57    @BeforeClass

59      register(ResourcesEndpoint.class);

62    @Test
63    public void getResourcesRefNoLongerExists() {
64      when(SERVICE_MOCK.readResource(any(UUID.class))).thenReturn(Optional.empty());
65      Response response = target(ROOTPATH).path(UUID.randomUUID().toString()).path("/ref").request().get();
66      assertThat(response.getStatus()).isEqualTo(Status.NOT_FOUND.getStatusCode());
67    }
68  
69    @Test
70    public void testGettingANonExistingResourceGivesANotFoundError() {
71      when(SERVICE_MOCK.readResource(any(UUID.class))).thenReturn(Optional.empty());
72      Response response = target(ROOTPATH).path("c28626d4-493a-4204-83d9-e9ae17e15654").request().get();
73      Log.debug("response={}", response);
74      assertThat(response.getStatus()).isEqualTo(Status.NOT_FOUND.getStatusCode());
75    }
76  
77    @Test
78    public void testPost() {
79      TentativeAntiochProvenance provenance = mock(TentativeAntiochProvenance.class);
80      UUID id = UUID.randomUUID();
81      AntiochResource resource = new AntiochResource(id, provenance);
82      when(SERVICE_MOCK.readResource(any(UUID.class))).thenReturn(Optional.of(resource));
83      Response response = target(ROOTPATH).request().post(jsonEntity("{'resource':{'ref':'REF'}}"));
84      Log.debug("response={}", response);
85      assertThat(response.getLocation().toString()).contains("/resources/" + id);
86      assertThat(response.getStatus()).isEqualTo(Status.CREATED.getStatusCode());
87    }
