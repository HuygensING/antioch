27  import javax.inject.Inject;
28  import javax.ws.rs.core.UriBuilder;
29  
30  import nl.knaw.huygens.antioch.api.EndpointPaths;
31  import nl.knaw.huygens.antioch.config.AntiochConfiguration;
32  import nl.knaw.huygens.antioch.model.Identifiable;
33  import nl.knaw.huygens.antioch.model.IdentifiablePointer;
34  
35  public class LocationBuilder {
36    private final AntiochConfiguration config;
37    private final EndpointPathResolver resolver;
38  
39    @Inject
40    public LocationBuilder(AntiochConfiguration config, EndpointPathResolver resolver) {
41      this.config = config;
42      this.resolver = resolver;
43    }
44  
45    public URI locationOf(Identifiable identifiable, String... subPaths) {
46      return locationOf(identifiable.getClass(), identifiable.getId(), subPaths);
47    }
48  
49    private URI locationOf(Class<? extends Identifiable> identifiableClass, UUID uuid, String... subPaths) {
50      return locationOf(identifiableClass, uuid.toString(), subPaths);
51    }
52  
53    public URI locationOf(IdentifiablePointer<? extends Identifiable> identifiablePointer, String... subPaths) {
54      return locationOf(identifiablePointer.getIdentifiableClass(), identifiablePointer.getIdentifier(), subPaths);
55    }
56  
57    public URI locationOf(Class<? extends Identifiable> identifiableClass, String uuid, String... subPaths) {
58      if (uuid.contains(".")) {
59        // Special case: uuid of deprecated annotation
60        String[] parts = uuid.split("\\.");
61        UriBuilder uriBuilder = UriBuilder.fromUri(config.getBaseURI()) //
62            .path(pathOf(identifiableClass)) //
63            .path("{uuid}") //
64            .path(EndpointPaths.REV) //
65            .path("{rev}");
66        for (String subPath : subPaths) {
67          uriBuilder = uriBuilder.path(subPath);
68        }
69        return uriBuilder //
70            .build(parts, true);
71      }
72  
73      UriBuilder uriBuilder = UriBuilder.fromUri(config.getBaseURI()) //
74          .path(pathOf(identifiableClass)) //
75          .path("{uuid}");
76      for (String subPath : subPaths) {
77        uriBuilder = uriBuilder.path(subPath);
78      }
79      return uriBuilder.build(uuid);
80    }
81  
82    public URI locationOf(Object... subPaths) {
83      UriBuilder uriBuilder = UriBuilder.fromUri(config.getBaseURI());
84      // Log.info("subPaths.size={}", subPaths.length);
85      for (Object subPath : subPaths) {
86        // Log.info("subPath=[{}]", subPath);
87        uriBuilder = uriBuilder.path(subPath.toString());
88      }
89      return uriBuilder.build("X");
90    }
91  
92    // -- private methods --//
93  
94    private String pathOf(Class<? extends Identifiable> identifiableClass, String... subPaths) {
95      return resolver.pathOf(identifiableClass).orElseThrow(unknownIdentifiableClass(identifiableClass));
96    }
97  
98    private Supplier<RuntimeException> unknownIdentifiableClass(Class<? extends Identifiable> identifiableClass) {
99      return () -> new RuntimeException("unknown Identifiable class " + identifiableClass);
100   }
101 }
