23  import java.net.URI;
24  
25  import javax.ws.rs.Produces;
26  import javax.ws.rs.core.MediaType;
27  import javax.ws.rs.core.Response;
28  import javax.ws.rs.core.Response.Status;
29  
30  //rfc4627: JSON text SHALL be encoded in Unicode. The default encoding is UTF-8.
31  @Produces(MediaType.APPLICATION_JSON)
32  public abstract class JSONEndpoint {
33    protected Response created(URI location) {
34      return Response.created(location).build();
35    }
36  
37    protected Response methodNotImplemented() {
38      return Response.status(Status.NOT_IMPLEMENTED).build();
39    }
40  
41    protected Response noContent() {
42      return Response.noContent().build();
43    }
44  
45    protected Response conflict() {
46      return Response.status(Status.CONFLICT).build();
47    }
48  
49    protected Response ok() {
50      return Response.ok().build();
51    }
52  
53    protected Response ok(Object entity) {
54      return Response.ok(entity).build();
55    }
56  
57    protected Response notImplemented() {
58      return Response.status(Status.NOT_IMPLEMENTED).build();
59    }
60  
61    protected Response notImplemented(Object entity) {
62      return Response.status(Status.NOT_IMPLEMENTED).entity(entity).build();
63    }
64  
65  }
