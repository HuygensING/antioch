71    public ObjectNode validateAndStore(ObjectNode annotationNode) {
72      try {
73        String json = new ObjectMapper().writeValueAsString(annotationNode);
74  
75        String resourceRef = extractResourceRef(annotationNode);
76        AntiochResource antiochResource = extractAntiochResource(resourceRef);
77        AntiochAnnotation annotation = createWebAnnotation(json, antiochResource);
78  
79        annotationNode.set("@id", new TextNode(webAnnotationURI(annotation.getId()).toString()));
80        return annotationNode;
81  
82      } catch (IOException | JsonLdError e) {
83        throw new BadRequestException(e.getMessage());
84      }
85    }

87    public WebAnnotation validateAndStore(WebAnnotationPrototype prototype) {
88      try {
89        String json = new ObjectMapper().writeValueAsString(prototype);
90        Map<String, Object> jsonObject = (Map<String, Object>) JsonUtils.fromString(json);
91  
92        String resourceRef = extractResourceRef(jsonObject);
93        AntiochResource antiochResource = extractAntiochResource(resourceRef);
94        AntiochAnnotation annotation = createWebAnnotation(json, antiochResource);
95  
96        jsonObject.put("@id", webAnnotationURI(annotation.getId()));
97        String json2 = new ObjectMapper().writeValueAsString(jsonObject);
98        return new WebAnnotation(annotation.getId())//
99                .setJson(json2)//
100               .setETag(String.valueOf(prototype.getModified().hashCode()));
101 
102     } catch (IOException | JsonLdError e) {
103       throw new BadRequestException(e.getMessage());
104     }
105   }

107   URI webAnnotationURI(UUID annotationUUID) {
108     return UriBuilder.fromUri(config.getBaseURI())//
109             .path(EndpointPaths.WEB_ANNOTATIONS)//
110             .path(annotationUUID.toString())//
111             .build();
112   }

114   private String extractResourceRef(Object annotationObject) throws JsonLdError {
115     try {
116       String json = new ObjectMapper().writeValueAsString(annotationObject);
117       Object jsonObject = JsonUtils.fromString(json);
118       JsonLdOptions options = new JsonLdOptions();
119 //      LOG.info("annotationObject={}", annotationObject);
120       Map<Object, Object> context = new HashMap<>();
121       Map<String, Object> compacted = JsonLdProcessor.compact(jsonObject, context, options);
122 //      LOG.info("compacted={}", compacted);
123       Map<String, Object> target = (Map<String, Object>) compacted.get(OA_HAS_TARGET_IRI);
124       String resourceRef = "";
125       if (target == null) {
126         LOG.error("target==null!, compacted={}", compacted); // TODO!
127       } else {
128         if (target.containsKey("@id")) {
129           resourceRef = (String) target.get("@id");
130         } else if (target.containsKey(OA_HAS_SOURCE_IRI)) {
131           Map<String, Object> source = (Map<String, Object>) target.get(OA_HAS_SOURCE_IRI);
132           resourceRef = (String) source.get("@id");
133         }
134       }
135       return resourceRef.replaceFirst("#.*", "");
136     } catch (IOException e) {
137       e.printStackTrace();
138       throw new RuntimeException(e);
139     }
140   }

142   private AntiochResource extractAntiochResource(String resourceRef) {
143     // first see if there's already a resource with this ref, if so, use this.
144     return cache.get(resourceRef).orElseGet(() -> {
145       AntiochResource resourceWithRef = service.readResourceWithUniqueRef(resourceRef).orElseGet(() -> {
146         UUID resourceUUID = UUID.randomUUID();
147         TentativeAntiochProvenance provenance = new TentativeAntiochProvenance(ThreadContext.getUserName(), Instant.now(), AntiochProvenance.DEFAULT_WHY);
148         return service.createResource(resourceUUID, resourceRef, provenance, AntiochState.CONFIRMED);
149       });
150       cache.add(resourceWithRef);
151       return resourceWithRef;
152     });
153   }
154 
155   private AntiochAnnotation createWebAnnotation(String json, AntiochResource antiochResource) {
156     UUID annotationBodyUUID = UUID.randomUUID();
157     // TODO: use information from prototype to create provenance
158     TentativeAntiochProvenance annotationProvenance = new TentativeAntiochProvenance(ThreadContext.getUserName(), Instant.now(), AntiochProvenance.DEFAULT_WHY);
159     AntiochAnnotationBody annotationBody = service.createAnnotationBody(annotationBodyUUID, WEBANNOTATION_TYPE, json, annotationProvenance);
160     AntiochAnnotation annotation = service.annotate(antiochResource, annotationBody, annotationProvenance);
161     service.confirmAnnotation(annotation.getId());
162     return annotation;
163   }
