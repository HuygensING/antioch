98    @POST
99    @Consumes(JSONLD_MEDIATYPE)
100   @Produces(JSONLD_MEDIATYPE)
101   public Response addWebAnnotation(//
102       @HeaderParam("Accept") String acceptHeader, //
103       WebAnnotationPrototype prototype//
104   ) {
105     String expectedProfile = extractProfile(acceptHeader);
106     prototype.setCreated(Instant.now().toString());
107     WebAnnotation webAnnotation = webAnnotationService.validateAndStore(prototype);
108     String profiledWebAnnotation = profile(webAnnotation.json(), expectedProfile);
109     return Response.created(webAnnotationService.webAnnotationURI(webAnnotation.getId()))//
110         .link(RESOURCE_TYPE_URI, "type")//
111         .link(ANNOTATION_TYPE_URI, "type")//
112         .tag(webAnnotation.eTag())//
113         .allow(ALLOWED_METHODS)//
114         .entity(profiledWebAnnotation)//
115         .build();
116   }

118   @GET
119   @Path("{uuid}")
120   @Produces(JSONLD_MEDIATYPE)
121   public Response getWebAnnotation(//
122       @HeaderParam("Accept") String acceptHeader, //
123       @PathParam("uuid") UUIDParam uuidParam //
124   ) {
125     String expectedProfile = extractProfile(acceptHeader);
126     AntiochAnnotation antiochAnnotation = service.readAnnotation(uuidParam.getValue()) //
127         .orElseThrow(annotationNotFoundForId(uuidParam));
128     if (antiochAnnotation.getState().equals(AntiochState.DELETED)) {
129       return Response.status(Status.GONE).build();
130     }
131 
132     WebAnnotation webAnnotation = asWebAnnotation(antiochAnnotation);
133     String profiledWebAnnotation = profile(webAnnotation.json(), expectedProfile);
134     return Response.ok(profiledWebAnnotation)//
135         .link(RESOURCE_TYPE_URI, "type")//
136         .link(ANNOTATION_TYPE_URI, "type")//
137         .tag(webAnnotation.eTag())//
138         .allow(ALLOWED_METHODS)//
139         .build();
140   }

142   @PUT
143   @Path("{uuid}")
144   @Consumes(JSONLD_MEDIATYPE)
145   @Produces(JSONLD_MEDIATYPE)
146   public Response updateWebAnnotation(//
147       @HeaderParam("Accept") String acceptHeader, //
148       @PathParam("uuid") UUIDParam uuidParam, //
149       WebAnnotationPrototype prototype//
150   ) {
151     // TODO: what if the resource changes?
152     String expectedProfile = extractProfile(acceptHeader);
153     Instant modificationInstant = Instant.now();
154     prototype.setModified(modificationInstant.toString());
155     UUID annotationUuid = uuidParam.getValue();
156     AntiochAnnotation antiochAnnotation = service.readAnnotation(annotationUuid) //
157         .orElseThrow(annotationNotFoundForId(uuidParam));
158     try {
159       String json = new ObjectMapper().writeValueAsString(prototype);
160       TentativeAntiochProvenance provenance = new TentativeAntiochProvenance(ThreadContext.getUserName(), modificationInstant, AntiochProvenance.DEFAULT_WHY);
161       AntiochAnnotationBody body = service.findAnnotationBodyWithTypeAndValue(WEBANNOTATION_TYPE, json)//
162           .orElseGet(() -> new AntiochAnnotationBody(UUID.randomUUID(), WEBANNOTATION_TYPE, json, provenance));
163       AntiochAnnotation newAnnotation = new AntiochAnnotation(annotationUuid, body, provenance);
164       antiochAnnotation = service.deprecateAnnotation(annotationUuid, newAnnotation);
165       WebAnnotation webAnnotation = asWebAnnotation(antiochAnnotation);
166       String profiledWebAnnotation = profile(webAnnotation.json(), expectedProfile);
167       return Response.ok(profiledWebAnnotation)//
168           .link(RESOURCE_TYPE_URI, "type")//
169           .link(ANNOTATION_TYPE_URI, "type")//
170           .tag(webAnnotation.eTag())//
171           .allow(ALLOWED_METHODS)//
172           .build();
173 
174     } catch (JsonProcessingException e) {
175       e.printStackTrace();
176       throw new RuntimeException(e);
177     }
178 
179   }

208   private String profile(String json, String expectedProfile) {
209     try {
210       Map<String, Object> jsonObject = (Map<String, Object>) JsonUtils.fromString(json);
211       JsonLdOptions options = new JsonLdOptions();
212       List<Object> expanded = JsonLdProcessor.expand(jsonObject, options);
213       Map<String, Object> profiled = JsonLdProcessor.compact(expanded, expectedProfile, options);
214       profiled.put("@context", expectedProfile);
215       return new ObjectMapper().writeValueAsString(profiled);
216 
217     } catch (IOException | JsonLdError e) {
218       throw new RuntimeException(e);
219     }
220   }

222   private List<Object> findWebAnnotationsAbout(String uri) {
223     AntiochQuery query = new AntiochQuery()//
224         .setPageSize(1000)//
225         .setFind("annotation")//
226         .setWhere("type:eq(\"" + WEBANNOTATION_TYPE + "\") resource.ref:eq(\"" + uri + "\")")//
227         .setReturns("id,value");
228 
229     SearchResult result = service.execute(query);
230     List<Object> webannotations = Lists.newArrayList();
231     result.getResults().forEach(resultMap -> {
232       String json = (String) resultMap.get("value");
233       UUID uuid = UUID.fromString((String) resultMap.get("id"));
234       try {
235         Map<String, Object> map = enrichJson(json);
236         map.put("@id", webAnnotationService.webAnnotationURI(uuid));
237         webannotations.add(map);
238       } catch (IOException e) {
239         throw new RuntimeException(e);
240       }
241     });
242     return webannotations;
243   }

245   private Map<String, Object> enrichJson(String json) throws IOException {
246     Map<String, Object> map = new ObjectMapper().readValue(json, Map.class);
247     return map;
248   }

250   private static final Pattern PROFILE_PATTERN = Pattern.compile(".*profile=\"(.*?)\".*");
251 
252   private String extractProfile(String accept) {
253     Matcher matcher = PROFILE_PATTERN.matcher(accept);
254     if (matcher.matches()) {
255       return matcher.group(1);
256     }
257     return DEFAULT_PROFILE;
258   }

260   private WebAnnotation asWebAnnotation(AntiochAnnotation antiochAnnotation) {
261     AntiochAnnotationBody body = antiochAnnotation.getBody();
262     String type = body.getType();
263     String value = body.getValue();
264     String json = "";
265     Instant when = antiochAnnotation.getProvenance().getWhen();
266     if (WEBANNOTATION_TYPE.equals(type)) {
267       json = addIdToValue(antiochAnnotation, value);
268 
269     } else {
270       Map<String, Object> webAnnotationMap = Maps.newHashMap();
271       webAnnotationMap.put("@context", "http://www.w3.org/ns/anno.jsonld");
272       webAnnotationMap.put("@id", locationBuilder.locationOf(antiochAnnotation));
273       webAnnotationMap.put("type", "Annotation");
274       webAnnotationMap.put("created", when.toString());
275       Map<String, String> bodyMap = ImmutableMap.of("type", "TextualBody", "value", type + ": " + value);
276       webAnnotationMap.put("body", bodyMap);
277       webAnnotationMap.put("target", locationBuilder.locationOf(antiochAnnotation.getAnnotatablePointer()));
278 
279       try {
280         json = new ObjectMapper().writeValueAsString(webAnnotationMap);
281       } catch (JsonProcessingException e) {
282         throw new RuntimeException(e);
283       }
284     }
285     return new WebAnnotation(antiochAnnotation.getId())//
286         .setJson(json)//
287         .setETag(String.valueOf(when.hashCode()));
288   }
289 
290   private String addIdToValue(AntiochAnnotation antiochAnnotation, String value) {
291     String json = value;
292     try {
293       Map<String, Object> webAnnotationMap = enrichJson(json);
294       webAnnotationMap.put("@id", webAnnotationService.webAnnotationURI(antiochAnnotation.getId()));
295       json = new ObjectMapper().writeValueAsString(webAnnotationMap);
296     } catch (IOException e) {
297       throw new RuntimeException(e);
298     }
299     return json;
300   }

302   private static Supplier<NotFoundException> annotationNotFoundForId(Object id) {
303     return () -> new NotFoundException(NoAnnotationFoundWithId(id));
304   }

306   private static String NoAnnotationFoundWithId(Object id) {
307     return "No annotation found with id " + id;
