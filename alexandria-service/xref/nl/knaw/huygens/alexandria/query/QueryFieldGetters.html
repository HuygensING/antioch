25  import java.util.Map;
26  import java.util.function.Function;
27  
28  import com.google.common.collect.ImmutableMap;
29  
30  import nl.knaw.huygens.alexandria.api.model.search.QueryField;
31  import nl.knaw.huygens.alexandria.storage.frames.AnnotationVF;
32  
33  public class QueryFieldGetters {
34    static final Map<QueryField, Function<AnnotationVF, Object>> getterMap = ImmutableMap.<QueryField, Function<AnnotationVF, Object>> builder()//
35        .put(QueryField.id, AlexandriaQueryParser::getAnnotationId) //
36        .put(QueryField.url, AlexandriaQueryParser::getAnnotationURL) //
37        .put(QueryField.when, AnnotationVF::getProvenanceWhen) //
38        .put(QueryField.who, AnnotationVF::getProvenanceWho) //
39        .put(QueryField.why, AnnotationVF::getProvenanceWhy) //
40        .put(QueryField.type, AnnotationVF::getType) //
41        .put(QueryField.value, AnnotationVF::getValue) //
42        .put(QueryField.state, AnnotationVF::getState) //
43        .put(QueryField.resource_id, AnnotationVF::getResourceId) //
44        .put(QueryField.subresource_id, AnnotationVF::getSubResourceId) //
45        .put(QueryField.resource_url, AlexandriaQueryParser::getResourceURL) //
46        .put(QueryField.subresource_url, AlexandriaQueryParser::getSubResourceURL)//
47        .put(QueryField.resource_ref, AlexandriaQueryParser::getResourceRef) //
48        .put(QueryField.subresource_sub, AlexandriaQueryParser::getSubResourceSub)//
49        .build();
50  
51    public static Function<AnnotationVF, Object> get(QueryField queryField) {
52      return getterMap.get(queryField);
53    }
54  }
