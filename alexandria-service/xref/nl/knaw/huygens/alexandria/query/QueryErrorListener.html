3   import java.util.BitSet;
4   import java.util.Collection;
5   
6   import org.antlr.v4.runtime.BaseErrorListener;
7   import org.antlr.v4.runtime.Parser;
8   import org.antlr.v4.runtime.RecognitionException;
9   import org.antlr.v4.runtime.Recognizer;
10  import org.antlr.v4.runtime.atn.ATNConfigSet;
11  import org.antlr.v4.runtime.dfa.DFA;
12  
13  import com.google.common.collect.Lists;
14  
15  import nl.knaw.huygens.Log;
16  
17  public class QueryErrorListener extends BaseErrorListener {
18    private Collection<String> parseErrors = Lists.newArrayList();
19  
20    @Override
21    public void reportAmbiguity(Parser recognizer, DFA dfa, int startIndex, int stopIndex, boolean exact, BitSet ambigAlts, ATNConfigSet configs) {
22      Log.warn("reportAmbiguity:");
23      Log.warn("recognizer={}", recognizer);
24      Log.warn("dfa={}", dfa);
25      Log.warn("startIndex={}", startIndex);
26      Log.warn("stopIndex={}", stopIndex);
27      Log.warn("exact={}", exact);
28      Log.warn("ambigAlts={}", ambigAlts);
29      Log.warn("configs={}", configs);
30      parseErrors.add("where: ambiguity at (" + startIndex + ".." + stopIndex + ")");
31    }
32  
33    @Override
34    public void reportContextSensitivity(Parser recognizer, DFA dfa, int startIndex, int stopIndex, int prediction, ATNConfigSet configs) {
35      Log.warn("reportContextSensitivity:");
36      Log.warn("recognizer={}", recognizer);
37      Log.warn("dfa={}", dfa);
38      Log.warn("startIndex={}", startIndex);
39      Log.warn("stopIndex={}", stopIndex);
40      Log.warn("prediction={}", prediction);
41      Log.warn("configs={}", configs);
42      parseErrors.add("where: contextSensitivity at (" + startIndex + ".." + stopIndex + ")");
43    }
44  
45    @Override
46    public void reportAttemptingFullContext(Parser recognizer, DFA dfa, int startIndex, int stopIndex, BitSet conflictingAlts, ATNConfigSet configs) {
47      Log.warn("reportAttemptingFullContext:");
48      Log.warn("recognizer={}", recognizer);
49      Log.warn("dfa={}", dfa);
50      Log.warn("startIndex={}", startIndex);
51      Log.warn("stopIndex={}", stopIndex);
52      Log.warn("conflictingAlts={}", conflictingAlts);
53      Log.warn("configs={}", configs);
54      parseErrors.add("where: attemptingFullContext at (" + startIndex + ".." + stopIndex + ")");
55    }
56  
57    @Override
58    public void syntaxError(Recognizer<?, ?> recognizer, Object offendingSymbol, int line, int charPositionInLine, String msg, RecognitionException e) {
59      Log.warn("syntaxError:");
60      Log.warn("recognizer={}", recognizer);
61      Log.warn("offendingSymbol={}", offendingSymbol);
62      Log.warn("line={}", line);
63      Log.warn("charPositionInLine={}", charPositionInLine);
64      Log.warn("msg={}", msg);
65      // Log.warn("e={}", e);
66      parseErrors.add("where: (" + line + ":" + charPositionInLine + ") " + msg);
67    }
68  
69    public Collection<String> getParseErrors() {
70      return parseErrors;
71    }
72  
73    public boolean heardErrors() {
74      return !parseErrors.isEmpty();
75    }
76  
77  }
