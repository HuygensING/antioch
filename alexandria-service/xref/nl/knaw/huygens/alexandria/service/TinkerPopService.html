82      AlexandriaResource resource;

86        resource = readResource(uuid).get();

95      createOrUpdateResource(resource);

125     Class<? extends Accountable> aClass = pointer.getIdentifiableClass();
126     UUID uuid = UUID.fromString(pointer.getIdentifier());
127     if (AlexandriaResource.class.equals(aClass)) {
128       return readResource(uuid);

131       return readAnnotation(uuid);

133     } else {
134       throw new RuntimeException("unexpected accountableClass: " + aClass.getName());
135     }

210   @Override
211   public AlexandriaAnnotation deprecateAnnotation(UUID annotationId, AlexandriaAnnotation updatedAnnotation) {
212     storage.startTransaction();
213 
214     // check if there's an annotation with the given id
215     AnnotationVF oldAnnotationVF = storage.readVF(AnnotationVF.class, annotationId)//
216         .orElseThrow(annotationNotFound(annotationId));
217     if (oldAnnotationVF.isTentative()) {
218       throw incorrectStateException(annotationId, "tentative");
219     } else if (oldAnnotationVF.isDeleted()) {
220       throw new BadRequestException("annotation " + annotationId + " is " + "deleted");
221     } else if (oldAnnotationVF.isDeprecated()) {
222       throw new BadRequestException("annotation " + annotationId + " is " + "already deprecated");
223     }
224 
225     AlexandriaAnnotationBody newBody = updatedAnnotation.getBody();
226     Optional<AlexandriaAnnotationBody> optionalBody = findAnnotationBodyWithTypeAndValue(newBody.getType(), newBody.getValue());
227     AlexandriaAnnotationBody body;
228     if (optionalBody.isPresent()) {
229       body = optionalBody.get();
230     } else {
231       AnnotationBodyVF annotationBodyVF = frameAnnotationBody(newBody);
232       updateState(annotationBodyVF, AlexandriaState.CONFIRMED);
233       body = newBody;
234     }
235 
236     AlexandriaProvenance tmpProvenance = updatedAnnotation.getProvenance();
237     TentativeAlexandriaProvenance provenance = new TentativeAlexandriaProvenance(tmpProvenance.getWho(), tmpProvenance.getWhen(), tmpProvenance.getWhy());
238     AlexandriaAnnotation newAnnotation = new AlexandriaAnnotation(updatedAnnotation.getId(), body, provenance);
239     AnnotationVF newAnnotationVF = frameAnnotation(newAnnotation);
240 
241     AnnotationVF annotatedAnnotation = oldAnnotationVF.getAnnotatedAnnotation();
242     if (annotatedAnnotation != null) {
243       newAnnotationVF.setAnnotatedAnnotation(annotatedAnnotation);
244     } else {
245       ResourceVF annotatedResource = oldAnnotationVF.getAnnotatedResource();
246       newAnnotationVF.setAnnotatedResource(annotatedResource);
247     }
248     newAnnotationVF.setDeprecatedAnnotation(oldAnnotationVF);
249     newAnnotationVF.setRevision(oldAnnotationVF.getRevision() + 1);
250     updateState(newAnnotationVF, AlexandriaState.CONFIRMED);
251 
252     oldAnnotationVF.setAnnotatedAnnotation(null);
253     oldAnnotationVF.setAnnotatedResource(null);
254     oldAnnotationVF.setUuid(oldAnnotationVF.getUuid() + "." + oldAnnotationVF.getRevision());
255     updateState(oldAnnotationVF, AlexandriaState.DEPRECATED);
256 
257     AlexandriaAnnotation resultAnnotation = deframeAnnotation(newAnnotationVF);
258     storage.commitTransaction();
259     return resultAnnotation;
260   }

285   @Override
286   public void deleteAnnotation(AlexandriaAnnotation annotation) {
287     storage.startTransaction();
288     UUID uuid = annotation.getId();
289     AnnotationVF annotationVF = storage.readVF(AnnotationVF.class, uuid).get();
290     if (annotation.isTentative()) {
291       // remove from database
292 
293       AnnotationBodyVF body = annotationVF.getBody();
294       List<AnnotationVF> ofAnnotations = body.getOfAnnotationList();
295       if (ofAnnotations.size() == 1) {
296         String annotationBodyId = body.getUuid();
297         storage.removeVertexWithId(annotationBodyId);
298       }
299 
300       // remove has_body edge
301       annotationVF.setBody(null);
302 
303       // remove annotates edge
304       annotationVF.setAnnotatedAnnotation(null);
305       annotationVF.setAnnotatedResource(null);
306 
307       String annotationId = uuid.toString();
308       storage.removeVertexWithId(annotationId);
309 
310     } else {
311       // set state
312       updateState(annotationVF, AlexandriaState.DELETED);
313     }
314 
315     storage.commitTransaction();
316   }

451   private AlexandriaResource deframeResource(ResourceVF rvf) {
452     TentativeAlexandriaProvenance provenance = deframeProvenance(rvf);
453     UUID uuid = getUUID(rvf);
454     AlexandriaResource resource = new AlexandriaResource(uuid, provenance);
455     resource.setCargo(rvf.getCargo());
456     resource.setState(AlexandriaState.valueOf(rvf.getState()));
457     resource.setStateSince(Instant.ofEpochSecond(rvf.getStateSince()));
458     for (AnnotationVF annotationVF : rvf.getAnnotatedBy()) {
459       AlexandriaAnnotation annotation = deframeAnnotation(annotationVF);
460       resource.addAnnotation(annotation);
461     }
462     ResourceVF parentResource = rvf.getParentResource();
463     if (parentResource != null) {
464       resource.setParentResourcePointer(new IdentifiablePointer<>(AlexandriaResource.class, parentResource.getUuid()));
465     }
466     rvf.getSubResources().stream()//
467         .forEach(vf -> resource.addSubResourcePointer(new IdentifiablePointer<>(AlexandriaResource.class, vf.getUuid())));
468     return resource;
469   }

471   private AlexandriaAnnotation deframeAnnotation(AnnotationVF annotationVF) {
472     TentativeAlexandriaProvenance provenance = deframeProvenance(annotationVF);
473     UUID uuid = getUUID(annotationVF);
474     AlexandriaAnnotationBody body = deframeAnnotationBody(annotationVF.getBody());
475     AlexandriaAnnotation annotation = new AlexandriaAnnotation(uuid, body, provenance);
476     annotation.setState(AlexandriaState.valueOf(annotationVF.getState()));
477     annotation.setStateSince(Instant.ofEpochSecond(annotationVF.getStateSince()));
478     if (annotationVF.getRevision() == null) { // update old data
479       annotationVF.setRevision(0);
480     }
481     annotation.setRevision(annotationVF.getRevision());
482 
483     AnnotationVF annotatedAnnotation = annotationVF.getAnnotatedAnnotation();
484     if (annotatedAnnotation == null) {
485       ResourceVF annotatedResource = annotationVF.getAnnotatedResource();
486       if (annotatedResource != null) {
487         annotation.setAnnotatablePointer(new IdentifiablePointer<>(AlexandriaResource.class, annotatedResource.getUuid()));
488       }
489     } else {
490       annotation.setAnnotatablePointer(new IdentifiablePointer<>(AlexandriaAnnotation.class, annotatedAnnotation.getUuid()));
491     }
492     for (AnnotationVF avf : annotationVF.getAnnotatedBy()) {
493       AlexandriaAnnotation annotationAnnotation = deframeAnnotation(avf);
494       annotation.addAnnotation(annotationAnnotation);
495     }
496     return annotation;
497   }

586   @SuppressWarnings("unchecked")
587   private void findAllConfirmedAnnotationsRelatedToResource(String uuid) {
588     // case: find all annotations related to a given resource (or its subresources)
589     // start with the resource
590     // from there: find subresources, add them to resource list
591     // foreach resource in the list, get the confirmed annotations of that resource, add them to annotations list
592     // foreach annotation in the annotationlist, find the confirmed annotations of that annotation, add them to annotations list
593     GraphTraversal<Vertex, Vertex> traversal = storage.getVertexTraversal();
594     traversal.has("uuid", uuid).in(ResourceVF.PART_OF).in("annotates").has("state", "CONFIRMED").as("a").out("has_body").as("b").toList();
595     traversal.has("uuid", uuid)
596         .union(//
597             __.in(ResourceVF.PART_OF).in(AnnotationVF.ANNOTATES_RESOURCE), //
598             __.in(AnnotationVF.ANNOTATES_RESOURCE)//
599     ).has("state", AlexandriaState.CONFIRMED.name()).as("a").out(AnnotationVF.HAS_BODY).as("b").toList();
600   }

602   private List<Map<String, Object>> dummyResults() {
603     List<Map<String, Object>> list = Lists.newArrayList();
604     int num = new Random().nextInt(100);
605     for (int i = 0; i < num; i++) {
606       String displayName = "Annotation " + i;
