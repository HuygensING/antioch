25  import java.nio.file.Files;
26  import java.nio.file.Paths;
27  
28  import javax.inject.Inject;
29  import javax.inject.Singleton;
30  
31  import nl.knaw.huygens.Log;
32  import nl.knaw.huygens.alexandria.config.AlexandriaConfiguration;
33  import nl.knaw.huygens.alexandria.endpoint.LocationBuilder;
34  import nl.knaw.huygens.alexandria.text.TextService;
35  
36  @Singleton
37  public class PersistentTinkerGraphService extends TinkerGraphService {
38    private static final String DUMPFILE = "alexandria.graphml";
39  
40    @Inject
41    public PersistentTinkerGraphService(AlexandriaConfiguration config, LocationBuilder locationBuilder, TextService textService) {
42      super(locationBuilder, textService);
43      String dumpfile = config.getStorageDirectory() + "/" + DUMPFILE;
44      STORAGE.setDumpFile(dumpfile);
45      if (!STORAGE.supportsPersistence() && Files.exists(Paths.get(dumpfile))) {
46        Log.info("reading stored db from {}", dumpfile);
47        STORAGE.loadFromDisk(dumpfile);
48      }
49    }
50  }
