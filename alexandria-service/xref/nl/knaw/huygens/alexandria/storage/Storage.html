92    public <T extends Object> T runInTransaction(Supplier<T> supplier) {
93      boolean inOpenTransaction = transactionOpen;
94      if (!inOpenTransaction) {
95        startTransaction();
96      }
97      try {
98        T result = supplier.get();
99        if (!inOpenTransaction) {
100         commitTransaction();
101       }
102       return result;
103 
104     } catch (Exception e) {
105       e.printStackTrace();
106       rollbackTransaction();
107       throw e;
108     }
109   }
110 
111   public void runInTransaction(Runnable runner) {
112     boolean inOpenTransaction = transactionOpen;
113     if (!inOpenTransaction) {
114       startTransaction();
115     }
116     try {
117       runner.run();
118       if (!inOpenTransaction) {
119         commitTransaction();
120       }
121 
122     } catch (Exception e) {
123       e.printStackTrace();
124       rollbackTransaction();
125       throw e;
126     }
127   }

180   public Map<String, Object> getMetadata() {
181     assertInTransaction();
182     Map<String, Object> metadata = Maps.newLinkedHashMap();
183     metadata.put("features", graph.features().toString().split(System.lineSeparator()));
184     GraphTraversalSource traversal = graph.traversal();
185     metadata.put("vertices", count(traversal.V()));
186     metadata.put("edges", count(traversal.E()));
187     return metadata;
188   }

262   private void startTransaction() {
263     assertTransactionIsClosed();
264     if (supportsTransactions()) {
265       framedGraph.tx().rollback();
266     }
267     transactionOpen = true;
268   }
269 
270   private void commitTransaction() {
271     assertTransactionIsOpen();
272     if (supportsTransactions()) {
273       framedGraph.tx().commit();
274       // framedGraph.tx().close();
275     }
276     if (!supportsPersistence()) {
277       saveToDisk(getDumpFile());
278     }
279     transactionOpen = false;
280   }
281 
282   private void rollbackTransaction() {
283     assertTransactionIsOpen();
284     if (supportsTransactions()) {
285       framedGraph.tx().rollback();
286       // framedGraph.tx().close();
287     } else {
288       Log.error("rollback called, but transactions are not supported by graph {}", graph);
289     }
290     transactionOpen = false;
291   }
292 
293   private void assertInTransaction() {
294     Preconditions.checkState(transactionOpen, "We should be in open transaction at this point, use runInTransaction()!");
295   }
296 
297   private void assertClass(final Class<? extends AlexandriaVF> clazz) {
298     Preconditions.checkState(//
299         clazz.getAnnotationsByType(peapod.annotations.Vertex.class).length > 0, //
300         "Class " + clazz + " has no peapod @Vertex annotation, are you sure it's the correct class?"//
301     );
302   }
303 
304   private void assertTransactionIsClosed() {
305     Preconditions.checkState(!transactionOpen, "We're already inside an open transaction!");
306   }
307 
308   private void assertTransactionIsOpen() {
309     Preconditions.checkState(transactionOpen, "We're not in an open transaction!");
310   }
