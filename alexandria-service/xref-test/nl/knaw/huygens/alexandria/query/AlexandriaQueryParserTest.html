92    public void testUserStory4a() {
93      String userId = "USERID";
94      AlexandriaQuery aQuery = new AlexandriaQuery();
95      aQuery.setFind("annotation");
96      aQuery.setWhere(//
97          "type:eq(\"Tag\")"//
98              + " who:eq(\"" + userId + "\")"//
99              + " state:eq(\"CONFIRMED\")"//
100     );
101     aQuery.setSort("when");
102     aQuery.setFields("when,value,resource.id,subresource.id");
103     ParsedAlexandriaQuery paq = alexandriaQueryParser.parse(aQuery);
104 
105     AnnotationVF passAnnotation = mock(AnnotationVF.class);
106     AnnotationBodyVF passBody = mock(AnnotationBodyVF.class);
107     when(passBody.getType()).thenReturn("Tag");
108     // .put("resource.id", AnnotationVF::getResourceId)//
109     // .put("subresource.id", AnnotationVF::getSubResourceId)//
110 
111     when(passAnnotation.getUuid()).thenReturn("uuid");
112     when(passAnnotation.getBody()).thenReturn(passBody);
113     when(passAnnotation.getType()).thenReturn("Tag");
114     when(passAnnotation.getValue()).thenReturn("Value");
115     when(passAnnotation.getProvenanceWhen()).thenReturn("1");
116     when(passAnnotation.getProvenanceWho()).thenReturn(userId);
117     when(passAnnotation.getProvenanceWhy()).thenReturn("because");
118     when(passAnnotation.getResourceId()).thenReturn("resourceId");
119     when(passAnnotation.getSubResourceId()).thenReturn("subresourceId");
120     when(passAnnotation.getState()).thenReturn("CONFIRMED");
121     when(passAnnotation.isConfirmed()).thenReturn(true);
122 
123     AnnotationVF failAnnotation = mock(AnnotationVF.class);
124     when(failAnnotation.getBody()).thenReturn(passBody);
125     when(failAnnotation.getProvenanceWhen()).thenReturn("2");
126     when(failAnnotation.getProvenanceWho()).thenReturn(userId);
127     when(failAnnotation.getState()).thenReturn("TENTATIVE");
128     when(failAnnotation.isConfirmed()).thenReturn(false);
129 
130     // find: test VFClass
131     assertThat(paq.getVFClass()).isEqualTo(AnnotationVF.class);
132 
133     // test predicate
134     Predicate<Traverser<AnnotationVF>> predicate = paq.getPredicate();
135     assertThat(predicate).isNotNull();
136     Log.info("predicate={}", predicate);
137 
138     Traverser<AnnotationVF> pass = mock(Traverser.class);
139     when(pass.get()).thenReturn(passAnnotation);
140     assertThat(predicate.test(pass)).isTrue();
141 
142     Traverser<AnnotationVF> fail = mock(Traverser.class);
143     when(fail.get()).thenReturn(failAnnotation);
144     // assertThat(predicate.test(fail)).isFalse();
145 
146     // sort: test resultComparator
147     Comparator<AnnotationVF> resultComparator = paq.getResultComparator();
148     int compare1 = resultComparator.compare(passAnnotation, failAnnotation);
149     assertThat(compare1).isEqualTo(-1);
150     int compare2 = resultComparator.compare(failAnnotation, passAnnotation);
151     assertThat(compare2).isEqualTo(1);
152     int compare3 = resultComparator.compare(passAnnotation, passAnnotation);
153     assertThat(compare3).isEqualTo(0);
154 
155     // return: test returnFields + resultMapper
156     assertThat(paq.getReturnFields()).containsExactly("when", "value", "resource.id", "subresource.id");
157     Function<AnnotationVF, Map<String, Object>> resultMapper = paq.getResultMapper();
158     Map<String, Object> resultMap = resultMapper.apply(passAnnotation);
159     assertThat(resultMap).contains( //
160         MapEntry.entry("when", "1"), //
161         MapEntry.entry("value", "Value"), //
162         MapEntry.entry("resource.id", "resourceId"), //
163         MapEntry.entry("subresource.id", "subresourceId") //
164     );
165   }

188   @Test
189   public void testWhereTokenization() {
190     String where = "type:eq(\"Tag\")"//
191         + " who:eq(\"nederlab\")"//
192         + " state:eq(\"CONFIRMED\")"//
193         + " resource.id:inSet(\"11111-111-111-11-111\",\"11111-111-111-11-112\")";
194     List<WhereToken> tokens = alexandriaQueryParser.tokenize(where);
195     Log.info("errors:{}", alexandriaQueryParser.parseErrors);
196     assertThat(tokens).hasSize(4);
197 
198     WhereToken typeToken = tokens.get(0);
199     assertThat(typeToken.getProperty()).isEqualTo("type");
200     assertThat(typeToken.getFunction()).isEqualTo(MatchFunction.eq);
201     assertThat(typeToken.getParameters()).containsExactly("Tag");
202 
203     WhereToken whoToken = tokens.get(1);
204     assertThat(whoToken.getProperty()).isEqualTo("who");
205     assertThat(whoToken.getFunction()).isEqualTo(MatchFunction.eq);
206     assertThat(whoToken.getParameters()).containsExactly("nederlab");
207 
208     WhereToken stateToken = tokens.get(2);
209     assertThat(stateToken.getProperty()).isEqualTo("state");
210     assertThat(stateToken.getFunction()).isEqualTo(MatchFunction.eq);
211     assertThat(stateToken.getParameters()).containsExactly("CONFIRMED");
212 
213     WhereToken resourceToken = tokens.get(3);
214     assertThat(resourceToken.getProperty()).isEqualTo("resource.id");
215     assertThat(resourceToken.getFunction()).isEqualTo(MatchFunction.inSet);
216     assertThat(resourceToken.getParameters()).containsExactly("11111-111-111-11-111", "11111-111-111-11-112");
217   }
