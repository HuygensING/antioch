111   @Test
112   public void testWhereTokenization() {
113     String where = "type:eq(\"Tag\")"//
114         + " who:eq(\"nederlab\")"//
115         + " state:eq(\"CONFIRMED\")"//
116         + " resource.id:inSet(\"11111-111-111-11-111\",\"11111-111-111-11-112\")";
117     List<WhereToken> tokens = alexandriaQueryParser.tokenize(where);
118     Log.info("errors:{}", alexandriaQueryParser.parseErrors);
119     assertThat(tokens).hasSize(4);
120 
121     WhereToken typeToken = tokens.get(0);
122     assertThat(typeToken.getProperty()).isEqualTo(QueryField.type);
123     assertThat(typeToken.getFunction()).isEqualTo(QueryFunction.eq);
124     assertThat(typeToken.getParameters()).containsExactly("Tag");
125 
126     WhereToken whoToken = tokens.get(1);
127     assertThat(whoToken.getProperty()).isEqualTo(QueryField.who);
128     assertThat(whoToken.getFunction()).isEqualTo(QueryFunction.eq);
129     assertThat(whoToken.getParameters()).containsExactly("nederlab");
130 
131     WhereToken stateToken = tokens.get(2);
132     assertThat(stateToken.getProperty()).isEqualTo(QueryField.state);
133     assertThat(stateToken.getFunction()).isEqualTo(QueryFunction.eq);
134     assertThat(stateToken.getParameters()).containsExactly("CONFIRMED");
135 
136     WhereToken resourceToken = tokens.get(3);
137     assertThat(resourceToken.getProperty()).isEqualTo(QueryField.resource_id);
138     assertThat(resourceToken.getFunction()).isEqualTo(QueryFunction.inSet);
139     assertThat(resourceToken.getParameters()).containsExactly("11111-111-111-11-111", "11111-111-111-11-112");
140   }

241   public void testUserStory4a() {
242     String userId = "USERID";
243     AlexandriaQuery aQuery = new AlexandriaQuery();
244     aQuery.setFind("annotation");
245     aQuery.setWhere(//
246         "type:eq(\"Tag\")"//
247             + " who:eq(\"" + userId + "\")"//
248             + " state:eq(\"CONFIRMED\")"//
249     );
250     aQuery.setSort("when");
251     aQuery.setFields("when,value,resource.id,subresource.id");
252     ParsedAlexandriaQuery paq = alexandriaQueryParser.parse(aQuery);
253 
254     AnnotationVF passAnnotation = mock(AnnotationVF.class);
255     AnnotationBodyVF passBody = mock(AnnotationBodyVF.class);
256     when(passBody.getType()).thenReturn("Tag");
257     // .put(QueryField.resource_id, AnnotationVF::getResourceId)//
258     // .put("subresource.id", AnnotationVF::getSubResourceId)//
259 
260     when(passAnnotation.getUuid()).thenReturn("uuid");
261     when(passAnnotation.getBody()).thenReturn(passBody);
262     when(passAnnotation.getType()).thenReturn("Tag");
263     when(passAnnotation.getValue()).thenReturn("Value");
264     when(passAnnotation.getProvenanceWhen()).thenReturn("1");
265     when(passAnnotation.getProvenanceWho()).thenReturn(userId);
266     when(passAnnotation.getProvenanceWhy()).thenReturn("because");
267     when(passAnnotation.getResourceId()).thenReturn("resourceId");
268     when(passAnnotation.getSubResourceId()).thenReturn("subresourceId");
269     when(passAnnotation.getState()).thenReturn("CONFIRMED");
270 
271     AnnotationVF failAnnotation = mock(AnnotationVF.class);
272     when(failAnnotation.getBody()).thenReturn(passBody);
273     when(failAnnotation.getProvenanceWhen()).thenReturn("2");
274     when(failAnnotation.getProvenanceWho()).thenReturn(userId);
275     when(failAnnotation.getState()).thenReturn("TENTATIVE");
276     when(failAnnotation.getType()).thenReturn("Tag");
277 
278     // find: test VFClass
279     assertThat(paq.getVFClass()).isEqualTo(AnnotationVF.class);
280 
281     // test predicate
282     Predicate<AnnotationVF> predicate = paq.getPredicate();
283     Log.info("predicate={}", predicate);
284     assertThat(predicate).isNotNull();
285     assertThat(predicate.test(passAnnotation)).isTrue();
286     assertThat(predicate.test(failAnnotation)).isFalse();
287 
288     // sort: test resultComparator
289     Comparator<AnnotationVF> resultComparator = paq.getResultComparator();
290     int compare1 = resultComparator.compare(passAnnotation, failAnnotation);
291     assertThat(compare1).isEqualTo(-1);
292     int compare2 = resultComparator.compare(failAnnotation, passAnnotation);
293     assertThat(compare2).isEqualTo(1);
294     int compare3 = resultComparator.compare(passAnnotation, passAnnotation);
295     assertThat(compare3).isEqualTo(0);
296 
297     // return: test returnFields + resultMapper
298     assertThat(paq.getReturnFields()).containsExactly("when", "value", "resource.id", "subresource.id");
299     Function<AnnotationVF, Map<String, Object>> resultMapper = paq.getResultMapper();
300     Map<String, Object> resultMap = resultMapper.apply(passAnnotation);
301     assertThat(resultMap).contains( //
302         MapEntry.entry("when", "1"), //
303         MapEntry.entry("value", "Value"), //
304         MapEntry.entry("resource.id", "resourceId"), //
305         MapEntry.entry("subresource.id", "subresourceId") //
306     );
