71    private static Module alexandriaModule() {
72      return new AbstractModule() {
73        @Override
74        protected void configure() {
75          Log.trace("setting up Guice bindings");
76          bind(Storage.class).to(TinkerGraphStorage.class);
77          bind(AlexandriaService.class).toInstance(service);
78          bind(AlexandriaConfiguration.class).toInstance(CONFIG);
79          bind(AnnotationEntityBuilder.class).in(Scopes.SINGLETON);
80          bind(EndpointPathResolver.class).in(Scopes.SINGLETON);
81          bind(ResourceEntityBuilder.class).in(Scopes.SINGLETON);
82        }
83      };
84    }
