1   package nl.knaw.huygens.alexandria.concordion;
2   
3   import javax.ws.rs.ext.ContextResolver;
4   import javax.ws.rs.ext.Provider;
5   
6   import com.fasterxml.jackson.annotation.JsonAutoDetect.Visibility;
7   import com.fasterxml.jackson.annotation.PropertyAccessor;
8   import com.fasterxml.jackson.databind.ObjectMapper;
9   import com.fasterxml.jackson.databind.SerializationFeature;
10  import nl.knaw.huygens.Log;
11  
12  @Provider
13  public class JsonConfiguration implements ContextResolver<ObjectMapper> {
14  
15    private final ObjectMapper defaultObjectMapper;
16  
17    public JsonConfiguration() {
18      defaultObjectMapper = createDefaultMapper();
19    }
20  
21    @Override
22    public ObjectMapper getContext(final Class<?> type) {
23      Log.trace("Returning Jackson ObjectMapper for type: {}", type);
24      return defaultObjectMapper;
25    }
26  
27    private static ObjectMapper createDefaultMapper() {
28      final ObjectMapper mapper = new ObjectMapper();
29      Log.debug("Configuring Jackson ObjectMapper: [" + mapper + "]");
30  
31      mapper.enable(SerializationFeature.INDENT_OUTPUT);
32      mapper.disable(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS);
33      // mapper.disable(SerializationFeature.WRITE_DURATIONS_AS_TIMESTAMPS);
34  
35      mapper.setVisibility(PropertyAccessor.FIELD, Visibility.ANY);
36  
37      return mapper;
38    }
39  }
