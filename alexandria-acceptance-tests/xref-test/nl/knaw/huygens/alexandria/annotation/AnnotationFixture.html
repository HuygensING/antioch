57    public void setUpAnnotation(String id, String tag) {
58      List<String> tags = annotatedReferences.get(id);
59  
60      if (tags == null) {
61        tags = Lists.newArrayList();
62        annotatedReferences.put(id, tags);
63      }
64  
65      tags.add(tag);
66    }
67  
68    public String addAnnotation(String id, String tag) {
69      if (Strings.isNullOrEmpty(id) || Strings.isNullOrEmpty(tag)) {
70        return "400 Bad Request";
71      }
72  
73      List<String> tags = annotatedReferences.get(id);
74      if (tags == null) {
75        tags = Lists.newArrayList();
76        annotatedReferences.put(id, tags);
77      }
78  
79      if (tags.contains(tag)) {
80        return "409 Conflict";
81      }
82  
83      tags.add(tag);
84      return "201 Created";
85    }
86  
87    public MultiValueResult getAnnotation(String id) {
88      if (Strings.isNullOrEmpty(id)) {
89        return multiValueResult().with("status", "400 Bad Request");
90      }
91  
92      if (annotatedReferences.containsKey(id)) {
93        return multiValueResult().with("status", "200 Ok").with("value", annotatedReferences.get(id));
94      }
95  
96      return multiValueResult().with("status", "404 Not Found");
97    }
