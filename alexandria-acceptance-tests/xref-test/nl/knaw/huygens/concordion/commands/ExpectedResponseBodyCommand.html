3   import java.util.Optional;

5   import nl.knaw.huygens.concordion.HuygensCommand;
6   import org.concordion.api.CommandCall;
7   import org.concordion.api.Element;
8   import org.concordion.api.Evaluator;
9   import org.concordion.api.ResultRecorder;
10  import org.concordion.internal.listener.AssertResultRenderer;
11  
12  @HuygensCommand(name = "responseBody")
13  public class ExpectedResponseBodyCommand extends AbstractHuygensCommand {
14  
15    public ExpectedResponseBodyCommand() {
16      addListener(new AssertResultRenderer());
17    }
18  
19    @Override
20    public void verify(CommandCall commandCall, Evaluator evaluator, ResultRecorder resultRecorder) {
21      final Element element = commandCall.getElement();
22  
23      final String expectedBody = element.getText();
24      final Optional<String> actualBody = getFixture(evaluator).response();
25  
26      if (expectedBody.isEmpty()) {
27        element.addStyleClass("empty");
28        if (actualBody.isPresent()) {
29          fail(resultRecorder, element, actualBody.get(), "(not set)");
30        } else {
31          succeed(resultRecorder, element);
32        }
33      } else {
34        if (actualBody.isPresent()) {
35          succeed(resultRecorder, element);
36        } else {
37          fail(resultRecorder, element, "(not set)", expectedBody);
38        }
39      }
40    }
41  }
