12  import org.concordion.api.CommandCall;
13  import org.concordion.api.Element;
14  import org.concordion.api.Evaluator;
15  import org.concordion.api.ResultRecorder;
16  import org.concordion.internal.listener.AssertResultRenderer;
17  
18  @HuygensCommand(name = "location")
19  public class ExpectedLocationCommand extends AbstractHuygensCommand {
20  
21    public ExpectedLocationCommand() {
22      addListener(new AssertResultRenderer());
23    }
24  
25    @Override
26    public void verify(CommandCall commandCall, Evaluator evaluator, ResultRecorder resultRecorder) {
27      final Element element = commandCall.getElement();
28      element.addStyleClass("location");
29  
30      final String expectedLocation = element.getText();
31      final String type = typeFrom(element);
32  
33      final RestFixture fixture = getFixture(evaluator);
34      final String actualLocation = fixture.location().map(l -> extract(l, type)).orElse("(not set)");
35  
36      if (actualLocation.equals(expectedLocation)) {
37        succeed(resultRecorder, element);
38      } else {
39        fail(resultRecorder, element, actualLocation, expectedLocation);
40      }
41    }
42  
43    private String extract(String location, String type) {
44      switch (type) {
45        case "base":
46          return baseOf(location);
47        case "full":
48          return location;
49        case "uuid":
50          return uuidQuality(location);
51        default:
52          throw new IllegalArgumentException("Illegal type: " + type);
53      }
54    }
55  
56    private String typeFrom(Element element) {
57      String type = element.getAttributeValue("type");
58      return Optional.ofNullable(type).orElse("full");
59    }
60  
61    private String baseOf(String s) {
62      return s.substring(0, s.lastIndexOf('/') + 1);
63    }
64  
65    private String uuidQuality(String location) {
66      final String idStr = tailOf(location);
67      return parse(idStr).map(uuid -> "well-formed UUID").orElseGet(malformedDescription(idStr));
68    }
69  
70    private String tailOf(String s) {
71      return Iterables.getLast(Splitter.on('/').split(s));
72    }
73  
74    private Optional<UUID> parse(String idStr) {
75      return UUIDParser.fromString(idStr).get();
76    }
77  
78    private Supplier<String> malformedDescription(String idStr) {
79      return () -> "malformed UUID: " + idStr;
80    }
81  
82  }
