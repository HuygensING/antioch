1   package nl.knaw.huygens.concordion.commands;
2   
3   import nl.knaw.huygens.concordion.FixtureEvaluator;
4   import nl.knaw.huygens.concordion.RestFixture;
5   import org.concordion.api.AbstractCommand;
6   import org.concordion.api.Element;
7   import org.concordion.api.Evaluator;
8   import org.concordion.api.Result;
9   import org.concordion.api.ResultRecorder;
10  import org.concordion.api.listener.AssertEqualsListener;
11  import org.concordion.api.listener.AssertFailureEvent;
12  import org.concordion.api.listener.AssertSuccessEvent;
13  import org.concordion.internal.util.Announcer;
14  
15  public abstract class AbstractHuygensCommand extends AbstractCommand {
16    private final Announcer<AssertEqualsListener> listeners = Announcer.to(AssertEqualsListener.class);
17  
18    protected void addListener(AssertEqualsListener listener) {
19      listeners.addListener(listener);
20    }
21  
22    protected RestFixture getFixture(Evaluator evaluator) {
23      return ((FixtureEvaluator) evaluator).getFixture();
24    }
25  
26    protected void succeed(ResultRecorder resultRecorder, Element element) {
27      resultRecorder.record(Result.SUCCESS);
28      announce().successReported(new AssertSuccessEvent(element));
29    }
30  
31    protected void fail(ResultRecorder resultRecorder, Element element, String actual, String expected) {
32      resultRecorder.record(Result.FAILURE);
33      announce().failureReported(new AssertFailureEvent(element, expected, actual));
34    }
35  
36    private AssertEqualsListener announce() {
37      return listeners.announce();
38    }
39  }
