9   import org.concordion.internal.listener.AssertResultRenderer;
10  
11  @HuygensCommand(name = "success", htmlTag = "span")
12  public class ExpectedSuccessCommand extends AbstractHuygensCommand {
13  
14    public ExpectedSuccessCommand() {
15      addListener(new AssertResultRenderer());
16    }
17  
18    @Override
19    public void verify(CommandCall commandCall, Evaluator evaluator, ResultRecorder resultRecorder) {
20      final Element element = commandCall.getElement();
21      final String expectedStatus = element.getText();
22  
23      final RestFixture fixture = getFixture(evaluator);
24      final String actualStatus = fixture.status();
25  
26      if (fixture.wasSuccessful()) {
27        succeed(resultRecorder, element);
28      } else {
29        fail(resultRecorder, element, actualStatus, expectedStatus);
30      }
31    }
32  }
