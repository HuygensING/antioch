17  public class AQLTest {
18  
19    @Test
20    public void testCorrectAQLStatement() {
21      String statement = "type:eq(\"Tag\")"//
22          + " who:eq(\"nedlab\")" //
23          + " state:eq(\"CONFIRMED\")"//
24          + " resource.id:inSet(1,2)"//
25          + " value:match(\"whatever\")";
26      CharStream stream = new ANTLRInputStream(statement);
27      AQLLexer lex = new AQLLexer(stream);
28      List<? extends Token> allTokens = lex.getAllTokens();
29      for (Token token : allTokens) {
30        Log.info("token: [{}] <<{}>>", lex.getRuleNames()[token.getType() - 1], token.getText());
31      }
32      lex.reset();
33  
34      CommonTokenStream tokens = new CommonTokenStream(lex);
35      AQLParser parser = new AQLParser(tokens);
36      parser.setBuildParseTree(true);
37      ParseTree tree = parser.root();
38      Log.info("tree={}", tree.toStringTree(parser));
39      assertThat(tree.getChildCount()).isEqualTo(5); // 5 subqueries
40  
41      int numberOfSyntaxErrors = parser.getNumberOfSyntaxErrors();
42      assertThat(numberOfSyntaxErrors).isEqualTo(0); // no syntax errors
43      Log.info("numberOfSyntaxErrors={}", numberOfSyntaxErrors);
44  
45      for (int i = 0; i < tree.getChildCount(); i++) {
46        Log.info("root.child={}", tree.getChild(i).getText());
47      }
48      assertThat(allTokens).hasSize(32);
49    }
