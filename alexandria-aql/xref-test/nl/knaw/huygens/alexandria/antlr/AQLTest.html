19    @Test
20    public void testCorrectAQLStatement() {
21      String statement = "type:eq(\"Tag\")"//
22          + " who:eq(\"nedlab\")" //
23          + " state:eq(\"CONFIRMED\")"//
24          + " resource.id:inSet(1,2)"//
25          + " value:match(\"what.*\")"//
26          + " when:inRange(\"2015-01-01T00:00:00Z\",\"2016-01-01T00:00:00Z\")"//
27          ;
28      CharStream stream = new ANTLRInputStream(statement);
29      AQLLexer lex = new AQLLexer(stream);
30      List<? extends Token> allTokens = lex.getAllTokens();
31      for (Token token : allTokens) {
32        Log.info("token: [{}] <<{}>>", lex.getRuleNames()[token.getType() - 1], token.getText());
33      }
34      lex.reset();
35  
36      CommonTokenStream tokens = new CommonTokenStream(lex);
37      AQLParser parser = new AQLParser(tokens);
38      parser.setBuildParseTree(true);
39      ParseTree tree = parser.root();
40      Log.info("tree={}", tree.toStringTree(parser));
41      assertThat(tree.getChildCount()).isEqualTo(6); // 6 subqueries
42  
43      int numberOfSyntaxErrors = parser.getNumberOfSyntaxErrors();
44      assertThat(numberOfSyntaxErrors).isEqualTo(0); // no syntax errors
45      Log.info("numberOfSyntaxErrors={}", numberOfSyntaxErrors);
46  
47      for (int i = 0; i < tree.getChildCount(); i++) {
48        Log.info("root.child={}", tree.getChild(i).getText());
49      }
50      assertThat(allTokens).hasSize(40);
51    }

64      }
65      lex.reset();
66
