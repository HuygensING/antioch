25  import static org.assertj.core.api.Assertions.assertThat;

27  import java.util.List;

29  import org.antlr.v4.runtime.ANTLRInputStream;
30  import org.antlr.v4.runtime.CharStream;
31  import org.antlr.v4.runtime.CommonTokenStream;
32  import org.antlr.v4.runtime.Token;
33  import org.antlr.v4.runtime.tree.ParseTree;
34  import org.junit.Test;
35  
36  import nl.knaw.huygens.Log;

38  public class AQLTest {
39  
40    @Test
41    public void testCorrectAQLStatement() {
42      String statement = "type:eq(\"Tag\")"//
43          + " who:eq(\"nedlab\")" //
44          + " state:eq(\"CONFIRMED\")"//
45          + " resource.id:inSet(1,2)"//
46          + " value:match(\"what.*\")"//
47          + " when:inRange(\"2015-01-01T00:00:00Z\",\"2016-01-01T00:00:00Z\")"//
48          ;
49      CharStream stream = new ANTLRInputStream(statement);
50      AQLLexer lex = new AQLLexer(stream);
51      List<? extends Token> allTokens = lex.getAllTokens();
52      for (Token token : allTokens) {
53        Log.info("token: [{}] <<{}>>", lex.getRuleNames()[token.getType() - 1], token.getText());
54      }
55      lex.reset();
56  
57      CommonTokenStream tokens = new CommonTokenStream(lex);
58      AQLParser parser = new AQLParser(tokens);
59      parser.setBuildParseTree(true);
60      ParseTree tree = parser.root();
61      Log.info("tree={}", tree.toStringTree(parser));
62      assertThat(tree.getChildCount()).isEqualTo(6); // 6 subqueries
63  
64      int numberOfSyntaxErrors = parser.getNumberOfSyntaxErrors();
65      assertThat(numberOfSyntaxErrors).isEqualTo(0); // no syntax errors
66      Log.info("numberOfSyntaxErrors={}", numberOfSyntaxErrors);
67  
68      for (int i = 0; i < tree.getChildCount(); i++) {
69        Log.info("root.child={}", tree.getChild(i).getText());
70      }
71      assertThat(allTokens).hasSize(40);
72    }
73  
74    @Test
75    public void testSpacesInStringsAreParsedCorrectly() {
76      String statement = "type:eq(\"Tag\")"//
77          + " who:eq(\"Jan Klaassen\")" //
78          + " value:eq(\"code: red\")" //
79          ;
80      CharStream stream = new ANTLRInputStream(statement);
81      AQLLexer lex = new AQLLexer(stream);
82      List<? extends Token> allTokens = lex.getAllTokens();
83      for (Token token : allTokens) {
84        Log.info("token: [{}] <<{}>>", lex.getRuleNames()[token.getType() - 1], token.getText());
85      }
86      lex.reset();
87  
88      CommonTokenStream tokens = new CommonTokenStream(lex);
89      AQLParser parser = new AQLParser(tokens);
90      parser.setBuildParseTree(true);
91      ParseTree tree = parser.root();
92      Log.info("tree={}", tree.toStringTree(parser));
93  
94      int numberOfSyntaxErrors = parser.getNumberOfSyntaxErrors();
95      assertThat(numberOfSyntaxErrors).isEqualTo(0); // no syntax errors
96      Log.info("numberOfSyntaxErrors={}", numberOfSyntaxErrors);
97  
98      assertThat(tree.getChildCount()).isEqualTo(3); // 2 subqueries
99  
100     for (int i = 0; i < tree.getChildCount(); i++) {
101       Log.info("root.child={}", tree.getChild(i).getText());
102     }
103     assertThat(allTokens).hasSize(18);
104   }
