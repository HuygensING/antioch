1   package nl.knaw.huygens.alexandria;
2   
3   import java.util.List;
4   
5   import com.google.common.base.Preconditions;
6   import com.google.common.collect.Lists;
7   import nl.knaw.huygens.alexandria.reference.ReferenceStore;
8   
9   public class InMemoryReferenceStore implements ReferenceStore {
10    private final List<String> references = Lists.newArrayList();
11  
12    @Override
13    public boolean createReference(final String id) {
14      Preconditions.checkNotNull(id, "id must not be null");
15      Preconditions.checkArgument(!id.isEmpty(), "id must not be empty");
16      
17      if (references.contains(id)) {
18        return false;
19      }
20  
21      references.add(id);
22      return true;
23    }
24  }
