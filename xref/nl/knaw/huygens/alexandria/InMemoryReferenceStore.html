1   package nl.knaw.huygens.alexandria;
2   
3   import java.util.Map;
4   import java.util.UUID;
5   
6   import com.google.common.base.Preconditions;
7   import com.google.common.base.Strings;
8   import com.google.common.collect.Maps;
9   import nl.knaw.huygens.alexandria.external.ReferenceStore;
10  
11  public class InMemoryReferenceStore implements ReferenceStore {
12    private final Map<UUID, String> references = Maps.newConcurrentMap();
13  
14    @Override
15    public boolean createReference(final UUID uuid, final String ref) {
16      Preconditions.checkNotNull(uuid, "id must not be null");
17      Preconditions.checkArgument(!Strings.isNullOrEmpty(ref), "ref must not be null or empty");
18  
19      if (references.containsKey(uuid)) {
20        return false;
21      }
22  
23      references.put(uuid, ref);
24      return true;
25    }
26  }
