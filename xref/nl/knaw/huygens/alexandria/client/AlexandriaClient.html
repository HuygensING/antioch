83    public AlexandriaClient(final URI alexandriaURI, SSLContext sslContext) {
84      this.alexandriaURI = alexandriaURI;
85      final ObjectMapper objectMapper = new ObjectMapper()//
86          .registerModule(new Jdk8Module())//
87          .registerModule(new JavaTimeModule());
88      final JacksonJaxbJsonProvider jacksonProvider = new JacksonJaxbJsonProvider();
89      jacksonProvider.setMapper(objectMapper);
90      if (sslContext == null) {
91        if ("https".equals(alexandriaURI.getScheme())) {
92          throw new RuntimeException("SSL connections need an SSLContext, use: new AlexandriaClient(uri, sslContext) instead.");
93        }
94        client = ClientBuilder.newClient(new ClientConfig(jacksonProvider));
95  
96      } else {
97        client = ClientBuilder.newBuilder()//
98            .sslContext(sslContext)//
99            .withConfig(new ClientConfig(jacksonProvider))//
100           .build();
101     }
102     client.property(ClientProperties.CONNECT_TIMEOUT, 60000);
103     client.property(ClientProperties.READ_TIMEOUT, 60000);
104     rootTarget = client.target(alexandriaURI);
105   }

153   public RestResult<UUID> addResource(final ResourcePrototype resource) {
154     final WebTarget path = rootTarget.path(EndpointPaths.RESOURCES);
155     final Entity<ResourcePrototype> entity = Entity.json(resource);
156     final Supplier<Response> responseSupplier = authorizedPost(path, entity);
157     final RestRequester<UUID> requester = RestRequester.withResponseSupplier(responseSupplier);
158     final RestResult<UUID> addResult = requester//
159         .onStatus(Status.CREATED, this::uuidFromLocationHeader)//
160         .getResult();
161     if (autoConfirm && !addResult.hasFailed()) {
162       confirmResource(addResult.get());
163     }
164 
165     return addResult;
166   }

396       confirmAnnotation(annotateResult.get());
397     }
398     return annotateResult;

412     final StatePrototype state = new StatePrototype().setState(AlexandriaState.CONFIRMED);
413     final Entity<StatePrototype> confirmation = Entity.json(state);
414     final WebTarget path = rootTarget//
415         .path(endpoint)//
416         .path(resourceUUID.toString())//
417         .path("state");
418     final Supplier<Response> responseSupplier = authorizedPut(path, confirmation);
419     final RestRequester<Void> requester = RestRequester.withResponseSupplier(responseSupplier);
420     return requester//
421         .onStatus(Status.NO_CONTENT, (response) -> new RestResult<>())//
422         .getResult();
423   }

481   private <E> RestResult<E> toEntityRestResult(final Response response, final Class<E> entityClass) {
482     final RestResult<E> result = new RestResult<>();
483     final E cargo = response.readEntity(entityClass);
484     result.setCargo(cargo);
485     return result;
486   }

488   private RestResult<URI> uriFromLocationHeader(final Response response) {
489     final RestResult<URI> result = new RestResult<>();
490     final String location = response.getHeaderString("Location");
491     final URI uri = URI.create(location);
492     result.setCargo(uri);
493     return result;
494   }
495 
496   private RestResult<UUID> uuidFromLocationHeader(final Response response) {
497     final RestResult<UUID> result = new RestResult<>();
498     final String location = response.getHeaderString("Location");
499     final UUID uuid = UUID.fromString(location.replaceFirst(".*/", ""));
500     result.setCargo(uuid);
501     return result;
502   }
