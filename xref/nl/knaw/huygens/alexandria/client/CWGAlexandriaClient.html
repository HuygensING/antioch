46    public List<UUID> findChildResourcesByRef(UUID parentUUID, String ref) throws AlexandriaException {
47      AlexandriaQuery query = new AlexandriaQuery()//
48          .setFind("resource")//
49          .setWhere(MessageFormat.format("subresource.sub:eq(\"{0}\") resource.id:eq(\"{1}\")", ref, parentUUID))//
50          .setReturns(QueryField.subresource_id.toString());
51      UUID searchId = addSearch(query);
52      SearchResultPage searchResultPage = getSearchResultPage(searchId);
53      return searchResultPage.getRecords()//
54          .stream()//
55          .map(r -> r.get(QueryField.subresource_id.toString()))//
56          .map(String.class::cast)//
57          .map(UUID::fromString)//
58          .collect(toList());
59    }
