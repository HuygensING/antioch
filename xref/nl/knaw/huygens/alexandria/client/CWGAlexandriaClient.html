48    public List<UUID> findChildResourcesByRef(UUID parentUUID, String ref) throws AlexandriaException {
49      AlexandriaQuery query = new AlexandriaQuery()//
50          .setFind("resource")//
51          .setWhere(MessageFormat.format("subresource.sub:eq(\"{0}\") resource.id:eq(\"{1}\")", ref, parentUUID))//
52          .setReturns(QueryField.subresource_id.toString());
53      UUID searchId = addSearch(query);
54      SearchResultPage searchResultPage = getSearchResultPage(searchId);
55      return searchResultPage.getRecords()//
56          .stream()//
57          .map(r -> r.get(QueryField.subresource_id.toString()))//
58          .map(String.class::cast)//
59          .map(UUID::fromString)//
60          .collect(toList());
61    }
