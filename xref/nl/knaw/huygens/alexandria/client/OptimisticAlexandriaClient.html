109   public TextAnnotationImportStatus addResourceTextRangeAnnotationsSynchronously(UUID resourceUUID, TextRangeAnnotationList textAnnotations) {
110     unwrap(delegate.addResourceTextRangeAnnotations(resourceUUID, textAnnotations));
111     TextAnnotationImportStatus finalStatus = textAnnotationImportStatusWhenFinished(resourceUUID);
112     // // throw exception when there are errors.
113     if (finalStatus.hasErrors()) {
114       throw new AlexandriaException(finalStatus.getBreakingErrorMessage());
115     }
116 
117     return finalStatus;
118   }

312   private <T> T unwrap(RestResult<T> restResult) {
313     if (restResult.hasFailed()) {
314       Optional<Response> response = restResult.getResponse();
315       String status = response.isPresent() ? response.get().getStatus() + ": " : "";
316       String message = status + restResult.getFailureCause().orElse("Unspecified error");
317       throw new AlexandriaException(message);
318     }
319     return restResult.get();
320   }

330   private TextImportStatus textImportStatusWhenFinished(UUID resourceUUID) {
331     TextImportStatus status = null;
332     boolean goOn = true;
333     while (goOn) {
334       try {
335         Thread.sleep(500);
336       } catch (InterruptedException e) {
337         e.printStackTrace();
338       }
339       status = unwrap(delegate.getTextImportStatus(resourceUUID));
340       goOn = !status.isDone();
341     }
342     return status;
343   }

345   private TextAnnotationImportStatus textAnnotationImportStatusWhenFinished(UUID resourceUUID) {
346     TextAnnotationImportStatus status = null;
347     boolean goOn = true;
348     while (goOn) {
349       try {
350         Thread.sleep(500);
351       } catch (InterruptedException e) {
352         e.printStackTrace();
353       }
354       status = unwrap(delegate.getResourceTextRangeAnnotationBatchImportStatus(resourceUUID));
355       goOn = !status.isDone();
356     }
357     return status;
358   }
