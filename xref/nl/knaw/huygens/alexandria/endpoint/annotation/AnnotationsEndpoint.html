54    @GET
55    @Path("{uuid}")
56    @ApiOperation(value = "get the annotation", response = AnnotationEntity.class)
57    public Response readAnnotation(@PathParam("uuid") UUIDParam uuidParam) {
58      final AlexandriaAnnotation annotation = service.readAnnotation(uuidParam.getValue())//
59          .orElseThrow(annotationNotFoundForId(uuidParam));
60      return Response.ok(entityBuilder.build(annotation)).build();
61    }

73    @POST
74    @Path("{uuid}/deprecation")
75    @Consumes(MediaType.APPLICATION_JSON)
76    @ApiOperation(value = "make a new annotation from the payload and use it to deprecate the annotation with the given uuid")
77    public Response deprecateAnnotation(@PathParam("uuid") UUIDParam uuidParam, AnnotationPrototype prototype) {
78      UUID uuid = uuidParam.getValue();
79      prototype.setState(AlexandriaState.TENTATIVE);
80      AnnotationDeprecationRequest request = requestBuilder.ofAnnotation(uuid).build(prototype);
81      AlexandriaAnnotation newAnnotation = request.execute(service);
82      return Response.created(locationBuilder.locationOf(newAnnotation)).build();
83    }
84  
85    @DELETE
86    @Path("{uuid}")
87    public Response deleteNotSupported(@PathParam("uuid") final UUIDParam uuidParam) {
88      UUID uuid = uuidParam.getValue();
89      final AlexandriaAnnotation annotation = service.readAnnotation(uuid)//
90          .orElseThrow(annotationNotFoundForId(uuidParam));
91      if (!annotation.getAnnotations().isEmpty()) {
92        throw new ConflictException("annotation " + uuid + " still has annotations");
93      }
94  
95      service.deleteAnnotation(annotation);
96  
97      return Response.noContent().build();
98    }
99  
100   @PUT
101   @Path("{uuid}/state")
102   @Consumes(MediaType.APPLICATION_JSON)
103   @ApiOperation(value = "update the state of the annotation (only state=CONFIRMED accepted for now)")
104   public Response setAnnotationState(@PathParam("uuid") final UUIDParam uuidParam, @NotNull StatePrototype protoType) {
105     Log.trace("protoType=[{}]", protoType);
106     UUID id = uuidParam.getValue();
107     AlexandriaAnnotation annotation = service.readAnnotation(id)//
108         .orElseThrow(annotationNotFoundForId(uuidParam));
109     if (protoType.isConfirmed()) {
110       if (!annotation.isActive()) {
111         throw new ConflictException(annotation.getState() + " annotations cannot be set to CONFIRMED");
112       }
113       service.confirmAnnotation(id);
114       return Response.ok().build();
115     }
116     throw new BadRequestException("for now, you can only set the state to CONFIRMED");
117   }
