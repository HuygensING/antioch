3   import javax.ws.rs.GET;
4   import javax.ws.rs.core.Response;
5   import java.util.Set;
6   import java.util.UUID;
7   
8   import nl.knaw.huygens.alexandria.model.AlexandriaAnnotation;
9   import nl.knaw.huygens.alexandria.service.ResourceService;
10  
11  public class ResourceAnnotations extends JSONEndpoint {
12    private final ResourceService resourceService;
13    private final UUID uuid;
14  
15    public ResourceAnnotations(ResourceService resourceService, UUID uuid) {
16      this.resourceService = resourceService;
17      this.uuid = uuid;
18    }
19  
20    @GET
21    public Response get() {
22      final Set<AlexandriaAnnotation> annotations = resourceService.readResource(uuid).getAnnotations();
23  
24      /*
25        return Response.ok(annotations).build() unfortunately yields:
26  
27        [{ "id" : "42f22020-cc82-11e4-aec9-1be06e873083", "annotations" : []},
28         { "id" : "92af4396-cc77-11e4-9b9b-1f1561a91434", "annotations" : []}]
29  
30        i.e.,
31          1) without the desired outer "annotations" wrapper, and
32          2) without the desired "annotation" wrapper around each annotation.
33  
34        Solution (for now?) is to use a static wrapper class (q.v.).
35       */
36  
37      return Response.ok(new AnnotationsWrapper(annotations)).build();
38    }
39  
40    static class AnnotationsWrapper {
41      private final Set<AlexandriaAnnotation> annotations;
42  
43      public AnnotationsWrapper(Set<AlexandriaAnnotation> annotations) {
44        this.annotations = annotations;
45      }
46  
47      public Set<AlexandriaAnnotation> getAnnotations() {
48        return annotations;
49      }
50    }
51  }
