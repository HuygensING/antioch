3   import javax.inject.Singleton;
4   import java.util.Map;
5   import java.util.Optional;
6   
7   import com.google.common.collect.ImmutableMap;
8   import nl.knaw.huygens.alexandria.endpoint.search.SearchResult;
9   import nl.knaw.huygens.alexandria.model.AlexandriaAnnotation;
10  import nl.knaw.huygens.alexandria.model.AlexandriaAnnotationBody;
11  import nl.knaw.huygens.alexandria.model.AlexandriaResource;
12  import nl.knaw.huygens.alexandria.model.Identifiable;
13  
14  @Singleton
15  public class EndpointPathResolver {
16    private static final Map<Class<? extends Identifiable>, String> IDENTIFIABLE_ENDPOINTS //
17        = ImmutableMap.<Class<? extends Identifiable>, String>builder()
18        .put(AlexandriaAnnotation.class, EndpointPaths.ANNOTATIONS)
19        .put(AlexandriaAnnotationBody.class, EndpointPaths.ANNOTATIONBODIES)
20        .put(AlexandriaResource.class, EndpointPaths.RESOURCES)
21        .put(SearchResult.class, EndpointPaths.SEARCHES)
22        .build();
23  
24    public Optional<String> pathOf(Class<? extends Identifiable> identifiableClass) {
25      return Optional.ofNullable(IDENTIFIABLE_ENDPOINTS.get(identifiableClass));
26    }
27  
28  }
