27  @Path(EndpointPaths.ANNOTATIONBODIES)
28  @Api(EndpointPaths.ANNOTATIONBODIES)
29  public class AnnotationBodiesEndpoint extends JSONEndpoint {
30    private final AlexandriaService service;
31    private final AnnotationBodyEntityBuilder entityBuilder;
32    private final AnnotationBodyCreationRequestBuilder requestBuilder;
33  
34    @Inject
35    public AnnotationBodiesEndpoint(AlexandriaService service, //
36        AnnotationBodyCreationRequestBuilder requestBuilder, //
37        AnnotationBodyEntityBuilder entityBuilder) {
38      this.service = service;
39      this.requestBuilder = requestBuilder;
40      this.entityBuilder = entityBuilder;
41    }
42  
43    @GET
44    @Path("{uuid}")
45    @ApiOperation(value = "get the annotation body", response = AnnotationBodyEntity.class)
46    public Response readAnnotationBody(@PathParam("uuid") UUIDParam uuidParam) {
47      final AlexandriaAnnotationBody annotationBody = service.readAnnotationBody(uuidParam.getValue())//
48          .orElseThrow(() -> new NotFoundException("No annotationbody found with id " + uuidParam));
49      return Response.ok(entityBuilder.build(annotationBody)).build();
50    }
51  
52    @POST
53    @Consumes(MediaType.APPLICATION_JSON)
54    @ApiOperation("create a new annotation body")
55    public Response createAnnotationBody(final AnnotationBodyPrototype prototype) {
56      final AnnotationBodyCreationRequest request = requestBuilder.build(prototype);
57      request.execute(service);
58  
59      if (request.wasCreated()) {
60        return Response.created(locationOf(prototype.getId().getValue())).build();
61      }
62  
63      return Response.status(Status.CONFLICT).build();
64    }
65  
66    @DELETE
67    @Path("{uuid}")
68    public Response deleteNotSupported(@PathParam("uuid") final UUIDParam paramId) {
69      return methodNotImplemented();
70    }
71  
72    private URI locationOf(UUID uuid) {
73      return URI.create(String.format("%s/%s", EndpointPaths.ANNOTATIONBODIES + "/", uuid));
74    }
