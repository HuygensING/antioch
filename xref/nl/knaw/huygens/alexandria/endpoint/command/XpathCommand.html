79      return Commands.XPATH;

82    @Override
83    public CommandResponse runWith(Map<String, Object> parameterMap) {
84      Parameters parameters = validateParameters(parameterMap);
85      Map<String, XPathResult> resultMap = new HashMap<>();
86      if (commandResponse.parametersAreValid()) {
87        for (ResourceViewId resourceViewId : parameters.resourceViewIds) {
88          service.runInTransaction(() -> {
89            UUID resourceId = resourceViewId.getResourceId();
90            String viewName = resourceViewId.getTextViewName().orElse("");
91            StreamingOutput xmlOutputStream = TextGraphUtil.xmlOutputStream(service, resourceId, viewName);
92            StringBuilderWriter sbWriter = new StringBuilderWriter();
93            WriterOutputStream writerOutputStream = new WriterOutputStream(sbWriter);
94            try {
95              xmlOutputStream.write(writerOutputStream);
96              writerOutputStream.close();
97              String xml = sbWriter.toString();
98              XPathResult result = testXPath(parameters.xpath, xml);
99              resultMap.put(resourceViewId.toString(), result);
100           } catch (WebApplicationException | IOException | XPathExpressionException e) {
101             commandResponse.addErrorLine(resourceId + ": " + e.getMessage());
102             e.printStackTrace();
103           }
104         });
105       }
106     }
107     commandResponse.setResult(resultMap);
108     return commandResponse;
109   }

111   static XPathResult testXPath(String xpathQuery, String xml) throws XPathExpressionException {
112     QueryableDocument qDoc = QueryableDocument.createFromXml(xml, true);
113 
114     try {
115       NodeList nodelist = qDoc.evaluateXPathToW3CNodeList(xpathQuery);
116       return new XPathResult(Type.NODESET, toStringList(nodelist));
117     } catch (XPathException e) {
118       if (e.getMessage().contains("#BOOLEAN")) {
119         Object b = qDoc.evaluateXPathToBoolean(xpathQuery);
120         return new XPathResult(Type.BOOLEAN, b);
121       }
122       if (e.getMessage().contains("#NUMBER")) {
123         Object n = qDoc.evaluateXPathToDouble(xpathQuery);
124         return new XPathResult(Type.NUMBER, n);
125       }
126     }
127 
128     String string = qDoc.evaluateXPathToString(xpathQuery);
129     return new XPathResult(Type.STRING, string);
130   }

132   private static List<String> toStringList(NodeList nodelist) {
133     List<String> list = new ArrayList<>(nodelist.getLength());
134     for (int i = 0; i < nodelist.getLength(); i++) {
135       Node node = nodelist.item(i);
136       list.add(toString(node));
137     }
138     return list;
139   }

141   public static String toString(Node node) {
142     if (node == null) {
143       throw new IllegalArgumentException("node is null.");
144     }
145 
146     try {
147       // Remove unwanted whitespaces
148       XPath xpath = XPathFactory.newInstance().newXPath();
149       XPathExpression expr = xpath.compile("//text()[normalize-space()='']");
150       NodeList nodeList = (NodeList) expr.evaluate(node, XPathConstants.NODESET);
151 
152       for (int i = 0; i < nodeList.getLength(); ++i) {
153         Node nd = nodeList.item(i);
154         nd.getParentNode().removeChild(nd);
155       }
156 
157       // Create and setup transformer
158       Transformer transformer = TransformerFactory.newInstance().newTransformer();
159       transformer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
160 
161       transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
162 
163       // transformer.setOutputProperty(OutputKeys.INDENT, "yes");
164       // transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "4");
165 
166       // Turn the node into a string
167       StringWriter writer = new StringWriter();
168       transformer.transform(new DOMSource(node), new StreamResult(writer));
169       return writer.toString();
170     } catch (TransformerException | XPathExpressionException e) {
171       throw new RuntimeException(e);
172     }
173   }

175   private Parameters validateParameters(Map<String, Object> parameterMap) {
176     final Parameters parameters = new Parameters();
177     parameters.resourceViewIds = validateResourceViewIds(parameterMap, commandResponse, service);
178     boolean valid = (commandResponse.getErrorLines().isEmpty());
179     if (!parameterMap.containsKey(PARAMETER_XPATH)) {
180       addXPathError();
181       valid = false;
182 
183     } else {
184       try {
185         parameters.xpath = (String) parameterMap.get(PARAMETER_XPATH);
186       } catch (ClassCastException e) {
187         addXPathError();
188         valid = false;
189       }
190     }
191 
192     if (valid) {
193       commandResponse.setParametersAreValid(true);
194     }
195     return parameters;
196   }
197 
198   private void addXPathError() {
199     commandResponse.addErrorLine("Parameter '" + PARAMETER_XPATH + "' should be a valid xpath query.");
200   }
