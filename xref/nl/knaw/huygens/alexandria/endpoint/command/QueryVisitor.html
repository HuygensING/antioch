25  import java.util.ArrayList;
26  import java.util.List;
27  
28  import org.antlr.v4.runtime.tree.TerminalNode;
29  
30  import nl.knaw.huygens.alexandria.antlr.AQL2BaseVisitor;
31  import nl.knaw.huygens.alexandria.antlr.AQL2Parser.ParameterContext;
32  import nl.knaw.huygens.alexandria.antlr.AQL2Parser.RootContext;
33  
34  public class QueryVisitor extends AQL2BaseVisitor<Void> {
35    public static final String QUOTE = String.valueOf('"');
36    String function;
37    List<Object> parameters = new ArrayList<>();
38  
39    @Override
40    public Void visitRoot(RootContext ctx) {
41      TerminalNode functionNode = ctx.FUNCTION();
42      if (functionNode == null) {
43        throw new RuntimeException("parse error: no function");
44      }
45      function = functionNode.getText();
46      return super.visitRoot(ctx);
47    }
48  
49    @Override
50    public Void visitParameter(ParameterContext ctx) {
51      parameters.add(parseParameterString(ctx.getText()));
52      return super.visitParameter(ctx);
53    }
54  
55    private static Object parseParameterString(String parameterString) {
56      if (parameterString.startsWith(QUOTE) && parameterString.endsWith(QUOTE)) {
57        return parameterString.replace(QUOTE, "");
58      }
59      return Long.valueOf(parameterString);
60    }
61  
62    public String getFunction() {
63      return function;
64    }
65  
66    public List<Object> getParameters() {
67      return parameters;
68    }
69  
70  }
