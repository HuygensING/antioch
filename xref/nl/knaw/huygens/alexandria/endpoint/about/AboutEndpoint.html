19  @Singleton
20  @Path("about")
21  @Api("about")
22  public class AboutEndpoint extends JSONEndpoint {
23    private static final Instant startedAt = Instant.now();
24  
25    private static PropertyResourceBundle propertyResourceBundle;
26  
27    private static synchronized String getProperty(String key) {
28      if (propertyResourceBundle == null) {
29        try {
30          propertyResourceBundle = new PropertyResourceBundle(//
31              Thread.currentThread().getContextClassLoader().getResourceAsStream("about.properties"));
32        } catch (IOException e) {
33          e.printStackTrace();
34        }
35      }
36  
37      try {
38        return propertyResourceBundle.getString(key);
39      } catch (MissingResourceException e) {
40        Log.warn("Missing expected resource: [{}] -- winging it", key);
41        return "missing";
42      } catch (ClassCastException e) {
43        Log.warn("Property value for key [{}] cannot be transformed to String -- winging it", key);
44        return "malformed";
45      }
46    }
47  
48    /**
49     * Show information about the back-end
50     *
51     * @return about-data map
52     */
53    @GET
54    @ApiOperation("get information about the server (version,buildDate,commitId,startedAt)")
55    public Response getMetadata() {
56      final Map<String, String> data = Maps.newLinkedHashMap();
57      data.put("buildDate", getProperty("buildDate"));
58      data.put("commitId", getProperty("commitId"));
59      data.put("startedAt", startedAt.toString());
60      data.put("version", getProperty("version"));
61      return Response.ok(data).build();
62    }
