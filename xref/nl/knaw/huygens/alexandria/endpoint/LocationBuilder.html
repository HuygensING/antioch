1   package nl.knaw.huygens.alexandria.endpoint;
2   
3   import javax.inject.Inject;
4   import javax.ws.rs.core.UriBuilder;
5   import java.net.URI;
6   import java.util.UUID;
7   import java.util.function.Supplier;
8   
9   import nl.knaw.huygens.alexandria.config.AlexandriaConfiguration;
10  import nl.knaw.huygens.alexandria.model.Identifiable;
11  import nl.knaw.huygens.alexandria.model.IdentifiablePointer;
12  
13  public class LocationBuilder {
14    private final AlexandriaConfiguration config;
15    private final EndpointPathResolver resolver;
16  
17    @Inject
18    public LocationBuilder(AlexandriaConfiguration config, EndpointPathResolver resolver) {
19      this.config = config;
20      this.resolver = resolver;
21    }
22  
23    public URI locationOf(Identifiable identifiable) {
24      return locationOf(identifiable.getClass(), identifiable.getId());
25    }
26  
27    public URI locationOf(Class<? extends Identifiable> identifiableClass, UUID uuid) {
28      return locationOf(identifiableClass, uuid.toString());
29    }
30  
31    public URI locationOf(Class<? extends Identifiable> identifiableClass, String uuid) {
32      return UriBuilder.fromUri(config.getBaseURI()) //
33          .path(pathOf(identifiableClass)) //
34          .path("{uuid}") //
35          .build(uuid);
36    }
37  
38    public URI locationOf(IdentifiablePointer<? extends Identifiable> identifiablePointer) {
39      return locationOf(identifiablePointer.getIdentifiableClass(), identifiablePointer.getIdentifier());
40    }
41  
42    private String pathOf(Class<? extends Identifiable> identifiableClass) {
43      return resolver.pathOf(identifiableClass).orElseThrow(unknownIdentifiableClass(identifiableClass));
44    }
45  
46    private Supplier<RuntimeException> unknownIdentifiableClass(Class<? extends Identifiable> identifiableClass) {
47      return () -> new RuntimeException("unknown Identifiable class " + identifiableClass);
48    }
49  }
