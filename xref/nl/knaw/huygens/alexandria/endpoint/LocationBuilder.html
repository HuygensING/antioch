1   package nl.knaw.huygens.alexandria.endpoint;
2   
3   import java.net.URI;
4   
5   import javax.inject.Inject;
6   import javax.ws.rs.core.UriBuilder;
7   
8   import nl.knaw.huygens.alexandria.config.AlexandriaConfiguration;
9   import nl.knaw.huygens.alexandria.model.Identifiable;
10  import nl.knaw.huygens.alexandria.model.IdentifiablePointer;
11  
12  public class LocationBuilder {
13  
14    private AlexandriaConfiguration config;
15    private EndpointPathResolver resolver;
16  
17    @Inject
18    public LocationBuilder(AlexandriaConfiguration config, EndpointPathResolver resolver) {
19      this.config = config;
20      this.resolver = resolver;
21    }
22  
23    public URI locationOf(Identifiable accountable) {
24      String path = resolver.pathOf(accountable)//
25          .orElseThrow(() -> new RuntimeException("unknown Identifiable class " + accountable.getClass()));
26      return UriBuilder.fromUri(config.getBaseURI())//
27          .path(path).path("{uuid}").build(accountable.getId());
28    }
29  
30    public URI locationOf(Class<? extends Identifiable> accountableClass, String uuid) {
31      String path = resolver.pathOf(accountableClass)//
32          .orElseThrow(() -> new RuntimeException("unknown Identifiable class " + accountableClass));
33      return UriBuilder.fromUri(config.getBaseURI())//
34          .path(path).path("{uuid}").build(uuid);
35    }
36  
37    public URI locationOf(IdentifiablePointer<? extends Identifiable> annotatablePointer) {
38      return locationOf(annotatablePointer.getIdentifiableClass(), annotatablePointer.getIdentifier());
39    }
40  }
