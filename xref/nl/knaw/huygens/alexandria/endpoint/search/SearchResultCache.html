3   import java.util.Optional;
4   import java.util.UUID;
5   import java.util.concurrent.Callable;
6   import java.util.concurrent.ExecutionException;
7   
8   import com.google.common.cache.Cache;
9   import com.google.common.cache.CacheBuilder;
10  
11  public class SearchResultCache {
12    static Cache<UUID, Optional<SearchResult>> cache = CacheBuilder.newBuilder()//
13        .maximumSize(1000)//
14        .build();
15  
16    public static void add(SearchResult searchResult) {
17      cache.put(searchResult.getId(), Optional.ofNullable(searchResult));
18    }
19  
20    static final Callable<Optional<SearchResult>> EMPTY_WHEN_MISSING = () -> Optional.empty();
21  
22    public static Optional<SearchResult> get(UUID id) {
23      Optional<SearchResult> oSearchResult = Optional.empty();
24      try {
25        oSearchResult = cache.get(id, EMPTY_WHEN_MISSING);
26      } catch (ExecutionException e) {
27        e.printStackTrace();
28      }
29      return oSearchResult;
30    }
31  
32  }
