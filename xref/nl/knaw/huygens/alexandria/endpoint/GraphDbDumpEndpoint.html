3   import javax.inject.Inject;
4   import javax.inject.Singleton;
5   import javax.ws.rs.GET;
6   import javax.ws.rs.Path;
7   import javax.ws.rs.Produces;
8   import javax.ws.rs.core.MediaType;
9   import javax.ws.rs.core.Response;
10  import javax.ws.rs.core.StreamingOutput;
11  
12  import nl.knaw.huygens.alexandria.service.TinkerPopService;
13  
14  @Singleton
15  @Path("dump")
16  public class GraphDbDumpEndpoint extends JSONEndpoint {
17    @Inject
18    TinkerPopService service;
19  
20    @GET
21    @Path("graphson")
22    public Response dumpGraphDbAsJSON() {
23      StreamingOutput stream = os -> {
24        service.dumpToGraphSON(os);
25      };
26      return Response.ok(stream).build();
27    }
28  
29    @GET
30    @Path("graphml")
31    @Produces(MediaType.APPLICATION_XML)
32    public Response dumpGraphDbAsGraphML() {
33      StreamingOutput stream = os -> {
34        service.dumpToGraphML(os);
35      };
36      return Response.ok(stream).build();
37    }
38  }
