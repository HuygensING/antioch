25  import java.lang.annotation.ElementType;
26  import java.lang.annotation.Retention;
27  import java.lang.annotation.RetentionPolicy;
28  import java.lang.annotation.Target;
29  import java.util.List;
30  
31  import javax.inject.Inject;
32  import javax.validation.Constraint;
33  import javax.validation.ConstraintValidator;
34  import javax.validation.ConstraintValidatorContext;
35  import javax.validation.Payload;
36  import javax.ws.rs.core.PathSegment;
37  import javax.ws.rs.core.UriInfo;
38  
39  import nl.knaw.huygens.Log;
40  import nl.knaw.huygens.alexandria.endpoint.UUIDParam;
41  
42  @Target({ElementType.FIELD, ElementType.PARAMETER})
43  @Retention(RetentionPolicy.RUNTIME)
44  @Constraint(validatedBy = MatchesPathId.Validator.class)
45  public @interface MatchesPathId {
46    String message() default "{nl.knaw.huygens.alexandria.endpoint.resource.MatchesPathId.message}";
47  
48    Class<?>[] groups() default {};
49  
50    Class<? extends Payload>[] payload() default {};
51  
52    class Validator implements ConstraintValidator<MatchesPathId, ResourcePrototype> {
53      final UriInfo uriInfo;
54  
55      @Inject
56      public Validator(UriInfo uriInfo) {
57        this.uriInfo = uriInfo;
58      }
59  
60      @Override
61      public void initialize(MatchesPathId constraintAnnotation) {
62        Log.info("initialize: annotation={}", constraintAnnotation);
63      }
64  
65      @Override
66      public boolean isValid(ResourcePrototype prototype, ConstraintValidatorContext context) {
67        if (prototype == null) {
68          return true; // Non-null validation, if desired, should be enforced via @NotNull
69        }
70  
71        final UUIDParam prototypeId = prototype.getId();
72  
73        // Either no id was passed in prototype, or it MUST equal the id in URI
74        return prototypeId == null || value(uriId()).equals(value(prototypeId));
75      }
76  
77      private String value(PathSegment segment) {
78        return segment.getPath();
79      }
80  
81      private String value(UUIDParam uuid) {
82        return String.valueOf(uuid.getValue());
83      }
84  
85      private PathSegment uriId() {
86        return uuidSegment(uriSegments());
87      }
88  
89      private PathSegment uuidSegment(List<PathSegment> pathSegments) {
90        final int uuidSegmentIndex = uriSegments().size() - 1;
91        return pathSegments.get(uuidSegmentIndex);
92      }
93  
94      private List<PathSegment> uriSegments() {
95        return uriInfo.getPathSegments();
96      }
97    }
