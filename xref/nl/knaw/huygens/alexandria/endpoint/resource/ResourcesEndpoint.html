77    @GET
78    @Path("{uuid}")
79    @ApiOperation(value = "Get the resource with the given uuid", response = ResourceEntity.class)
80    public Response getResourceByID(@PathParam("uuid") final UUIDParam uuidParam) {
81      return ok(readExistingResource(uuidParam));
82    }
83  
84    @POST
85    @Consumes(MediaType.APPLICATION_JSON)
86    @ApiOperation("create new Resource")
87    public Response createResource(@NotNull @Valid @WithoutId ResourcePrototype protoType) {
88      // Log.trace("protoType=[{}]", protoType);
89  
90      protoType.setState(AlexandriaState.TENTATIVE);
91      final ResourceCreationRequest request = requestBuilder.build(protoType);
92      AlexandriaResource resource = request.execute(service);
93  
94      if (request.wasExecutedAsIs()) {
95        return noContent();
96      }
97  
98      return created(resource);
99    }

101   @PUT
102   @Path("{uuid}/state")
103   @Consumes(MediaType.APPLICATION_JSON)
104   @ApiOperation(value = "update the state of the resource (only state=CONFIRMED accepted for now)")
105   public Response setResourceState(@PathParam("uuid") final UUIDParam uuidParam, @NotNull StatePrototype protoType) {
106     // Log.trace("protoType=[{}]", protoType);
107     AlexandriaResource resource = readExistingResource(uuidParam);
108     if (protoType.isConfirmed()) {
109       if (!resource.isActive()) {
110         throw new ConflictException(resource.getState() + " resources cannot be set to CONFIRMED");
111       }
112       service.confirmResource(resource.getId());
113       return noContent();
114     }
115     throw new BadRequestException("for now, you can only set the state to CONFIRMED");
116   }

124   @PUT
125   @Path("{uuid}")
126   @Consumes(MediaType.APPLICATION_JSON)
127   @ApiOperation(value = "update/create the resource with the given uuid")
128   public Response setResourceAtSpecificID(@PathParam("uuid") final UUIDParam uuid, //
129       @NotNull @Valid @MatchesPathId ResourcePrototype protoType) {
130     // Log.trace("protoType=[{}]", protoType);
131 
132     protoType.setState(AlexandriaState.CONFIRMED);
133     protoType.setId(uuid); // in case the prototype has no id, get it from the Path
134     Optional<AlexandriaResource> existingResource = service.readResource(uuid.getValue());
135     if (existingResource.isPresent() && isNotConfirmed(existingResource.get())) {
136       throw new ConflictException("This resource has state " + existingResource.get().getState() + "; only CONFIRMED resources can be updated.");
137     }
138     final ResourceCreationRequest request = requestBuilder.build(protoType);
139     AlexandriaResource resource = request.execute(service);
140 
141     if (request.newResourceWasCreated()) {
142       return created(resource);
143     }
144 
145     if (request.wasExecutedAsIs()) {
146       return noContent();
147     }
148 
149     return ok(resource);
150   }

185   private Response created(AlexandriaResource resource) {
186     return created(locationBuilder.locationOf(resource));
187   }
188 
189   private Response ok(AlexandriaResource resource) {
190     return ok(entityBuilder.build(resource));
191   }
192 
193   private AlexandriaResource readExistingResource(UUIDParam id) {
194     return service.readResource(id.getValue()).orElseThrow(resourceNotFoundForId(id));
195   }
196 
197   private boolean isNotConfirmed(AlexandriaResource resource) {
198     return !resource.getState().equals(AlexandriaState.CONFIRMED);
199   }
200 
201   private void assertResourceIsConfirmed(UUIDParam uuidParam) {
202     AlexandriaResource resource = readExistingResource(uuidParam);
203     if (isNotConfirmed(resource)) {
204       throw new ConflictException("This resource has state " + resource.getState() + "; it needs to be CONFIRMED first.");
205     }
206   }
