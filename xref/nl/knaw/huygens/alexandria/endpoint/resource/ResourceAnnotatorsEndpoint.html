51  
52  public class ResourceAnnotatorsEndpoint extends JSONEndpoint {
53  
54    private AlexandriaService service;
55    private LocationBuilder locationBuilder;
56    private AlexandriaResource resource;
57  
58    @Inject
59    public ResourceAnnotatorsEndpoint(AlexandriaService service, //
60        LocationBuilder locationBuilder, //
61        ResourceValidatorFactory validatorFactory, //
62        @PathParam("uuid") final UUIDParam uuidParam) {
63      this.service = service;
64      this.locationBuilder = locationBuilder;
65      this.resource = validatorFactory.validateExistingResource(uuidParam).notTentative().get();
66    }
67  
68    @GET
69    @ApiOperation("get annotators")
70    public Response getAnnotators() {
71      return ok(readAllAnnotatorsForResource());
72    }
73  
74    @PUT
75    @Path("{code}")
76    @Consumes(MediaType.APPLICATION_JSON)
77    @ApiOperation("set annotator")
78    public Response setAnnotator(@PathParam("code") final String code, //
79        @NotNull @ValidAnnotator Annotator annotator) {
80      annotator.setCode(code);
81      verifyAnnotatorCodeNotInUseByAncestorResource(code);
82      Optional<Annotator> existingAnnotator = service.readResourceAnnotator(resource.getId(), code);
83      service.setResourceAnnotator(resource.getId(), annotator);
84      if (existingAnnotator.isPresent()) {
85        return noContent();
86      }
87  
88      URI uri = locationBuilder.locationOf(resource, EndpointPaths.ANNOTATORS, code);
89      return created(uri);
90    }
91  
92    private void verifyAnnotatorCodeNotInUseByAncestorResource(String code) {
93      UUID resourceUUID = resource.getId();
94      AnnotatorList resourceAnnotators = service.readResourceAnnotators(resourceUUID);
95      Optional<Annotator> ancestorAnnotator = resourceAnnotators.stream()//
96          .filter(a -> a.getCode().equals(code))//
97          .filter(a -> !a.getResourceURI().toString().contains(resourceUUID.toString()))//
98          .findAny();
99      if (ancestorAnnotator.isPresent()) {
100       throw new ConflictException("Annotator '" + code + "' already defined in (sub)resource chain.");
101     }
102 
103   }
104 
105   @GET
106   @Path("{code}")
107   @ApiOperation("get annotator")
108   public Response getAnnotator(@PathParam("code") final String code) {
109     return ok(readExisitingAnnotator(code));
110   }
111 
112   private Annotator readExisitingAnnotator(String code) {
113     return service.readResourceAnnotator(resource.getId(), code).orElseThrow(annotatorNotFoundForCode(code));
114   }
115 
116   private AnnotatorList readAllAnnotatorsForResource() {
117     return service.readResourceAnnotators(resource.getId());
118   }
119 
120   private Supplier<NotFoundException> annotatorNotFoundForCode(String code) {
121     return () -> new NotFoundException("No annotator found with code " + code);
122   }
123 
124 }
