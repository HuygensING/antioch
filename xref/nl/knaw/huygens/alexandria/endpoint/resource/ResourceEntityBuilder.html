3   import javax.inject.Inject;

5   import nl.knaw.huygens.Log;
6   import nl.knaw.huygens.alexandria.endpoint.AbstractAnnotatableEntity;
7   import nl.knaw.huygens.alexandria.endpoint.LocationBuilder;
8   import nl.knaw.huygens.alexandria.model.AlexandriaResource;
9   
10  public class ResourceEntityBuilder {
11  
12    private final LocationBuilder locationBuilder;
13  
14    @Inject
15    public ResourceEntityBuilder(LocationBuilder locationBuilder) {
16      Log.trace("ResourceCreationRequestBuilder created: locationBuilder=[{}]", locationBuilder);
17      this.locationBuilder = locationBuilder;
18    }
19  
20    public AbstractAnnotatableEntity build(AlexandriaResource resource) {
21      return resource.isSubResource() ? SubResourceEntity.of(resource).withLocationBuilder(locationBuilder) //
22          : ResourceEntity.of(resource).withLocationBuilder(locationBuilder);
23    }
24  }
