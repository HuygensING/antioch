15  public abstract class AccountableProvenanceEndpoint extends JSONEndpoint {
16    protected final AlexandriaService service;
17    protected final UUID uuid;
18    private final LocationBuilder locationBuilder;
19  
20    protected AccountableProvenanceEndpoint(AlexandriaService service, //
21        final UUIDParam uuidParam, LocationBuilder locationBuilder) {
22      this.locationBuilder = locationBuilder;
23      this.service = service;
24      this.uuid = uuidParam.getValue();
25      Log.trace("resourceService=[{}], uuidParam=[{}]", service, uuidParam);
26    }
27  
28    protected abstract Accountable getAccountable();
29  
30    @GET
31    @ApiOperation(value = "get the provenance", response = ProvenanceEntity.class)
32    public Response get() {
33      AlexandriaProvenance provenance = getAccountable().getProvenance();
34      ProvenanceEntity entity = ProvenanceEntity.of(provenance).withLocationBuilder(locationBuilder);
35      return Response.ok(entity).build();
36    }
