1   package nl.knaw.huygens.alexandria.resource;
2   
3   import javax.ws.rs.GET;
4   import javax.ws.rs.POST;
5   import javax.ws.rs.Path;
6   import javax.ws.rs.PathParam;
7   import javax.ws.rs.core.Response;
8   import javax.ws.rs.core.Response.Status;
9   import java.net.URI;
10  
11  import nl.knaw.huygens.alexandria.service.AnnotationService;
12  
13  @Path("/annotations")
14  public class Annotations {
15    private final AnnotationService service = new AnnotationService();
16  
17    @POST
18    public Response createAnnotation(String key, String value) {
19      int id = service.createAnnotation(key, value);
20      return Response.created(URI.create(String.valueOf(id))).build();
21    }
22  
23    @GET
24    @Path("/{id}")
25    public Response readAnnotation(@PathParam("id") int id) {
26      return service.getAnnotation(id).map(AnnotationService.Anno::toString).map(Response::ok) //
27          .orElse(Response.status(Status.NOT_FOUND)).build();
28    }
29  }
