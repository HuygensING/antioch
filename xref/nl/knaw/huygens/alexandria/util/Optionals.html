12  // Thanks to Christoffer Sawicki (qerub) for https://gist.github.com/qerub/9299fce67ab74e114633
13  public class Optionals {
14    private Optionals() {
15    }
16  
17    public static void main(String[] args) {
18      Optional<String> opt1 = Optional.of("1");
19      Optional<String> opt2 = Optional.of("2");
20      Optional<String> opt3 = Optional.empty();
21  
22      // Transform an optional into something else via a stream:
23      Set<String> res1 = stream(opt1).collect(toSet());
24  
25      // Flatten a collection of optionals:
26      List<String> res2 = Stream.of(opt1, opt2, opt3).flatMap(Optionals::stream).collect(toList());
27  
28      // Combining multiple optionals, keeping the first present value:
29      Optional<String> res3 = or(opt1, opt2, opt3);
30    }
31  
32    /*
33     * This will be included in Java 9, but until then...
34     * (https://bugs.openjdk.java.net/browse/JDK-8050820)
35     */
36    public static <T> Stream<T> stream(Optional<T> self) {
37      return self.map(Stream::of).orElseGet(Stream::empty);
38    }
39  
40    @SafeVarargs
41    public static <T> Optional<T> or(Optional<T>... xs) {
42      return Arrays.stream(xs).flatMap(Optionals::stream).findFirst();
43    }
44  }
