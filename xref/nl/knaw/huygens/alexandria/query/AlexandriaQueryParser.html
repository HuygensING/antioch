54    public ParsedAlexandriaQuery parse(final AlexandriaQuery query) {
55      parseErrors.clear();
56  
57      final ParsedAlexandriaQuery paq = new ParsedAlexandriaQuery()//
58          .setVFClass(parseFind(query.getFind()))//
59          .setResultComparator(parseSort(query.getSort()));
60  
61      setFilter(paq, query.getWhere());
62  
63      parseReturn(paq, query.getFields());
64  
65      if (!parseErrors.isEmpty()) {
66        throw new AlexandriaQueryParseException(parseErrors);
67      }
68  
69      return paq;
70    }

72    private void setFilter(final ParsedAlexandriaQuery paq, String where) {
73      final List<WhereToken> tokens = tokenize(where);
74  
75      // any tokens with resource.id or subresource.id need to be filtered out and lead to an annotationVFFinder
76      List<WhereToken> resourceWhereTokens = filterResourceWhereTokens(tokens);
77      if (!resourceWhereTokens.isEmpty()) {
78        paq.setAnnotationVFFinder(createAnnotationVFFinder(resourceWhereTokens));
79      }
80  
81      tokens.removeAll(resourceWhereTokens);
82  
83      // create a predicate for filtering the annotationVF stream based on the remaining tokens
84      paq.setPredicate(createPredicate(tokens));
85    }

87    private List<WhereToken> filterResourceWhereTokens(List<WhereToken> tokens) {
88      return tokens.stream()//
89          .filter(WhereToken::hasResourceProperty)//
90          .collect(toList());
91    }

93    private Function<Storage, Stream<AnnotationVF>> createAnnotationVFFinder(List<WhereToken> resourceWhereTokens) {
94      WhereToken resourceWhereToken = resourceWhereTokens.get(0);
95      if (resourceWhereTokens.size() == 1 && resourceWhereToken.getFunction().equals(QueryFunction.eq)) {
96        String uuid = (String) resourceWhereToken.getParameters().get(0);
97        return storage -> {
98          Optional<ResourceVF> optionalResource = storage.readVF(ResourceVF.class, UUID.fromString(uuid));
99          if (optionalResource.isPresent()) {
100           ResourceVF resourceVF = optionalResource.get();
101           Stream<AnnotationVF> resourceAnnotationsStream = resourceVF.getAnnotatedBy().stream();
102           Stream<AnnotationVF> subresourceAnnotationsStream = resourceVF.getSubResources().stream()//
103               .flatMap(rvf -> rvf.getAnnotatedBy().stream());//
104           Stream<AnnotationVF> annotationVFStream = Stream.concat(resourceAnnotationsStream, subresourceAnnotationsStream);
105 
106           return annotationVFStream;
107         }
108         // Should return error, since no resource found with given uuid
109         return ImmutableList.<AnnotationVF> of().stream();
110       };
111 
112     }
113     return null;
114   }

116   private Class<? extends AlexandriaVF> parseFind(final String find) {
117     if (find.equals("annotation")) {
118       return AnnotationVF.class;
119 
120     } else if (find.equals("resource")) {
121       parseErrors.add("find: type 'resource' not supported yet");
122       return ResourceVF.class;
123 
124     } else {
125       parseErrors.add("find: unknown type '" + find + "', should be 'annotation'");
126       // parseErrors.add("unknown type '" + find + "' in find, should be 'annotation' or 'resource'");
127     }
128     return null;
129   }

131   List<WhereToken> tokenize(String whereString) {
132     Log.info("whereString=<{}>", whereString);
133     if (StringUtils.isEmpty(whereString)) {
134       // parseErrors.add("empty or missing where");
135       return Lists.newArrayList();
136     }
137 
138     QueryErrorListener errorListener = new QueryErrorListener();
139     CharStream stream = new ANTLRInputStream(whereString);
140     AQLLexer lex = new AQLLexer(stream);
141     lex.removeErrorListeners();
142     lex.addErrorListener(errorListener);
143     CommonTokenStream tokenStream = new CommonTokenStream(lex);
144     AQLParser parser = new AQLParser(tokenStream);
145     parser.removeErrorListeners();
146     parser.addErrorListener(errorListener);
147     parser.setBuildParseTree(true);
148     ParseTree tree = parser.root();
149     Log.info("tree={}", tree.toStringTree(parser));
150     if (errorListener.heardErrors()) {
151       parseErrors.addAll(errorListener.getParseErrors());
152       return Lists.newArrayList();
153     }
154 
155     QueryVisitor visitor = new QueryVisitor();
156     visitor.visit(tree);
157     parseErrors.addAll(errorListener.getParseErrors());
158     return visitor.getWhereTokens();
159   }

161   private static Predicate<AnnotationVF> createPredicate(List<WhereToken> tokens) {
162     if (tokens.isEmpty()) {
163       return alwaysTrue();
164     }
165     Predicate<AnnotationVF> predicate = tokens.stream()//
166         .map(AlexandriaQueryParser::toPredicate)//
167         .reduce(alwaysTrue(), (p, np) -> p = p.and(np));
168     return predicate;
169   }

171   static Predicate<AnnotationVF> toPredicate(WhereToken whereToken) {
172     Function<AnnotationVF, Object> getter = whereToken.getProperty().getter;
173     // eq
174     if (QueryFunction.eq.equals(whereToken.getFunction())) {
175       Object eqValue = whereToken.getParameters().get(0);
176       return (AnnotationVF avf) -> {
177         return getter.apply(avf).equals(eqValue);
178       };
179     }
180 
181     // match
182     if (QueryFunction.match.equals(whereToken.getFunction())) {
183       // TODO: catch errors
184       String matchValue = (String) whereToken.getParameters().get(0);
185       Pattern p = Pattern.compile(matchValue);
186       return (AnnotationVF avf) -> {
187         String propertyValue = (String) getter.apply(avf);
188         Matcher matcher = p.matcher(propertyValue);
189         return matcher.matches();
190       };
191     }
192 
193     // inSet
194     if (QueryFunction.inSet.equals(whereToken.getFunction())) {
195       List<Object> possibleValues = whereToken.getParameters();
196       return (AnnotationVF avf) -> {
197         Object propertyValue = getter.apply(avf);
198         return possibleValues.contains(propertyValue);
199       };
200     }
201 
202     // inRange
203     if (QueryFunction.inRange.equals(whereToken.getFunction())) {
204       List<Object> rangeLimits = whereToken.getParameters();
205       Object lowerLimit = rangeLimits.get(0);
206       Object upperLimit = rangeLimits.get(1);
207       return (AnnotationVF avf) -> {
208         Object propertyValue = getter.apply(avf);
209         if (propertyValue instanceof String) {
210           return ((String) propertyValue).compareTo((String) lowerLimit) >= 0//
211               && ((String) propertyValue).compareTo((String) upperLimit) <= 0;
212         }
213         if (propertyValue instanceof Long) {
214           return ((Long) propertyValue).compareTo((Long) lowerLimit) >= 0//
215               && ((Long) propertyValue).compareTo((Long) upperLimit) <= 0;
216 
217         }
218         return true;
219       };
220     }
221 
222     return alwaysTrue();
223   }

227       return true;

246   private Comparator<AnnotationVF> parseSort(final String sortString) {
247     // TODO: cache resultcomparator?
248     final List<SortToken> sortTokens = parseSortString(sortString);
249     if (sortTokens == null) {
250       // there were parse errors
251       return null;
252     }
253     List<Ordering<AnnotationVF>> orderings = sortTokens.stream()//
254         .map(AlexandriaQueryParser::ordering)//
255         .collect(toList());
256     Ordering<AnnotationVF> order = orderings.remove(0);
257     for (final Ordering<AnnotationVF> suborder : orderings) {
258       order = order.compound(suborder);
259     }
260     return order;
261   }

263   private static Ordering<AnnotationVF> ordering(SortToken token) {
264     boolean ascending = token.isAscending();
265     Function<AnnotationVF, Object> function = token.getField().getter;
266     return new Ordering<AnnotationVF>() {
267       @SuppressWarnings("unchecked")
268       @Override
269       public int compare(final AnnotationVF left, final AnnotationVF right) {
270         return ascending//
271             ? ((Comparable<Object>) function.apply(left)).compareTo(function.apply(right))//
272             : ((Comparable<Object>) function.apply(right)).compareTo(function.apply(left));
273       }
274     };
275   }

277   private List<SortToken> parseSortString(final String sortString) {
278     List<String> sortTokenStrings = splitToList(sortString);
279 
280     List<String> sortParseErrors = sortTokenStrings.stream()//
281         .map(AlexandriaQueryParser::extractExternalName)//
282         .filter(externalName -> !QueryField.ALL_EXTERNAL_NAMES.contains(externalName))//
283         .map(invalidFieldName -> "sort: unknown field: " + invalidFieldName + ALLOWEDFIELDS)//
284         .collect(toList());
285     if (!sortParseErrors.isEmpty()) {
286       parseErrors.addAll(sortParseErrors);
287       return null;
288     }
289 
290     final List<SortToken> sortFields = sortTokenStrings.stream()//
291         .map(AlexandriaQueryParser::sortToken)//
292         .collect(toList());
293     return sortFields;
294   }

296   static SortToken sortToken(final String f) {
297     boolean ascending = !f.startsWith("-");
298     String externalName = extractExternalName(f);
299     return new SortToken()//
300         .setAscending(ascending)//
301         .setField(QueryField.fromExternalName(externalName));
302   }

304   private static String extractExternalName(final String sortParameter) {
305     return sortParameter.replaceFirst("^[\\-\\+]", "");
306   }

308   private void parseReturn(final ParsedAlexandriaQuery paq, final String fieldString) {
309     final List<String> fields = splitToList(fieldString);
310     final List<String> allowedFields = QueryField.ALL_EXTERNAL_NAMES;
311     final List<String> unknownFields = Lists.newArrayList(fields);
312     unknownFields.removeAll(allowedFields);
313     if (!unknownFields.isEmpty()) {
314       parseErrors.add("return: unknown field(s) " + Joiner.on(", ").join(unknownFields) + ALLOWEDFIELDS);
315 
316     } else {
317       paq.setReturnFields(fields);
318 
319       final Function<AnnotationVF, Map<String, Object>> mapper = avf -> fields.stream()//
320           .collect(toMap(Function.identity(), f -> QueryField.fromExternalName(f).getter.apply(avf)));
321       // TODO: cache resultmapper?
322       paq.setResultMapper(mapper);
323     }
324   }

326   private static List<String> splitToList(final String fieldString) {
327     return Splitter.on(",").trimResults().splitToList(fieldString);
328   }
