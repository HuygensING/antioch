6   import java.time.Instant;
7   import java.util.Collections;
8   import java.util.HashSet;
9   import java.util.Set;
10  import java.util.UUID;
11  
12  import com.fasterxml.jackson.annotation.JsonTypeInfo;
13  import com.fasterxml.jackson.annotation.JsonTypeInfo.As;
14  import com.fasterxml.jackson.annotation.JsonTypeInfo.Id;
15  import com.fasterxml.jackson.annotation.JsonTypeName;
16  
17  @JsonTypeInfo(use = Id.NAME, include = As.WRAPPER_OBJECT)
18  @JsonTypeName("resource")
19  public class AlexandriaResource {
20    private final UUID id;
21  
22    private final Set<AlexandriaAnnotation> annotations = new HashSet<>();
23  
24    private final Instant createdOn;
25  
26    private String ref;
27  
28    public AlexandriaResource() {
29      this(UUID.randomUUID());
30    }
31  
32    public AlexandriaResource(UUID id) {
33      this(id, now());
34    }
35  
36    public AlexandriaResource(UUID id, Instant createdOn) {
37      this.id = id;
38      this.createdOn = createdOn;
39    }
40  
41    public AlexandriaResource(AlexandriaResource protoType) {
42      id = ifProvided(protoType.id).orElse(UUID.randomUUID());
43      ref = protoType.ref;
44      createdOn = ifProvided(protoType.createdOn).orElse(now());
45      annotations.addAll(protoType.annotations);
46    }
47  
48    public UUID getId() {
49      return id;
50    }
51  
52    public String getCreatedOn() {
53      return createdOn.toString();
54    }
55  
56    public Set<AlexandriaAnnotation> getAnnotations() {
57      return Collections.unmodifiableSet(annotations);
58    }
59  
60    public void addAnnotation(AlexandriaAnnotation annotation) {
61      annotations.add(annotation);
62    }
63  
64    @Override
65    public String toString() {
66      return getClass().getSimpleName() + "[id=" + getId() + ",createdOn=" + createdOn + "]";
67    }
68  
69    public String getRef() {
70      return ref;
71    }
72  
73    public void setRef(String ref) {
74      this.ref = ref;
75    }
76  }
