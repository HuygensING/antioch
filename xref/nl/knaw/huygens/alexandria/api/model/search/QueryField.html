30  import com.google.common.collect.ImmutableList;

32  public enum QueryField {
33    id, //
34    url, //
35    when, //
36    who, //
37    why, //
38    type, //
39    value, //
40    state, //
41    resource_id, //
42    subresource_id, //
43    resource_ref, //
44    subresource_sub, //
45    resource_url, //
46    subresource_url;
47  
48    public static final ImmutableList<QueryField> RESOURCE_FIELDS = ImmutableList.of(//
49        resource_id, resource_url, resource_ref, //
50        subresource_id, subresource_url, subresource_sub//
51    );
52  
53    public static final List<String> ALL_EXTERNAL_NAMES = Arrays.stream(values())//
54        .map(QueryField::externalName)//
55        .collect(toList());
56  
57    public String externalName() {
58      return name().replace('_', '.');
59    }
60  
61    public static QueryField fromExternalName(String externalName) {
62      return valueOf(externalName.replace('.', '_'));
63    }
