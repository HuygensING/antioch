25  import java.util.ArrayList;
26  import java.util.Arrays;
27  import java.util.List;
28  import java.util.Objects;
29  
30  public class ElementDefinition {
31    String name = "";
32    List<String> includedAttributes = new ArrayList<>();
33    String condition = "";
34  
35    private ElementDefinition() {
36    }
37  
38    public static ElementDefinition withName(final String name) {
39      ElementDefinition baseElementDefinition = new ElementDefinition();
40      baseElementDefinition.setName(name);
41      return baseElementDefinition;
42    }
43  
44    public ElementDefinition withAttributes(final String... attributes) {
45      setIncludedAttributes(Arrays.asList(attributes));
46      return this;
47    }
48  
49    public void setName(final String name) {
50      this.name = name;
51    }
52  
53    public String getName() {
54      return name;
55    }
56  
57    public String getCondition() {
58      return condition;
59    }
60  
61    public void setCondition(String condition) {
62      this.condition = condition;
63    }
64  
65    public void setIncludedAttributes(final List<String> includedAttributes) {
66      this.includedAttributes = includedAttributes;
67    }
68  
69    public List<String> getIncludedAttributes() {
70      return includedAttributes;
71    }
72  
73    @Override
74    public int hashCode() {
75      return Objects.hash(name, includedAttributes);
76    }
77  
78    @Override
79    public boolean equals(Object other) {
80      return other instanceof ElementDefinition //
81          && Objects.equals(this.getName(), ((ElementDefinition) other).getName())//
82          && Objects.equals(this.getIncludedAttributes(), ((ElementDefinition) other).getIncludedAttributes());
83    }
84  
85  }
