25  import java.util.Map;
26  
27  import org.apache.commons.lang3.builder.CompareToBuilder;
28  
29  public class XmlAnnotation extends TextAnnotation implements Comparable<XmlAnnotation> {
30    private boolean isMilestone;
31    private Integer firstSegmentIndex;
32    private Integer lastSegmentIndex;
33  
34    public XmlAnnotation(String name, Map<String, String> attributes, int depth) {
35      super(name, attributes, depth);
36    }
37  
38    public XmlAnnotation setMilestone(boolean bool) {
39      this.isMilestone = bool;
40      return this;
41    }
42  
43    public boolean isMilestone() {
44      return isMilestone;
45    }
46  
47    public XmlAnnotation setFirstSegmentIndex(Integer firstSegmentIndex) {
48      this.firstSegmentIndex = firstSegmentIndex;
49      return this;
50    }
51  
52    public Integer getFirstSegmentIndex() {
53      return firstSegmentIndex;
54    }
55  
56    public XmlAnnotation setLastSegmentIndex(Integer lastSegmentIndex) {
57      this.lastSegmentIndex = lastSegmentIndex;
58      return this;
59    }
60  
61    public Integer getLastSegmentIndex() {
62      return lastSegmentIndex;
63    }
64  
65    @Override
66    public int compareTo(XmlAnnotation other) {
67      return new CompareToBuilder()//
68          .append(this.firstSegmentIndex, other.firstSegmentIndex)//
69          .append(other.lastSegmentIndex, this.lastSegmentIndex)//
70          .append(getDepth(), other.getDepth())//
71          .build();
72    }
73  
74    @Override
75    public String toString() {
76      return "XmlAnnotation[" + getDepth() + ":" + firstSegmentIndex + "-" + lastSegmentIndex + "] <" + getName() + " " + getAttributes() + (isMilestone ? "/>" : ">");
77    }
78  
79  }
