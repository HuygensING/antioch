25  import java.util.List;
26  
27  import nl.knaw.huygens.alexandria.storage.VertexLabels;
28  import peapod.FramedVertex;
29  import peapod.annotations.Edge;
30  import peapod.annotations.In;
31  import peapod.annotations.Out;
32  import peapod.annotations.Vertex;
33  
34  @Vertex(VertexLabels.RESOURCE)
35  public abstract class ResourceVF extends AlexandriaVF implements FramedVertex<ResourceVF> {
36    public static class Properties {
37      public static final String CARGO = "cargo";
38    }
39  
40    public static class EdgeLabels {
41      public static final String PART_OF = "part_of";
42    }
43  
44    public abstract String getCargo();
45  
46    public abstract void setCargo(String cargo);
47  
48    public abstract Boolean getHasText();
49  
50    public abstract void setHasText(Boolean hasText);
51  
52    public abstract void setSerializedTextViewMap(String json);
53  
54    public abstract String getSerializedTextViewMap();
55  
56    public abstract void setSerializedTextViewDefinitionMap(String json);
57  
58    public abstract String getSerializedTextViewDefinitionMap();
59  
60    @In
61    @Edge(nl.knaw.huygens.alexandria.storage.frames.AnnotationVF.EdgeLabels.ANNOTATES_RESOURCE)
62    public abstract List<AnnotationVF> getAnnotatedBy();
63  
64    @In
65    @Edge(EdgeLabels.PART_OF)
66    public abstract List<ResourceVF> getSubResources();
67  
68    @In
69    @Edge(nl.knaw.huygens.alexandria.storage.frames.AnnotatorVF.EdgeLabels.HAS_RESOURCE)
70    public abstract List<AnnotatorVF> getAnnotators();
71  
72    @In
73    @Edge(nl.knaw.huygens.alexandria.storage.frames.TextRangeAnnotationVF.EdgeLabels.HAS_RESOURCE)
74    public abstract List<TextRangeAnnotationVF> getTextRangeAnnotations();
75  
76    @Out
77    @Edge(EdgeLabels.PART_OF)
78    public abstract ResourceVF getParentResource();
79  
80    @Out
81    @Edge(EdgeLabels.PART_OF)
82    public abstract void setParentResource(ResourceVF resourceVF);
83  
84    public boolean isSubResource() {
85      return getParentResource() != null;
86    }
87  
88  }
