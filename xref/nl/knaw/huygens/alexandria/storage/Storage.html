25  public class Storage {
26    private static final String IDENTIFIER_PROPERTY = "uuid";
27  
28    private Graph graph;
29    private FramedGraph framedGraph;
30  
31    private Transaction tx;
32    private String dumpfile;
33    private boolean supportsTransactions;
34    private boolean supportsPersistence;
35  
36    public Storage(final Graph graph) {
37      setGraph(graph);
38    }
39  
40    // - public methods - //
41  
42    // This is for the acceptancetest's benefit, so the tests kan start with a clear graph
43    public void setGraph(final Graph graph) {
44      this.graph = graph;
45      this.framedGraph = new FramedGraph(graph, ResourceVF.class.getPackage());
46      final GraphFeatures graphFeatures = graph.features().graph();
47      this.supportsPersistence = graphFeatures.supportsPersistence();
48      this.supportsTransactions = graphFeatures.supportsTransactions();
49    }
50  
51    public boolean supportsTransactions() {
52      return supportsTransactions;
53    }
54  
55    public boolean supportsPersistence() {
56      return supportsPersistence;
57    }
58  
59    // framedGraph methods
60  
61    public void startTransaction() {
62      if (supportsTransactions()) {
63        tx = framedGraph.tx();
64      }
65    }
66  
67    public void commitTransaction() {
68      if (supportsTransactions()) {
69        tx.commit();
70        tx.close();
71      }
72      if (!supportsPersistence()) {
73        saveToDisk(getDumpFile());
74      }
75    }
76  
77    public void rollbackTransaction() {
78      if (supportsTransactions()) {
79        tx.rollback();
80        tx.close();
81      } else {
82        Log.error("rollback called, but transactions are not supported by graph {}", graph);
83      }
84    }
85  
86    public boolean existsVF(final Class<? extends AlexandriaVF> vfClass, final UUID uuid) {
87      assertClass(vfClass);
88      return find(vfClass, uuid).tryNext().isPresent();
89    }
90  
91    public <A extends AlexandriaVF> A createVF(final Class<A> vfClass) {
92      assertClass(vfClass);
93      return framedGraph.addVertex(vfClass);
94    }
95  
96    public <A extends AlexandriaVF> Optional<A> readVF(final Class<A> vfClass, final UUID uuid) {
97      assertClass(vfClass);
98      return firstOrEmpty(find(vfClass, uuid).toList());
99    }
100 
101   public <A extends AlexandriaVF> FramedGraphTraversal<Object, A> find(Class<A> vfClass) {
102     return framedGraph.V(vfClass);
103   }
104 
105   // graph methods
106 
107   public void removeExpiredTentatives(final Long threshold) {
108     graph.traversal().V()//
109         .has("state", AlexandriaState.TENTATIVE.name())//
110         .has("stateSince", lt(threshold))//
111         .forEachRemaining(Element::remove);
112   }
113 
114   public void removeVertexWithId(final String annotationBodyId) {
115     graph.traversal().V()//
116         .has(IDENTIFIER_PROPERTY, annotationBodyId).next().remove();
117   }
118 
119   public void dumpToGraphSON(final OutputStream os) throws IOException {
120     graph.io(new GraphSONIo.Builder()).writer().create().writeGraph(os, graph);
121   }
122 
123   public void dumpToGraphML(final OutputStream os) throws IOException {
124     graph.io(new GraphMLIo.Builder()).writer().create().writeGraph(os, graph);
125   }
126 
127   public void loadFromDisk(final String file) {
128     try {
129       graph.io(new GraphMLIo.Builder()).readGraph(file);
130     } catch (final IOException e) {
131       throw new RuntimeException(e);
132     }
133   }
134 
135   public void saveToDisk(final String file) {
136     if (file != null) {
137       try {
138         graph.io(new GraphMLIo.Builder()).writeGraph(file);
139       } catch (final IOException e) {
140         throw new RuntimeException(e);
141       }
142     }
143   }
144 
145   public void setDumpFile(final String dumpfile) {
146     this.dumpfile = dumpfile;
147   }
148 
149   // - private methods - //
150 
151   private <A extends AlexandriaVF> FramedGraphTraversal<Object, A> find(final Class<A> vfClass, final UUID uuid) {
152     return find(vfClass).has(IDENTIFIER_PROPERTY, uuid.toString());
153   }
154 
155   private String getDumpFile() {
156     return dumpfile;
157   }
158 
159   private <A> Optional<A> firstOrEmpty(final List<A> results) {
160     return results.isEmpty() ? Optional.empty() : Optional.ofNullable(results.get(0));
161   }
162 
163   private void assertClass(final Class<? extends AlexandriaVF> clazz) {
164     if (clazz.getAnnotationsByType(peapod.annotations.Vertex.class).length == 0) {
165       throw new RuntimeException("Class " + clazz + " has no peapod @Vertex annotation, are you sure it is the correct class?");
166     }
167   }
