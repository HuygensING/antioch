84    public void storeTextGraph(UUID resourceUUID, ParseResult result) {
85      storage.runInTransaction(() -> {
86        Vertex resource = getResourceVertex(resourceUUID);
87        Vertex text = storage.addVertex(T.label, VertexLabels.TEXTGRAPH);
88        resource.addEdge(EdgeLabels.HAS_TEXTGRAPH, text);
89        resource.property("hasText", true);
90        List<Vertex> textSegments = storeTextSegments(result.getTextSegments(), text);
91        storeTextAnnotations(result.getXmlAnnotations(), text, textSegments);
92        reindex(resourceUUID);
93      });
94    }

166   public void wrapContentInChildTextAnnotation(TextAnnotation parentTextAnnotation, TextAnnotation newChildTextAnnotation) {
167     Vertex parentVertex = getTextAnnotationVertex(parentTextAnnotation);
168     Iterator<Edge> parentOutEdges = parentVertex.edges(Direction.OUT, EdgeLabels.FIRST_TEXT_SEGMENT, EdgeLabels.LAST_TEXT_SEGMENT, EdgeLabels.NEXT);
169     Vertex childVertex = toVertex(newChildTextAnnotation);
170 
171     // copy FIRST_TEXT_SEGMENT, LAST_TEXT_SEGMENT, NEXT edges from parentVertex to childVertex
172     while (parentOutEdges.hasNext()) {
173       Edge outEdge = parentOutEdges.next();
174       childVertex.addEdge(outEdge.label(), outEdge.inVertex());
175     }
176     // remove existing NEXT edge for parentVertex, replace with NEXT edge pointing to childVertex
177     Iterator<Edge> parentNextEdgeIterator = parentVertex.edges(Direction.OUT, EdgeLabels.NEXT);
178     if (parentNextEdgeIterator.hasNext()) {
179       Edge nextEdge = parentNextEdgeIterator.next();
180       nextEdge.remove();
181       parentVertex.addEdge(EdgeLabels.NEXT, childVertex);
182 
183       // increase the depth of the next textannotations as long as the textrange they annotate overlaps with the textrange of the parent annotation
184       updateDepths(parentVertex, childVertex, parentTextAnnotation.getDepth());
185     }
186   }

188   private static void updateDepths(Vertex parentVertex, Vertex childVertex, int parentDepth) {
189     // TODO: use start/end textSegmentIndex to determine the relevant annotations to adjust.
190     Vertex firstTextSegment = firstTextSegment(parentVertex);
191     Vertex lastTextSegment = lastTextSegment(parentVertex);
192     Set<Vertex> updatedVertices = Sets.newHashSet();
193     updatedVertices.add(parentVertex);
194     updatedVertices.add(childVertex);
195     boolean goOn = true;
196     Vertex textSegment = firstTextSegment;
197     while (goOn) {
198       StreamUtil.stream(textSegment.vertices(Direction.IN, EdgeLabels.FIRST_TEXT_SEGMENT))//
199           .filter(v -> VertexLabels.TEXTANNOTATION.equals(v.label()))//
200           .filter(v -> !updatedVertices.contains(v))//
201           .forEach(v -> {
202             // LOG.debug("v={}", v);
203             // LOG.debug("updatedVertices={}", updatedVertices);
204             int currentDepth = getIntValue(v, TextAnnotation.Properties.depth);
205             if (currentDepth > parentDepth) {
206               v.property(TextAnnotation.Properties.depth, currentDepth + 1);
207               updatedVertices.add(v);
208             }
209           });
210       goOn = !(textSegment.equals(lastTextSegment));
211       if (goOn) {
212         Iterator<Vertex> nextTextSegment = textSegment.vertices(Direction.OUT, EdgeLabels.NEXT);
213         if (nextTextSegment.hasNext()) {
214           textSegment = nextTextSegment.next();
215         } else {
216           LOG.error("There seems to be something wrong with the graph.");
217           goOn = false;
218         }
219       }
220     }

223   public void updateTextAnnotationLink(TextRangeAnnotationVF vf, TextRangeAnnotation textRangeAnnotation, UUID resourceUUID) {
224     // LOG.debug("textRangeAnnotation={}", textRangeAnnotation);
225     // if the TextRangeAnnotationVF is already linked to a TextAnnotation, remove that TextAnnotation
226     FramedGraphTraversal<TextRangeAnnotationVF, Vertex> traversal = vf.out(nl.knaw.huygens.alexandria.storage.frames.TextRangeAnnotationVF.EdgeLabels.HAS_TEXTANNOTATION);
227     if (traversal.hasNext()) {
228       traversal.next().remove();
229     }
230 
231     Vertex newTextAnnotationVertex = createNewTextAnnotation(vf, textRangeAnnotation);
232     // LOG.debug("textRangeAnnotation={}", textRangeAnnotation);
233 
234     TextAnnotationInsertionContext context = new TextAnnotationInsertionContext(newTextAnnotationVertex, textRangeAnnotation);
235     Vertex parentTextAnnotationVertex = getVertexTraversalFromResource(resourceUUID)//
236         .out(EdgeLabels.HAS_TEXTGRAPH)//
237         .out(EdgeLabels.FIRST_ANNOTATION)//
238 
239         // find the TextAnnotation with the xml:id from annotation.position.xmlid
240         .until(__.has(TextAnnotation.Properties.xmlid, textRangeAnnotation.getAbsolutePosition().getXmlId()))//
241         .repeat(__.out(EdgeLabels.NEXT))//
242         .next();
243     if (context.annotationIsMilestone) {
244       handleMilestoneAnnotation(newTextAnnotationVertex, context, parentTextAnnotationVertex);
245     } else {
246       handleRegularAnnotation(resourceUUID, context, parentTextAnnotationVertex);
247     }
248     if (context.reindexNeeded) {
249       reindex(resourceUUID);
250     }
251     context.insertNewTextAnnotationVertex();
252     reindex(resourceUUID);
253     if (LOG.isDebugEnabled()) {
254       LOG.debug("newTextAnnotationVertex={}", visualizeVertex(newTextAnnotationVertex));
255     }
256   }

258   private void handleRegularAnnotation(UUID resourceUUID, TextAnnotationInsertionContext context, Vertex parentTextAnnotationVertex) {
259     LOG.debug("handleMilestoneAnnotation");
260     List<Vertex> list = storage.getVertexTraversal(parentTextAnnotationVertex)//
261         .out(EdgeLabels.FIRST_TEXT_SEGMENT)//
262         // find the textsegment where the textrange from annotation.position starts
263         .until(context::rangeStartsInThisTextSegment)//
264         .repeat(__.out(EdgeLabels.NEXT))//
265         .sideEffect(context::processFirstTextSegmentInRange)//
266 
267         // iterate over the next TextSegments until you find the one the textrange from annotation.position ends in
268         .until(context::rangeEndsInThisTextSegment)//
269         .repeat(__.out(EdgeLabels.NEXT))//
270 
271         .sideEffect(context::processLastTextSegmentInRange)//
272 
273         .toList();
274     if (list.size() != 1) {
275       LOG.error("listsize should be 1, is {}", list.size());
276     }

279   private void handleMilestoneAnnotation(Vertex newTextAnnotationVertex, TextAnnotationInsertionContext context, Vertex parentTextAnnotationVertex) {
280     LOG.debug("handleMilestoneAnnotation");
281     // TODO: refactor this mess!
282 
283     int startOffset = 0;
284     int endOffset = 0;
285     String text = "";
286     Vertex textSegment = parentTextAnnotationVertex.vertices(Direction.OUT, EdgeLabels.FIRST_TEXT_SEGMENT).next();
287     boolean parentTextAnnotationIsMilestone = parentTextAnnotationIsMilestone(parentTextAnnotationVertex, textSegment);
288 
289     if (parentTextAnnotationIsMilestone) {
290       context.linkTextAnnotationToTextSegment(newTextAnnotationVertex, textSegment);
291 
292     } else {
293       Vertex emptyTextSegment = newTextSegmentVertex("");
294       context.linkTextAnnotationToTextSegment(newTextAnnotationVertex, emptyTextSegment);
295 
296       // find the textsegment where the textrange from annotation.position starts
297       boolean goOn = true;
298       while (goOn) {
299         text = getStringValue(textSegment, TextSegment.Properties.text);
300         int segmentSize = text.length();
301         goOn = startOffset + segmentSize < context.rangeStart;
302         endOffset += segmentSize;
303         if (goOn) {
304           Iterator<Vertex> nextTextSegments = textSegment.vertices(Direction.OUT, EdgeLabels.NEXT);
305           if (nextTextSegments.hasNext()) {
306             startOffset += segmentSize;
307             textSegment = nextTextSegments.next();
308           } else {
309             goOn = false;
310           }
311         }
312       }
313       if (startOffset == context.rangeStart) {
314         context.insertNewBeforeCurrent(emptyTextSegment, textSegment);
315         StreamUtil.stream(textSegment.edges(Direction.IN, EdgeLabels.FIRST_TEXT_SEGMENT))//
316             .forEach(e -> context.moveEdge(e, EdgeLabels.FIRST_TEXT_SEGMENT, emptyTextSegment));
317 
318       } else if (endOffset == context.rangeEnd) {
319         context.insertNewAfterCurrent(emptyTextSegment, textSegment);
320         StreamUtil.stream(textSegment.edges(Direction.IN, EdgeLabels.LAST_TEXT_SEGMENT))//
321             .forEach(e -> context.moveEdge(e, EdgeLabels.LAST_TEXT_SEGMENT, emptyTextSegment));
322 
323       } else {
324         // split up textSegment?
325         int headLength = context.rangeStart - startOffset - 1;
326         String headText = text.substring(0, headLength);
327         String tailText = text.substring(headLength);
328         textSegment.property(TextSegment.Properties.text, headText);
329         Vertex newTail = newTextSegmentVertex(tailText);
330         StreamUtil.stream(textSegment.edges(Direction.IN, EdgeLabels.LAST_TEXT_SEGMENT))//
331             .forEach(e -> context.moveEdge(e, EdgeLabels.LAST_TEXT_SEGMENT, newTail));
332         context.insertNewAfterCurrent(newTail, textSegment);
333         context.insertNewAfterCurrent(emptyTextSegment, textSegment);
334       }
335     }
336 
337     context.reindexNeeded = true;
338   }

408     private void insertAfterParent() {
409       // LOG.debug("startingTextSegment={}", visualizeVertex(startingTextSegment));
410       Vertex parentVertex = storage.getVertexTraversal(startingTextSegment)//
411           .in(EdgeLabels.FIRST_TEXT_SEGMENT)//
412           .hasLabel(VertexLabels.TEXTANNOTATION)//
413           .has(TextAnnotation.Properties.xmlid, parentXmlId)//
414           .next();
415       // LOG.debug("parentVertex={}", visualizeVertex(parentVertex));
416       Iterator<Edge> edges = parentVertex.edges(Direction.OUT, EdgeLabels.NEXT);
417       if (edges.hasNext()) {
418         Edge oldNextEdge = edges.next();
419         Vertex nextTextAnnotation = oldNextEdge.inVertex();
420         newTextAnnotationVertex.addEdge(EdgeLabels.NEXT, nextTextAnnotation);
421         oldNextEdge.remove();
422       }
423       parentVertex.addEdge(EdgeLabels.NEXT, newTextAnnotationVertex);
424       int parentDepth = getDepth(parentVertex);
425       newTextAnnotationVertex.property(TextAnnotation.Properties.depth, parentDepth + 1);
426       updateDepths(parentVertex, newTextAnnotationVertex, parentDepth);
427     }

523     void processFirstTextSegmentInRange(Traverser<Vertex> t) {
524       LOG.debug("processFirstTextSegmentInRange()");
525       Vertex textSegmentVertex = t.get();
526       checkVertexLabel(textSegmentVertex, VertexLabels.TEXTSEGMENT);
527 
528       // if needed, split up the textsegment, preserving the TextAnnotation links
529       int tailLength = Math.min(textSize, textSize - rangeStart + 1);
530       LOG.debug("textSize = {}, tailLength = {}", textSize, tailLength);
531 
532       // link the new TextAnnotation to the tail if detaching was necessary, to the firstTextSegment otherwise
533       this.startingTextSegment = detachTail(textSegmentVertex, tailLength);
534       newTextAnnotationVertex.addEdge(EdgeLabels.FIRST_TEXT_SEGMENT, this.startingTextSegment);
535     }

537     boolean rangeEndsInThisTextSegment(Traverser<Vertex> t) {
538       incTextSize(t);
539       LOG.debug("textSize:{},rangeEnd:{}", textSize, rangeEnd);
540       return textSize >= rangeEnd;
541     }

566     private Vertex detachTail(Vertex textSegment, int tailLength) {
567       Preconditions.checkArgument(textSegment.label().equals(VertexLabels.TEXTSEGMENT));
568       String text = getStringValue(textSegment, TextSegment.Properties.text);
569       int length = text.length();
570       textSize = textSize - length;
571       int headLength = length - tailLength;
572       String headText = text.substring(0, headLength);
573       String tailText = text.substring(headLength);
574       LOG.debug("detachTail(): head = [{}], tail = [{}]", headText, tailText);
575       if (headLength == 0) {
576         // no detachment necessary?
577         return textSegment;
578 
579       } else {
580         textSegment.property(TextSegment.Properties.text, headText);
581         Vertex tailTextSegment = newTextSegmentVertex(tailText);
582         insertNewAfterCurrent(tailTextSegment, textSegment);
583         // move LAST_TEXT_SEGMENT edges to tailTextSegment
584         StreamUtil.stream(textSegment.edges(Direction.IN, EdgeLabels.LAST_TEXT_SEGMENT))//
585             .forEach(e -> moveEdge(e, EdgeLabels.LAST_TEXT_SEGMENT, tailTextSegment));
586         reindexNeeded = true;
587         return tailTextSegment;
588       }
589     }

591     private void insertNewAfterCurrent(Vertex newVertex, Vertex currentVertex) {
592       Iterator<Edge> edges = currentVertex.edges(Direction.OUT, EdgeLabels.NEXT);
593       if (edges.hasNext()) {
594         Edge nextEdge = edges.next();
595         Vertex nextVertex = nextEdge.inVertex();
596         nextEdge.remove();
597         newVertex.addEdge(EdgeLabels.NEXT, nextVertex);
598       }
599       currentVertex.addEdge(EdgeLabels.NEXT, newVertex);
600     }

602     private Vertex detachHead(Vertex textSegment, int tailLength) {
603       Preconditions.checkArgument(textSegment.label().equals(VertexLabels.TEXTSEGMENT));
604       String text = getStringValue(textSegment, TextSegment.Properties.text);
605       int length = text.length();
606       int headLength = length - tailLength;
607       String headText = text.substring(0, headLength);
608       String tailText = text.substring(headLength);
609       LOG.debug("detachHead(): head = [{}], tail = [{}]", headText, tailText);
610       if (tailLength == 0) {
611         // no detachment necessary
612         return textSegment;
613 
614       } else {
615         textSegment.property(TextSegment.Properties.text, tailText);
616         Vertex headTextSegment = newTextSegmentVertex(headText);
617         // move FIRST_TEXT_SEGMENT edges to tailTextSegment
618         insertNewBeforeCurrent(headTextSegment, textSegment);
619         StreamUtil.stream(textSegment.edges(Direction.IN, EdgeLabels.FIRST_TEXT_SEGMENT))//
620             .forEach(e -> moveEdge(e, EdgeLabels.FIRST_TEXT_SEGMENT, headTextSegment));
621         reindexNeeded = true;
622         return headTextSegment;
623       }
624     }

627       Vertex textAnnotationVertex = e.outVertex();
628       textAnnotationVertex.addEdge(firstTextSegment, headTextSegment);
629       e.remove();

632     private void insertNewBeforeCurrent(Vertex newVertex, Vertex currentVertex) {
633       Iterator<Edge> edges = currentVertex.edges(Direction.IN, EdgeLabels.NEXT);
634       if (edges.hasNext()) {
635         Edge prevEdge = edges.next();
636         Vertex prevVertex = prevEdge.outVertex();
637         prevEdge.remove();
638         prevVertex.addEdge(EdgeLabels.NEXT, newVertex);
639       }
640       newVertex.addEdge(EdgeLabels.NEXT, currentVertex);
641     }

671   private List<Vertex> storeTextSegments(List<String> textSegments, Vertex text) {
672     List<Vertex> textSegmentVertices = new ArrayList<>();
673     Vertex previous = null;
674     for (String s : textSegments) {
675       Vertex v = newTextSegmentVertex(s);
676       if (previous == null) {
677         text.addEdge(EdgeLabels.FIRST_TEXT_SEGMENT, v);
678       } else {
679         previous.addEdge(EdgeLabels.NEXT, v);
680       }
681       textSegmentVertices.add(v);
682       previous = v;
683     }
684     return textSegmentVertices;
685   }

695       } else {
696         previous.addEdge(EdgeLabels.NEXT, v);

698       previous = v;
699     }

708   private static void update(Vertex v, TextAnnotation textAnnotation) {
709     Map<String, String> attributes = textAnnotation.getAttributes();
710     String[] attributeKeys = new String[attributes.size()];
711     String[] attributeValues = new String[attributes.size()];
712     int i = 0;
713     for (Entry<String, String> kv : attributes.entrySet()) {
714       attributeKeys[i] = kv.getKey();
715       attributeValues[i] = kv.getValue();
716       i++;
717     }
718     v.property(TextAnnotation.Properties.name, textAnnotation.getName());
719     v.property(TextAnnotation.Properties.attribute_keys, attributeKeys);
720     v.property(TextAnnotation.Properties.attribute_values, attributeValues);
721     v.property(TextAnnotation.Properties.depth, textAnnotation.getDepth());
722     if (attributes.containsKey("xml:id")) {
723       v.property(TextAnnotation.Properties.xmlid, attributes.get("xml:id"));
724     }
725   }

727   private TextGraphSegment toTextGraphSegment(Vertex textSegment, List<List<String>> orderedLayerDefinitions) {
728     TextGraphSegment textGraphSegment = new TextGraphSegment();
729     if (textSegment.keys().contains(TextSegment.Properties.text)) {
730       textGraphSegment.setTextSegment(textSegment.value(TextSegment.Properties.text));
731     }
732     List<TextAnnotation> textAnnotationsToOpen = getTextAnnotationsToOpen(textSegment, orderedLayerDefinitions);
733     List<TextAnnotation> textAnnotationsToClose = getTextAnnotationsToClose(textSegment, orderedLayerDefinitions);
734 
735     boolean isMilestone = StringUtils.isEmpty(textGraphSegment.getTextSegment())//
736         && !textAnnotationsToOpen.isEmpty()//
737         && !textAnnotationsToClose.isEmpty();
738     if (isMilestone) {
739       TextAnnotation lastToOpen = textAnnotationsToOpen.get(textAnnotationsToOpen.size() - 1);
740       TextAnnotation firstToClose = textAnnotationsToClose.get(0);
741       if (lastToOpen.equals(firstToClose)) {
742         textAnnotationsToOpen.remove(lastToOpen);
743         textGraphSegment.setMilestoneAnnotation(lastToOpen);
744         textAnnotationsToClose.remove(firstToClose);
745       }
746     }
747     textGraphSegment.setAnnotationsToOpen(textAnnotationsToOpen);
748     textGraphSegment.setAnnotationsToClose(textAnnotationsToClose);
749     return textGraphSegment;
750   }

760   private static final Comparator<TextAnnotation> BY_INCREASING_DEPTH = Comparator.comparing(TextAnnotation::getDepth);

762   private List<TextAnnotation> getTextAnnotations(Vertex textSegment, String edgeLabel, List<List<String>> orderedLayerDefinitions) {
763     // LOG.info("orderedLayerDefinitions:'{}'",orderedLayerDefinitions);
764     // LOG.info("textsegment:'{}'",(String) textSegment.value("text"));
765     List<Vertex> textAnnotationVertexList = StreamUtil.stream(textSegment.vertices(Direction.IN, edgeLabel))//
766         .filter(v -> v.label().equals(VertexLabels.TEXTANNOTATION)).collect(toList());
767     // LOG.info("textAnnotationVertexList.size={}",textAnnotationVertexList.size());
768     List<List<Vertex>> vertexListPerLayer = new ArrayList<>();
769     AtomicInteger relevantVertexCount = new AtomicInteger(0);
770     List<Vertex> otherVertexList = Lists.newArrayList(textAnnotationVertexList);
771 
772     // process tags defined in orderedLayerDefinitions
773     orderedLayerDefinitions.forEach(layerTags -> {
774       List<Vertex> vertexList = textAnnotationVertexList.stream()//
775           .filter(v -> layerTags.contains(v.value(TextAnnotation.Properties.name)))//
776           .collect(toList());
777       vertexListPerLayer.add(vertexList);
778       relevantVertexCount.set(relevantVertexCount.get() + vertexList.size());
779       otherVertexList.removeAll(vertexList);
780     });
781     // put all other tags in separate layers, grouped by depth
782     // LOG.info("otherVertexList.size={}",otherVertexList.size());
783     vertexListPerLayer.addAll(groupByDecreasingDepth(otherVertexList));
784     relevantVertexCount.set(relevantVertexCount.get() + otherVertexList.size());
785 
786     boolean useLayerOrder = relevantVertexCount.get() > 1;
787     Map<Vertex, Integer> overriddenDepth = new HashMap<>();
788     if (useLayerOrder) {
789       createPairs(vertexListPerLayer).stream()//
790           .filter(this::hasSameTextRange)//
791           .forEach(pair -> {
792             Vertex leftVertex = pair.getLeft();
793             Vertex rightVertex = pair.getRight();
794             Integer leftDepth = getDepth(leftVertex, overriddenDepth);
795             Integer rightDepth = getDepth(rightVertex, overriddenDepth);
796             boolean swapDepths = leftDepth < rightDepth;
797             if (swapDepths) {
798               overriddenDepth.put(leftVertex, rightDepth);
799               overriddenDepth.put(rightVertex, leftDepth);
800             }
801           });
802     }
803     Function<Vertex, TextAnnotation> toTextAnnotationWithOverriddenDepth = vertex -> {
804       TextAnnotation textAnnotation = toTextAnnotation(vertex);
805       if (overriddenDepth.containsKey(vertex)) {
806         textAnnotation.setDepth(overriddenDepth.get(vertex));
807       }
808       return textAnnotation;
809     };
810     return textAnnotationVertexList.stream()//
811         .map(toTextAnnotationWithOverriddenDepth)//
812         .sorted(BY_INCREASING_DEPTH)//
813         .collect(toList());
814   }

816   private Integer getDepth(Vertex leftVertex, Map<Vertex, Integer> overriddenDepth) {
817     return overriddenDepth.containsKey(leftVertex)//
818         ? overriddenDepth.get(leftVertex)//
819         : (Integer) leftVertex.value(TextAnnotation.Properties.depth);
820   }

822   List<List<Vertex>> groupByDecreasingDepth(List<Vertex> vertexList) {
823     Map<Integer, List<Vertex>> groupedByDepth = vertexList.stream()//
824         .collect(groupingBy(v -> (Integer) v.value(TextAnnotation.Properties.depth)));
825     return groupedByDepth.keySet().stream()//
826         .sorted((d0, d1) -> d1.compareTo(d0))//
827         .map(d -> groupedByDepth.get(d))//
828         .collect(toList());
829   }

831   List<Pair<Vertex, Vertex>> createPairs(List<List<Vertex>> vertexListPerLayer) {
832     List<Pair<Vertex, Vertex>> pairList = Lists.newArrayList();
833     for (int i = 0; i < vertexListPerLayer.size() - 1; i++) {
834       List<Vertex> vertexLayer1 = vertexListPerLayer.get(i);
835       for (int j = i + 1; j < vertexListPerLayer.size(); j++) {
836         List<Vertex> vertexLayer2 = vertexListPerLayer.get(j);
837         pairList.addAll(createPairs(vertexLayer1, vertexLayer2));
838       }
839     }
840     return pairList;
841   }

843   private List<Pair<Vertex, Vertex>> createPairs(List<Vertex> vertexLayer1, List<Vertex> vertexLayer2) {

845     vertexLayer1.forEach(v1 -> {
846       vertexLayer2.forEach(v2 -> {
847         list.add(Pair.of(v1, v2));
848       });
849     });
850     return list;

853   Boolean hasSameTextRange(Pair<Vertex, Vertex> vertexPair) {
854     Vertex left = vertexPair.getLeft();
855     Vertex leftFirstTextSegment = left.vertices(Direction.OUT, EdgeLabels.FIRST_TEXT_SEGMENT).next();
856     Vertex leftLastTextSegment = left.vertices(Direction.OUT, EdgeLabels.LAST_TEXT_SEGMENT).next();
857 
858     Vertex right = vertexPair.getRight();
859     Vertex rightFirstTextSegment = right.vertices(Direction.OUT, EdgeLabels.FIRST_TEXT_SEGMENT).next();
860     Vertex rightLastTextSegment = right.vertices(Direction.OUT, EdgeLabels.LAST_TEXT_SEGMENT).next();
861 
862     return leftFirstTextSegment.equals(rightFirstTextSegment) //
863         && leftLastTextSegment.equals(rightLastTextSegment);
864   }
