196   public Optional<? extends Accountable> dereference(IdentifiablePointer<? extends Accountable> pointer) {
197     Class<? extends Accountable> aClass = pointer.getIdentifiableClass();
198     UUID uuid = UUID.fromString(pointer.getIdentifier());
199     if (AlexandriaResource.class.equals(aClass)) {
200       return readResource(uuid);
201 
202     } else if (AlexandriaAnnotation.class.equals(aClass)) {
203       return readAnnotation(uuid);
204 
205     } else {
206       throw new RuntimeException("unexpected accountableClass: " + aClass.getName());
207     }
208   }

356   @Override
357   public boolean overlapsWithExisitingTextRangeAnnotationForResource(TextRangeAnnotation annotation, UUID resourceUUID) {
358     return storage.runInTransaction(() -> {
359       boolean overlaps = false;
360       Optional<ResourceVF> oResourceVF = storage.readVF(ResourceVF.class, resourceUUID);
361       if (oResourceVF.isPresent()) {
362         FramedGraphTraversal<ResourceVF, Vertex> traversal = oResourceVF.get().start()//
363             .in(nl.knaw.huygens.alexandria.storage.frames.TextRangeAnnotationVF.EdgeLabels.HAS_RESOURCE)//
364             .has("name", annotation.getName())//
365             .has("annotatorCode", annotation.getAnnotator());
366         Integer start1 = annotation.getPosition().getOffset();
367         Integer end1 = start1 + annotation.getPosition().getLength();
368         Predicate<Vertex> overlapsWithAnnotation = t -> {
369           Integer start2 = (Integer) t.property("offset").value();
370           Integer end2 = start2 + (Integer) t.property("length").value();
371           return start1 <= end2 && start2 <= end1;
372         };
373         overlaps = StreamUtil.parallelStream(traversal)//
374             .filter(overlapsWithAnnotation)//
375             .findAny()//
376             .isPresent();
377       }
378       return overlaps;
379     });
380   }

388   private AnnotationVF deprecateAnnotationVF(UUID annotationId, AlexandriaAnnotation updatedAnnotation) {
389     // check if there's an annotation with the given id
390     AnnotationVF oldAnnotationVF = storage.readVF(AnnotationVF.class, annotationId)//
391         .orElseThrow(annotationNotFound(annotationId));
392     if (oldAnnotationVF.isTentative()) {
393       throw incorrectStateException(annotationId, "tentative");
394     } else if (oldAnnotationVF.isDeleted()) {
395       throwBadRequest(annotationId, "deleted");
396     } else if (oldAnnotationVF.isDeprecated()) {
397       throwBadRequest(annotationId, "already deprecated");
398     }
399 
400     AlexandriaAnnotationBody newBody = updatedAnnotation.getBody();
401     Optional<AlexandriaAnnotationBody> optionalBody = findAnnotationBodyWithTypeAndValue(newBody.getType(), newBody.getValue());
402     AlexandriaAnnotationBody body;
403     if (optionalBody.isPresent()) {
404       body = optionalBody.get();
405     } else {
406       AnnotationBodyVF annotationBodyVF = frameAnnotationBody(newBody);
407       updateState(annotationBodyVF, AlexandriaState.CONFIRMED);
408       body = newBody;
409     }
410 
411     AlexandriaProvenance tmpProvenance = updatedAnnotation.getProvenance();
412     TentativeAlexandriaProvenance provenance = new TentativeAlexandriaProvenance(tmpProvenance.getWho(), tmpProvenance.getWhen(), tmpProvenance.getWhy());
413     AlexandriaAnnotation newAnnotation = new AlexandriaAnnotation(updatedAnnotation.getId(), body, provenance);
414     AnnotationVF newAnnotationVF = frameAnnotation(newAnnotation);
415 
416     AnnotationVF annotatedAnnotation = oldAnnotationVF.getAnnotatedAnnotation();
417     if (annotatedAnnotation != null) {
418       newAnnotationVF.setAnnotatedAnnotation(annotatedAnnotation);
419     } else {
420       ResourceVF annotatedResource = oldAnnotationVF.getAnnotatedResource();
421       newAnnotationVF.setAnnotatedResource(annotatedResource);
422     }
423     newAnnotationVF.setDeprecatedAnnotation(oldAnnotationVF);
424     newAnnotationVF.setRevision(oldAnnotationVF.getRevision() + 1);
425     updateState(newAnnotationVF, AlexandriaState.CONFIRMED);
426 
427     oldAnnotationVF.setAnnotatedAnnotation(null);
428     oldAnnotationVF.setAnnotatedResource(null);
429     oldAnnotationVF.setUuid(oldAnnotationVF.getUuid() + "." + oldAnnotationVF.getRevision());
430     updateState(oldAnnotationVF, AlexandriaState.DEPRECATED);
431     return newAnnotationVF;
432   }

460   @Override
461   public void deleteAnnotation(AlexandriaAnnotation annotation) {
462     storage.runInTransaction(() -> {
463       UUID uuid = annotation.getId();
464       AnnotationVF annotationVF = storage.readVF(AnnotationVF.class, uuid).get();
465       if (annotation.isTentative()) {
466         // remove from database
467 
468         AnnotationBodyVF body = annotationVF.getBody();
469         List<AnnotationVF> ofAnnotations = body.getOfAnnotationList();
470         if (ofAnnotations.size() == 1) {
471           String annotationBodyId = body.getUuid();
472           storage.removeVertexWithId(annotationBodyId);
473         }
474 
475         // remove has_body edge
476         annotationVF.setBody(null);
477 
478         // remove annotates edge
479         annotationVF.setAnnotatedAnnotation(null);
480         annotationVF.setAnnotatedResource(null);
481 
482         String annotationId = uuid.toString();
483         storage.removeVertexWithId(annotationId);
484 
485       } else {
486         // set state
487         updateState(annotationVF, AlexandriaState.DELETED);
488       }
489     });
490   }

530   public List<TextView> getTextViewsForResource(UUID resourceUUID) {
531     List<TextView> textViews = new ArrayList<>();
532     Set<String> viewNames = Sets.newHashSet();
533 
534     return storage.runInTransaction(() -> {
535       ResourceVF resourceVF = storage.readVF(ResourceVF.class, resourceUUID).get();
536       while (resourceVF != null) {
537         String serializedTextViews = resourceVF.getSerializedTextViewMap();
538         UUID uuid = UUID.fromString(resourceVF.getUuid());
539         try {
540           deserializeToTextViews(serializedTextViews).stream().filter(v -> !viewNames.contains(v.getName())).forEach((tv) -> {
541             tv.setTextViewDefiningResourceId(uuid);
542             textViews.add(tv);
543             viewNames.add(tv.getName());
544           });
545 
546         } catch (Exception e) {
547           e.printStackTrace();
548           throw new RuntimeException(e);
549         }
550         resourceVF = resourceVF.getParentResource();
551       }
552       return textViews;
553     });
554   }

919   private AlexandriaAnnotation deframeAnnotation(AnnotationVF annotationVF) {
920     TentativeAlexandriaProvenance provenance = deframeProvenance(annotationVF);
921     UUID uuid = getUUID(annotationVF);
922     AlexandriaAnnotationBody body = deframeAnnotationBody(annotationVF.getBody());
923     AlexandriaAnnotation annotation = new AlexandriaAnnotation(uuid, body, provenance);
924     if (annotationVF.getLocator() != null) {
925       try {
926         annotation.setLocator(new TextLocatorFactory(this).fromString(annotationVF.getLocator()));
927       } catch (TextLocatorParseException e) {
928         e.printStackTrace();
929       }
930     }
931     annotation.setState(AlexandriaState.valueOf(annotationVF.getState()));
932     annotation.setStateSince(Instant.ofEpochSecond(annotationVF.getStateSince()));
933     if (annotationVF.getRevision() == null) { // update old data
934       annotationVF.setRevision(0);
935     }
936     annotation.setRevision(annotationVF.getRevision());
937 
938     AnnotationVF annotatedAnnotation = annotationVF.getAnnotatedAnnotation();
939     if (annotatedAnnotation == null) {
940       ResourceVF annotatedResource = annotationVF.getAnnotatedResource();
941       if (annotatedResource != null) {
942         annotation.setAnnotatablePointer(new IdentifiablePointer<>(AlexandriaResource.class, annotatedResource.getUuid()));
943       }
944     } else {
945       annotation.setAnnotatablePointer(new IdentifiablePointer<>(AlexandriaAnnotation.class, annotatedAnnotation.getUuid()));
946     }
947     for (AnnotationVF avf : annotationVF.getAnnotatedBy()) {
948       AlexandriaAnnotation annotationAnnotation = deframeAnnotation(avf);
949       annotation.addAnnotation(annotationAnnotation);
950     }
951     return annotation;
952   }

1003   private List<Map<String, Object>> processQuery(AlexandriaQuery query) {
1004     ParsedAlexandriaQuery pQuery = alexandriaQueryParser.parse(query);
1005 
1006     Stream<Map<String, Object>> mapStream = pQuery.getResultStreamMapper().apply(storage);
1007 
1008     if (pQuery.isDistinct()) {
1009       mapStream = mapStream.distinct();
1010     }
1011     if (pQuery.doGrouping()) {
1012       Stream<Map<String, Object>> groupByStream = mapStream//
1013           .collect(groupingBy(pQuery::concatenateGroupByFieldsValues))//
1014           .values().stream()//
1015           .map(pQuery::collectListFieldValues);
1016       mapStream = groupByStream;
1017     }
1018     List<Map<String, Object>> results = mapStream//
1019         .collect(toList());
1020     return results;
1021   }
