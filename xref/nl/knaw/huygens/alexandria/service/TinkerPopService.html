406   private void removeTextAnnotationFromChain(TextRangeAnnotationVF oldVF) {
407     FramedGraphTraversal<TextRangeAnnotationVF, Vertex> traversal = oldVF.out(TextRangeAnnotationVF.EdgeLabels.HAS_TEXTANNOTATION);
408 
409     // remove the old textAnnotationVertex without breaking the chain.
410     if (traversal.hasNext()) {
411       Vertex textAnnotationVertex = traversal.next();
412       Vertex leftVertex = null; // in the annotation chain, the vertex to the left of this textAnnotationVertex; there is always a left vertex
413       String leftEdgeLabel = null;
414       Vertex rightVertex = null; // in the annotation chain, the vertex to the right of this textAnnotationVertex; there might not be a right vertex
415       // it might be the first in the chain, so it has an incoming FIRST_ANNOTATION edge
416       Iterator<Edge> incomingFirstAnnotationEdgeIterator = textAnnotationVertex.edges(Direction.IN, nl.knaw.huygens.alexandria.storage.EdgeLabels.FIRST_ANNOTATION);
417       if (incomingFirstAnnotationEdgeIterator.hasNext()) {
418         // in that case, remove the edge, and reconnect the chain.
419         Edge incomingEdge = incomingFirstAnnotationEdgeIterator.next();
420         leftVertex = incomingEdge.outVertex();
421         leftEdgeLabel = nl.knaw.huygens.alexandria.storage.EdgeLabels.FIRST_ANNOTATION;
422         incomingEdge.remove();
423 
424       } else {
425         // otherwise, it's a NEXT edge
426         Iterator<Edge> incomingNextEdgeIterator = textAnnotationVertex.edges(Direction.IN, nl.knaw.huygens.alexandria.storage.EdgeLabels.NEXT);
427         if (incomingNextEdgeIterator.hasNext()) {
428           Edge incomingNextEdge = incomingNextEdgeIterator.next();
429           leftVertex = incomingNextEdge.outVertex();
430           incomingNextEdge.remove();
431         }
432         leftEdgeLabel = nl.knaw.huygens.alexandria.storage.EdgeLabels.NEXT;
433       }
434 
435       Iterator<Edge> outgoingNextEdgeIterator = textAnnotationVertex.edges(Direction.OUT, nl.knaw.huygens.alexandria.storage.EdgeLabels.NEXT);
436       if (outgoingNextEdgeIterator.hasNext()) {
437         Edge outgoingNextEdge = outgoingNextEdgeIterator.next();
438         rightVertex = outgoingNextEdge.inVertex();
439         outgoingNextEdge.remove();
440       }
441 
442       if (rightVertex != null) {
443         leftVertex.addEdge(leftEdgeLabel, rightVertex);
444       }
445 
446       textAnnotationVertex.remove();
447     }
448   }

491   @Override
492   public boolean nonNestingOverlapWithExistingTextRangeAnnotationForResource(TextRangeAnnotation annotation, UUID resourceUUID) {
493     return storage.runInTransaction(() -> {
494       AtomicBoolean overlaps = new AtomicBoolean(false);
495       storage.readVF(ResourceVF.class, resourceUUID).ifPresent(resourceVF -> {
496         FramedGraphTraversal<ResourceVF, Vertex> traversal = resourceVF.start()//
497           .in(TextRangeAnnotationVF.EdgeLabels.HAS_RESOURCE)//
498           .has("name", annotation.getName())//
499           .has("annotatorCode", annotation.getAnnotator())//
500           ;
501         AbsolutePosition absolutePosition = annotation.getAbsolutePosition();
502         String uuid1 = annotation.getId().toString();
503         String xmlId1 = absolutePosition.getXmlId();
504         Integer start1 = absolutePosition.getOffset();
505         Integer end1 = start1 + absolutePosition.getLength();
506         Predicate<Vertex> nonNestingOverlapWithAnnotation = t -> {
507           String xmlId2 = (String) t.property("absoluteXmlId").value();
508           Integer start2 = (Integer) t.property("absoluteOffset").value();
509           Integer end2 = start2 + (Integer) t.property("absoluteLength").value();
510           return xmlId1.equals(xmlId2)//
511             && (start1 < end2 && start2 < end1) // annotation overlaps with existing annotation t
512             && !((start1 <= start2 && start2 <= end1 && end2 <= end1) && !(start1 == start2 && end1 == end2)) // existing annotation t nested in annotation
513             && !((start2 <= start1 && start1 <= end2 && end1 <= end2) && !(start1 == start2 && end1 == end2)) // annotation nested in exisiting annotation t
514             ;
515         };
516         Predicate<Vertex> hasDifferentUUID = t -> {
517           String uuid2 = (String) t.property("uuid").value();
518           return !uuid1.equals(uuid2);
519         };
520         overlaps.set(StreamUtil.stream(traversal)//
521           .filter(hasDifferentUUID)//
522           .filter(nonNestingOverlapWithAnnotation)//
523           .findAny()//
524           .isPresent()//
525         );
526 
527       });
528       return overlaps.get();
529     });
530   }

538   private AnnotationVF deprecateAnnotationVF(UUID annotationId, AlexandriaAnnotation updatedAnnotation) {
539     // check if there's an annotation with the given id
540     AnnotationVF oldAnnotationVF = storage.readVF(AnnotationVF.class, annotationId)//
541       .orElseThrow(annotationNotFound(annotationId));
542     if (oldAnnotationVF.isTentative()) {
543       throw incorrectStateException(annotationId, "tentative");
544     } else if (oldAnnotationVF.isDeleted()) {
545       throwBadRequest(annotationId, "deleted");
546     } else if (oldAnnotationVF.isDeprecated()) {
547       throwBadRequest(annotationId, "already deprecated");
548     }
549 
550     AlexandriaAnnotationBody newBody = updatedAnnotation.getBody();
551     Optional<AlexandriaAnnotationBody> optionalBody = findAnnotationBodyWithTypeAndValue(newBody.getType(), newBody.getValue());
552     AlexandriaAnnotationBody body;
553     if (optionalBody.isPresent()) {
554       body = optionalBody.get();
555     } else {
556       AnnotationBodyVF annotationBodyVF = frameAnnotationBody(newBody);
557       updateState(annotationBodyVF, AlexandriaState.CONFIRMED);
558       body = newBody;
559     }
560 
561     // update the uuid of the (to be) deprecated annotation, so the annotationuuid can be used for the new annotation
562     oldAnnotationVF.setUuid(oldAnnotationVF.getUuid() + "." + oldAnnotationVF.getRevision());
563 
564     AlexandriaProvenance tmpProvenance = updatedAnnotation.getProvenance();
565     TentativeAlexandriaProvenance provenance = new TentativeAlexandriaProvenance(tmpProvenance.getWho(), tmpProvenance.getWhen(), tmpProvenance.getWhy());
566     AlexandriaAnnotation newAnnotation = new AlexandriaAnnotation(updatedAnnotation.getId(), body, provenance);
567     AnnotationVF newAnnotationVF = frameAnnotation(newAnnotation);
568 
569     AnnotationVF annotatedAnnotation = oldAnnotationVF.getAnnotatedAnnotation();
570     if (annotatedAnnotation != null) {
571       newAnnotationVF.setAnnotatedAnnotation(annotatedAnnotation);
572     } else {
573       ResourceVF annotatedResource = oldAnnotationVF.getAnnotatedResource();
574       newAnnotationVF.setAnnotatedResource(annotatedResource);
575     }
576     newAnnotationVF.setDeprecatedAnnotation(oldAnnotationVF);
577     newAnnotationVF.setRevision(oldAnnotationVF.getRevision() + 1);
578     updateState(newAnnotationVF, AlexandriaState.CONFIRMED);
579 
580     oldAnnotationVF.setAnnotatedAnnotation(null);
581     oldAnnotationVF.setAnnotatedResource(null);
582     updateState(oldAnnotationVF, AlexandriaState.DEPRECATED);
583     return newAnnotationVF;
584   }
