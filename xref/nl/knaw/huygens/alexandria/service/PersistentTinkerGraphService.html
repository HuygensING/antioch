6   import javax.inject.Inject;
7   import javax.inject.Singleton;
8   
9   import nl.knaw.huygens.Log;
10  import nl.knaw.huygens.alexandria.config.AlexandriaConfiguration;
11  import nl.knaw.huygens.alexandria.endpoint.LocationBuilder;
12  
13  @Singleton
14  public class PersistentTinkerGraphService extends TinkerGraphService {
15    private static final String DUMPFILE = "alexandria.graphml";
16  
17    @Inject
18    public PersistentTinkerGraphService(AlexandriaConfiguration config, LocationBuilder locationBuilder) {
19      super(locationBuilder);
20      String dumpfile = config.getStorageDirectory() + "/" + DUMPFILE;
21      STORAGE.setDumpFile(dumpfile);
22      if (!STORAGE.supportsPersistence() && Files.exists(Paths.get(dumpfile))) {
23        Log.info("reading stored db from {}", dumpfile);
24        STORAGE.loadFromDisk(dumpfile);
25      }
26    }
27  }
