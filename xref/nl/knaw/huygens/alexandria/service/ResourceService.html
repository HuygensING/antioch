1   package nl.knaw.huygens.alexandria.service;
2   
3   import javax.ws.rs.core.Context;
4   import java.util.UUID;
5   
6   import nl.knaw.huygens.alexandria.external.IllegalResourceException;
7   import nl.knaw.huygens.alexandria.external.ResourceExistsException;
8   import nl.knaw.huygens.alexandria.external.ResourceStore;
9   import nl.knaw.huygens.alexandria.util.UUIDParser;
10  
11  public class ResourceService {
12    private final ResourceStore resourceStore;
13  
14    public ResourceService(@Context ResourceStore resourceStore) {
15      this.resourceStore = resourceStore;
16    }
17  
18    public UUID createResource(final String ref) {
19      final UUID uuid = UUID.randomUUID();
20      resourceStore.createResource(uuid, ref);
21      return uuid;
22    }
23  
24    public void createResource(final UUID uuid, final String ref) throws ResourceExistsException {
25      if (!resourceStore.createResource(uuid, ref)) {
26        throw new ResourceExistsException();
27      }
28    }
29  
30    public void createResource(final String id, final String ref)
31        throws IllegalResourceException, ResourceExistsException {
32      UUID uuid = UUIDParser.fromString(id).get().orElseThrow(IllegalResourceException::new);
33      createResource(uuid, ref);
34    }
35  
36    public String getResource(String id) throws IllegalResourceException {
37      UUID uuid = UUIDParser.fromString(id).get().orElseThrow(IllegalResourceException::new);
38      return resourceStore.getReference(uuid);
39    }
40  }
