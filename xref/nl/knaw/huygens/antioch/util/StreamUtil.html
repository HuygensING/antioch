23  import java.util.Iterator;
24  import java.util.stream.Stream;
25  import java.util.stream.StreamSupport;
26  
27  import org.w3c.dom.Node;
28  import org.w3c.dom.NodeList;
29  
30  public class StreamUtil {
31    public static <S> Stream<S> stream(Iterator<S> input) {
32      return stream(input, false);
33    }
34  
35    public static <S> Stream<S> parallelStream(Iterator<S> input) {
36      return stream(input, true);
37    }
38  
39    private static <S> Stream<S> stream(Iterator<S> input, boolean parallel) {
40      Iterable<S> it = () -> input;
41      return StreamSupport.stream(it.spliterator(), parallel);
42    }
43  
44    public static <S> Stream<S> stream(Iterable<S> input) {
45      return stream(input, false);
46    }
47  
48    public static <S> Stream<S> parallelStream(Iterable<S> input) {
49      return stream(input, true);
50    }
51  
52    private static <S> Stream<S> stream(Iterable<S> input, boolean parallel) {
53      return StreamSupport.stream(input.spliterator(), parallel);
54    }
55  
56    public static class NodeIterator implements Iterator<Node> {
57      private final NodeList nodelist;
58      private final int size;
59      private int i;
60  
61      NodeIterator(NodeList nodelist) {
62        this.nodelist = nodelist;
63        this.size = nodelist.getLength();
64        this.i = 0;
65      }
66  
67      @Override
68      public boolean hasNext() {
69        return i < size;
70      }
71  
72      @Override
73      public Node next() {
74        return nodelist.item(i++);
75      }
76    }
77  
78    public static Stream<Node> stream(NodeList nodelist) {
79      NodeIterator nodeIterator = new NodeIterator(nodelist);
80      return stream(nodeIterator);
81    }
82  
83  }
