65    @POST
66    @Consumes(MediaType.APPLICATION_JSON)
67    @ApiOperation("create a new annotation body")
68    public Response createAnnotationBody(final AnnotationBodyPrototype prototype) {
69      final AnnotationBodyCreationRequest request = requestBuilder.build(prototype);
70      request.execute(service);
71  
72      if (request.wasCreated()) {
73        return created(locationOf(prototype.getId().getValue()));
74      }
75  
76      return conflict();
77    }
