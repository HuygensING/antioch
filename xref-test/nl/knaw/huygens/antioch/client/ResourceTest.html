43    @Test
44    public void testAddResourceReturnsValidUUID() {
45      client.setAuthKey(AUTHKEY);
46      client.setAutoConfirm(false);
47      String resourceRef = "corpus";
48      ResourcePrototype resource = new ResourcePrototype().setRef(resourceRef);
49      RestResult<UUID> result = client.addResource(resource);
50      assertRequestSucceeded(result);
51      UUID resourceUuid = result.get();
52      Log.info("resourceUUID = {}", resourceUuid);
53      softly.assertThat(resourceUuid).isNotNull();
54  
55      // retrieve the resource
56      RestResult<ResourcePojo> result2 = client.getResource(resourceUuid);
57      assertRequestSucceeded(result2);
58      ResourcePojo resourcePojo = result2.get();
59      softly.assertThat(resourcePojo).isNotNull();
60      softly.assertThat(resourcePojo.getRef()).as("ref").isEqualTo("corpus");
61      softly.assertThat(resourcePojo.getState().getValue()).as("state").isEqualTo(AntiochState.TENTATIVE);
62  
63      // confirm the resource
64      RestResult<Void> result3 = client.confirmResource(resourceUuid);
65      assertRequestSucceeded(result3);
66  
67      // retrieve the resource again
68      RestResult<ResourcePojo> result4 = client.getResource(resourceUuid);
69      assertRequestSucceeded(result4);
70      ResourcePojo resourcePojo2 = result4.get();
71      softly.assertThat(resourcePojo2).isNotNull();
72      softly.assertThat(resourcePojo2.getRef()).as("ref").isEqualTo("corpus");
73      softly.assertThat(resourcePojo2.getState().getValue()).as("state").isEqualTo(AntiochState.CONFIRMED);
74    }

97      client.setAutoConfirm(true);
98  
99      UUID resourceUuid = createResource("corpus");
100     UUID subresourceUUID = UUID.randomUUID();
101     String ref = "/some/sub/path";
102     SubResourcePrototype subresource = new SubResourcePrototype().setSub(ref);
103     RestResult<Void> setResult1 = client.setSubResource(resourceUuid, subresourceUUID, subresource);
104     assertRequestSucceeded(setResult1);
105 
106     // retrieve the resource again
107     RestResult<SubResourcePojo> getResult1 = client.getSubResource(subresourceUUID);
108     assertRequestSucceeded(getResult1);
109     SubResourcePojo subResourcePojo1 = getResult1.get();
110     softly.assertThat(subResourcePojo1).isNotNull();
111     softly.assertThat(subResourcePojo1.getSub()).as("sub").isEqualTo(ref);
112     softly.assertThat(subResourcePojo1.getState().getValue()).as("state").isEqualTo(AntiochState.CONFIRMED);
113 
114     // update the subresource
115     String ref2 = "/some/other/sub/path";
116     SubResourcePrototype newSubresource = new SubResourcePrototype().setSub(ref2);
117     RestResult<Void> setResult2 = client.setSubResource(resourceUuid, subresourceUUID, newSubresource);
118     assertRequestSucceeded(setResult2);
119 
120     RestResult<SubResourcePojo> getResult2 = client.getSubResource(subresourceUUID);
121     assertRequestSucceeded(getResult2);
122     SubResourcePojo subResourcePojo2 = getResult2.get();
123     softly.assertThat(subResourcePojo2).isNotNull();
124     softly.assertThat(subResourcePojo2.getSub()).as("sub").isEqualTo(ref2);
125     softly.assertThat(subResourcePojo2.getState().getValue()).as("state").isEqualTo(AntiochState.CONFIRMED);

159   @Test
160   public void testGetAnnotationsFromResourceUUID() {
161     client.setAuthKey(AUTHKEY);
162     String ref = "resource4";
163     ResourcePrototype resource = new ResourcePrototype().setRef(ref).setProvenance(new ProvenancePojo().setWho("test4").setWhy("because test4"));
164     UUID resourceId = UUID.fromString("11111111-1111-1111-1111-111111111112");
165     RestResult<Void> result = client.setResource(resourceId, resource);
166     assertRequestSucceeded(result);
167 
168     // add annotations
169     RestResult<UUID> annotateResourceResult1 = client.annotateResource(resourceId, new AnnotationPrototype().setType("typeUno").setValue("valueUno"));
170     assertRequestSucceeded(annotateResourceResult1);
171     AnnotationPojo annotation1 = client.getAnnotation(annotateResourceResult1.get()).get();
172 
173     RestResult<UUID> annotateResourceResult2 = client.annotateResource(resourceId, new AnnotationPrototype().setType("typeDos").setValue("valueDos"));
174     assertRequestSucceeded(annotateResourceResult2);
175     AnnotationPojo annotation2 = client.getAnnotation(annotateResourceResult2.get()).get();
176 
177     // retrieve the resource
178     RestResult<AnnotationList> annotationsResult = client.getResourceAnnotations(resourceId);
179     assertRequestSucceeded(annotationsResult);
180     AnnotationList annotationList = annotationsResult.get();
181     softly.assertThat(annotationList).as("entity != null").isNotNull();
182     softly.assertThat(annotationList).hasSize(2);
183     softly.assertThat(annotationList).containsExactly(annotation2, annotation1);
184   }

186   @Test
187   public void testGetSubresourcesFromResourceUUID() {
188     client.setAuthKey(AUTHKEY);
189     String ref = "resource5";
190     ResourcePrototype resource = new ResourcePrototype().setRef(ref).setProvenance(new ProvenancePojo().setWho("test5").setWhy("because test5"));
191     UUID resourceId = UUID.fromString("11111111-1111-1111-1111-111111111113");
192     RestResult<Void> result = client.setResource(resourceId, resource);
193     assertRequestSucceeded(result);
194 
195     // add subresources
196     RestResult<UUID> subResourceResult1 = client.addSubResource(resourceId, new SubResourcePrototype().setSub("Uno"));
197     assertRequestSucceeded(subResourceResult1);
198     SubResourcePojo subresource1 = client.getSubResource(subResourceResult1.get()).get();
199 
200     RestResult<UUID> subResourceResult2 = client.addSubResource(resourceId, new SubResourcePrototype().setSub("Dos"));
201     assertRequestSucceeded(subResourceResult2);
202     SubResourcePojo subresource2 = client.getSubResource(subResourceResult2.get()).get();
203 
204     // retrieve the resource
205     RestResult<SubResourceList> subresourcesResult = client.getSubResources(resourceId);
206     assertRequestSucceeded(subresourcesResult);
207     SubResourceList subresourceList = subresourcesResult.get();
208     softly.assertThat(subresourceList).as("entity != null").isNotNull();
209     softly.assertThat(subresourceList).hasSize(2);
210     // sorted on sub, so Dos before Uno
211     softly.assertThat(subresourceList).containsExactly(subresource2, subresource1);
212   }
