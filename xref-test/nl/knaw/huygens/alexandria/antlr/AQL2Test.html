25  import static org.assertj.core.api.Assertions.assertThat;
26  
27  import java.util.List;
28  
29  import org.antlr.v4.runtime.ANTLRInputStream;
30  import org.antlr.v4.runtime.CharStream;
31  import org.antlr.v4.runtime.CommonTokenStream;
32  import org.antlr.v4.runtime.Token;
33  import org.antlr.v4.runtime.tree.ParseTree;
34  import org.junit.Test;
35  
36  import nl.knaw.huygens.Log;
37  
38  public class AQL2Test {
39  
40    @Test
41    public void testCorrectAQL2Statement() {
42      String command = "hello(\"World\")";
43      CharStream stream = new ANTLRInputStream(command);
44      AQL2Lexer lex = new AQL2Lexer(stream);
45      List<? extends Token> allTokens = lex.getAllTokens();
46      for (Token token : allTokens) {
47        Log.info("token: [{}] <<{}>>", lex.getRuleNames()[token.getType() - 1], token.getText());
48      }
49      lex.reset();
50  
51      CommonTokenStream tokens = new CommonTokenStream(lex);
52      AQL2Parser parser = new AQL2Parser(tokens);
53      parser.setBuildParseTree(true);
54      ParseTree tree = parser.root();
55      Log.info("tree={}", tree.toStringTree(parser));
56      assertThat(tree.getChildCount()).isEqualTo(4);
57  
58      int numberOfSyntaxErrors = parser.getNumberOfSyntaxErrors();
59      assertThat(numberOfSyntaxErrors).isEqualTo(0); // no syntax errors
60      Log.info("numberOfSyntaxErrors={}", numberOfSyntaxErrors);
61  
62      for (int i = 0; i < tree.getChildCount(); i++) {
63        Log.info("root.child={}", tree.getChild(i).getText());
64      }
65      assertThat(allTokens).hasSize(4);
66    }
67  
68  }
