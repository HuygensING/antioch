295   @Test
296   public void testHasSameTextRangePasses() throws Exception {
297     Vertex firstTextSegment = mock(Vertex.class);
298     List<Vertex> firstTextSegmentList = ImmutableList.of(firstTextSegment);
299 
300     Vertex lastTextSegment = mock(Vertex.class);
301     List<Vertex> lastTextSegmentList = ImmutableList.of(lastTextSegment);
302 
303     Vertex v1 = mockVertex("1");
304     when(v1.vertices(Direction.OUT, EdgeLabels.FIRST_TEXT_SEGMENT)).thenReturn(firstTextSegmentList.iterator());
305     when(v1.vertices(Direction.OUT, EdgeLabels.LAST_TEXT_SEGMENT)).thenReturn(lastTextSegmentList.iterator());
306 
307     Vertex va = mockVertex("a");
308     when(va.vertices(Direction.OUT, EdgeLabels.FIRST_TEXT_SEGMENT)).thenReturn(firstTextSegmentList.iterator());
309     when(va.vertices(Direction.OUT, EdgeLabels.LAST_TEXT_SEGMENT)).thenReturn(lastTextSegmentList.iterator());
310 
311     Pair<Vertex, Vertex> pair = Pair.of(v1, va);
312     assertThat(tgs.hasSameTextRange(pair)).isTrue();
313   }

335   private void assertExpectation(String xml, String expected, UUID resourceUUID, TextRangeAnnotation textRangeAnnotation) {
336     Log.info("xml={}", xml);
337     setAbsolutePosition(textRangeAnnotation);
338     String xmlOut = storage.runInTransaction(() -> {
339       createResourceWithText(resourceUUID, xml);
340       showTextAnnotationList(resourceUUID);
341       TextRangeAnnotationVF vf = storage.createVF(TextRangeAnnotationVF.class);
342       // dumpDot(resourceUUID);
343       tgs.updateTextAnnotationLink(vf, textRangeAnnotation, resourceUUID);
344       // dumpDot(resourceUUID);
345       showTextAnnotationList(resourceUUID);
346       return getXML(resourceUUID);
347     });
348 
349     softly.assertThat(xmlOut).isEqualTo(expected);
350   }

361   private String logTextAnnotation(TextAnnotation ta) {
362     String string = "<" + ta.getName() + " depth=" + ta.getDepth()
363         + ta.getAttributes()//
364             .entrySet()//
365             .stream()//
366             .map(kv -> " " + kv.getKey() + "='" + kv.getValue() + "'")//
367             .collect(joining(" "))
368         + ">";
369     Log.info(string);
370     return string;
371   }

378   private String getAnnotatedText(String xml) {
379     TextRangeAnnotation textRangeAnnotation = new TextRangeAnnotation()//
380         .setId(UUID.randomUUID())//
381         .setName("word")//
382         .setAnnotator("ed")//
383         .setPosition(new Position().setXmlId("p-1").setOffset(6).setLength(2));
384     setAbsolutePosition(textRangeAnnotation);
385     UUID resourceUUID = UUID.randomUUID();
386     Log.info("xml={}", xml);
387     return storage.runInTransaction(() -> {
388       createResourceWithText(resourceUUID, xml);
389       TextRangeAnnotationVF vf = storage.createVF(TextRangeAnnotationVF.class);
390 
391       // dumpDot(resourceUUID);
392       tgs.updateTextAnnotationLink(vf, textRangeAnnotation, resourceUUID);
393       dumpDot(resourceUUID);
394       return getXML(resourceUUID);
395     });
396   }

398   private void setAbsolutePosition(TextRangeAnnotation textRangeAnnotation) {
399     setAbsolutePosition(textRangeAnnotation, 1);
400   }

402   private void setAbsolutePosition(TextRangeAnnotation textRangeAnnotation, int length) {
403     Position position = textRangeAnnotation.getPosition();
404     AbsolutePosition absolutePosition = new AbsolutePosition()//
405         .setXmlId(position.getXmlId().get())//
406         .setOffset(position.getOffset().orElse(1))//
407         .setLength(position.getLength().orElse(length))//
408     ;
409     textRangeAnnotation.setAbsolutePosition(absolutePosition);
410   }

412   private String getXML(UUID resourceUUID) {
413     StreamingOutput outputStream = TextGraphUtil.streamXML(service, resourceUUID);
414     return TextGraphUtil.asString(outputStream);
415   }

417   private void createResourceWithText(UUID resourceUUID, String xml) {
418     TentativeAlexandriaProvenance provenance = new TentativeAlexandriaProvenance("who", Instant.now(), "why");
419     AlexandriaResource resource = new AlexandriaResource(resourceUUID, provenance);
420     service.createOrUpdateResource(resource);
421     ParseResult parseresult = TextGraphUtil.parse(xml);
422     service.storeTextGraph(resourceUUID, parseresult);
423   }

425   void dumpDb() {
426     Log.info("dumping server graph as graphML:");
427     try {
428       ByteArrayOutputStream outputstream = new ByteArrayOutputStream();
429       service.dumpToGraphML(outputstream);
430       outputstream.flush();
431       outputstream.close();
432       System.out.println(outputstream.toString());
433     } catch (IOException e) {
434       e.printStackTrace();
435     }
436     Log.info("dumping done");
437   }
