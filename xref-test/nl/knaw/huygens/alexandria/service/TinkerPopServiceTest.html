57      UUID id = UUID.randomUUID();

59      Instant when = Instant.now();
60      String why = "why";

62      AlexandriaResource resource = new AlexandriaResource(id, provenance);
63      resource.setCargo("reference");
64      // logGraphAsJson();
65      Log.info("resource={}", resource);
66      service.createOrUpdateResource(resource);
67      // logGraphAsJson();
68      Log.info("after createOrUpdate");
69      AlexandriaResource read = service.readResource(id).get();
70      Log.info("after read");
71  
72      // logGraphAsJson();

75      assertThat(read.getProvenance()).isEqualToComparingOnlyGivenFields(resource.getProvenance(), "who", "why", "when");
76    }

137   @Test
138   public void testDeprecateAnnotation() {
139     // given
140     UUID resourceId = UUID.fromString("00000000-0000-0000-0000-000000000000");
141     TentativeAlexandriaProvenance provenance = new TentativeAlexandriaProvenance("who", Instant.now(), "why");
142     AlexandriaResource resource = new AlexandriaResource(resourceId, provenance);
143     service.createOrUpdateResource(resource);
144 
145     UUID annotationBodyId = UUID.fromString("22222222-2222-2222-2222-222222222222");
146     TentativeAlexandriaProvenance provenance1 = new TentativeAlexandriaProvenance("who1", Instant.now(), "why1");
147     AlexandriaAnnotationBody body1 = service.createAnnotationBody(annotationBodyId, "type", "value", provenance1);
148 
149     UUID annotationId = UUID.fromString("11111111-1111-1111-1111-111111111111");
150     TentativeAlexandriaProvenance provenance2 = new TentativeAlexandriaProvenance("who2", Instant.now(), "why2");
151     AlexandriaAnnotation annotation = new AlexandriaAnnotation(annotationId, body1, provenance2);
152     service.annotateResourceWithAnnotation(resource, annotation);
153     service.confirmAnnotation(annotationId);
154     assertThat(annotation.getRevision()).isEqualTo(0);
155 
156     UUID annotationBodyId2 = UUID.fromString("22222222-2222-2222-2222-222222222223");
157     TentativeAlexandriaProvenance provenance3 = new TentativeAlexandriaProvenance("who3", Instant.now(), "why3");
158     AlexandriaAnnotationBody body2 = service.createAnnotationBody(annotationBodyId2, "type", "updated value", provenance3);
159     TentativeAlexandriaProvenance provenance4 = new TentativeAlexandriaProvenance("who4", Instant.now(), "why4");
160     AlexandriaAnnotation updatedAnnotation = new AlexandriaAnnotation(annotationId, body2, provenance4);
161 
162     // when
163     AlexandriaAnnotation newAnnotation = service.deprecateAnnotation(annotationId, updatedAnnotation);
164 
165     // then expect
166     assertThat(newAnnotation.getId()).isEqualTo(annotationId);
167     assertThat(newAnnotation.getState()).isEqualTo(AlexandriaState.CONFIRMED);
168     assertThat(newAnnotation.getBody().getType()).isEqualTo("type");
169     assertThat(newAnnotation.getBody().getValue()).isEqualTo("updated value");
170     assertThat(newAnnotation.getAnnotatablePointer().getIdentifier()).isEqualTo(resourceId.toString());
171     assertThat(newAnnotation.getProvenance().getWhen()).isEqualTo(provenance4.getWhen());
172     assertThat(newAnnotation.getProvenance().getWho()).isEqualTo(provenance4.getWho());
173     assertThat(newAnnotation.getProvenance().getWhy()).isEqualTo(provenance4.getWhy());
174     assertThat(newAnnotation.getRevision()).isEqualTo(1);
175   }
