77    @Test
78    public void testUpdatingATentativeResourceLeadsToConflictException() {
79      Response response = target(ROOTPATH).request().post(jsonEntity("{'resource':{'ref':'REF'}}"));
80      Log.debug("response={}", response);
81      assertThat(response.getLocation().toString()).contains("/resources/");
82      assertThat(response.getStatus()).isEqualTo(Status.CREATED.getStatusCode());
83  
84      UUID id = extractId(response);
85      Log.debug("uuid={}", id);
86      AlexandriaResource resource = getService().readResource(id).get();
87      assertThat(resource.getState()).isEqualTo(AlexandriaState.TENTATIVE);
88  
89      response = target(ROOTPATH).path(id.toString()).request().put(jsonEntity("{'resource':{'id':'" + id + "','ref':'REF'}}"));
90      assertThat(response.getStatus()).isEqualTo(Status.CONFLICT.getStatusCode());
91  
92      resource = getService().readResource(id).get();
93      assertThat(resource.getState()).isEqualTo(AlexandriaState.TENTATIVE);
94    }

117   @Test
118   public void testPostResourceAnnotationSetsStateToTentativeAndPutConfirmedToState() {
119     Response response = target(ROOTPATH).request().post(jsonEntity("{'resource':{'ref':'REF'}}"));
120     Log.debug("response={}", response);
121     assertThat(response.getLocation().toString()).contains("/resources/");
122     assertThat(response.getStatus()).isEqualTo(Status.CREATED.getStatusCode());
123 
124     UUID id = extractId(response);
125     Log.debug("resource uuid={}", id);
126     AlexandriaResource resource = getService().readResource(id).get();
127     assertThat(resource.getState()).isEqualTo(AlexandriaState.TENTATIVE);
128 
129     response = target(ROOTPATH).path(id.toString()).path("state").request().put(jsonEntity("{'state':'CONFIRMED'}"));
130     assertThat(response.getStatus()).isEqualTo(Status.NO_CONTENT.getStatusCode());
131 
132     resource = getService().readResource(id).get();
133     assertThat(resource.getState()).isEqualTo(AlexandriaState.CONFIRMED);
134 
135     response = target(ROOTPATH).path(id.toString()).path("annotations").request().post(jsonEntity("{'annotation':{'value':'bladiebla'}}"));
136     assertThat(response.getStatus()).isEqualTo(Status.CREATED.getStatusCode());
137 
138     UUID annotationId = extractId(response);
139     Log.debug("annotation uuid={}", annotationId);
140     AlexandriaAnnotation annotation = getService().readAnnotation(annotationId).get();
141     assertThat(annotation.getState()).isEqualTo(AlexandriaState.TENTATIVE);
142 
143     response = target("annotations").path(annotationId.toString()).path("state").request().put(jsonEntity("{'state':'CONFIRMED'}"));
144     assertThat(response.getStatus()).isEqualTo(Status.NO_CONTENT.getStatusCode());
145 
146     annotation = getService().readAnnotation(annotationId).get();
147     assertThat(annotation.getState()).isEqualTo(AlexandriaState.CONFIRMED);
148   }

150   private UUID extractId(Response response) {
151     String[] parts = response.getLocation().getPath().split("/");
152     return UUID.fromString(parts[parts.length - 1]);
153   }

155   @Test
156   public void testPutAndAnnotate() {
157     UUID uuid = UUID.randomUUID();
158     Entity<String> resourcePrototype = jsonEntity("{'resource':{'id':'" + uuid.toString() + "','ref':'Jan Klaassen'}}");
159     Response response = target(ROOTPATH).path(uuid.toString()).request().put(resourcePrototype);
160     Log.debug("response={}", response);
161     assertThat(response.getLocation().toString()).contains("/resources/");
162     assertThat(response.getStatus()).isEqualTo(Status.CREATED.getStatusCode());
163 
164     Entity<String> annotationPrototype = jsonEntity("{'annotation':{'type':'Tag','value':'Bookmark'}}");
165     Response annotateResponse = target(ROOTPATH).path(uuid.toString()).path("annotations").request().post(annotationPrototype);
166     Log.debug("response={}", annotateResponse);
167     assertThat(annotateResponse.getStatus()).isEqualTo(Status.CREATED.getStatusCode());
168     Log.debug("response.location={}", annotateResponse.getLocation());
169   }
