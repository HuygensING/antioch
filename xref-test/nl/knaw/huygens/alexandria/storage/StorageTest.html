18    @Test
19    public void testReadAfterCreateIsIdentity() {
20      UUID id = UUID.randomUUID();
21      String who = "who";
22      Instant when = Instant.now();
23      String why = "why";
24      TentativeAlexandriaProvenance provenance = new TentativeAlexandriaProvenance(who, when, why);
25      AlexandriaResource resource = new AlexandriaResource(id, provenance);
26      resource.setCargo("reference");
27      // logGraphAsJson();
28      Log.info("resource={}", resource);
29      storage.createOrUpdateResource(resource);
30      // logGraphAsJson();
31      Log.info("after createOrUpdate");
32      AlexandriaResource read = storage.readResource(id).get();
33      Log.info("after read");
34  
35      // logGraphAsJson();
36      Log.info("read={}", read);
37      assertThat(read).isEqualToComparingOnlyGivenFields(resource, "cargo", "id", "state");
38      assertThat(read.getProvenance()).isEqualToComparingOnlyGivenFields(resource.getProvenance(), "who", "why", "when");
39    }
