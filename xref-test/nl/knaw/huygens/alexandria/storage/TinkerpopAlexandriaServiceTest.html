56    // @Test
57    public void test() {
58      Graph g = TinkerGraph.open();
59      Log.info("graph features: {}", g.features());
60      TentativeAlexandriaProvenance provenance = new TentativeAlexandriaProvenance("who", Instant.ofEpochSecond(1000000), "why");
61      UUID uuid = new UUID(1l, 1l);
62      String ref = "ref";
63      AlexandriaState state = AlexandriaState.CONFIRMED;
64      boolean created = service.createOrUpdateResource(uuid, ref, provenance, state);
65      assertThat(created).isTrue();
66    }

70    public void testDereferenceWithExistingAnnotation() {
71      UUID id = UUID.randomUUID();
72      AlexandriaAnnotationBody body = mock(AlexandriaAnnotationBody.class);
73      TentativeAlexandriaProvenance provenance = new TentativeAlexandriaProvenance("who", Instant.now(), "why");
74      AnnotationVF annotationVF = mock(AnnotationVF.class);
75      when(annotationVF.getUuid()).thenReturn(id.toString());
76      when(mockStorage.readVF(AnnotationVF.class, id)).thenReturn(Optional.of(annotationVF));
77      IdentifiablePointer<AlexandriaAnnotation> ap = new IdentifiablePointer<>(AlexandriaAnnotation.class, id.toString());
78  
79      Optional<? extends Accountable> optional = service.dereference(ap);
80      assertThat(optional.isPresent());
81      AlexandriaAnnotation annotation = (AlexandriaAnnotation) optional.get();
82      assertThat(annotation.getId()).isEqualTo(id);
83    }
