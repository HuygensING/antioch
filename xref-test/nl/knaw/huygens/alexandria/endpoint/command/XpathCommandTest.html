25  import static org.assertj.core.api.Assertions.assertThat;
26  
27  import java.util.Collections;
28  import java.util.List;
29  
30  import org.junit.Test;
31  
32  import nl.knaw.huygens.alexandria.endpoint.command.XpathCommand.XPathResult;
33  import nl.knaw.huygens.alexandria.endpoint.command.XpathCommand.XPathResult.Type;
34  import nl.knaw.huygens.alexandria.test.AlexandriaTest;
35  
36  public class XpathCommandTest extends AlexandriaTest {
37    String EXAMPLE = singleQuotesToDouble("<root xmlns:bar='http://www.bar.org' xmlns:foo='http://www.foo.org/'>"//
38        + "<employees>"//
39        + "<employee id='1'>Johnny Dapp</employee>"//
40        + "<employee id='2'>Al Pacino</employee>"//
41        + "<employee id='3'>Robert De Niro</employee>"//
42        + "<employee id='4'>Kevin Spacey</employee>"//
43        + "<employee id='5'>Denzel Washington</employee>"//
44        + "</employees>"//
45        + "<foo:companies>"//
46        + "<foo:company id='6'>Tata Consultancy Services</foo:company>"//
47        + "<foo:company id='7'>Wipro</foo:company>"//
48        + "<foo:company id='8'>Infosys</foo:company>"//
49        + "<foo:company id='9'>Microsoft</foo:company>"//
50        + "<foo:company id='10'>IBM</foo:company>"//
51        + "<foo:company id='11'>Apple</foo:company>"//
52        + "<foo:company id='12'>Oracle</foo:company>"//
53        + "</foo:companies>"//
54        + "</root>");
55  
56    @Test
57    public void selectRootElement() throws Exception {
58      XPathResult result = XpathCommand.testXPath("/root", EXAMPLE);
59      assertThat(result.getType()).isEqualTo(Type.NODESET);
60      assertThat(result.getResult()).isInstanceOf(List.class);
61      assertThat((List<String>) result.getResult()).containsExactly(EXAMPLE);
62    }
63  
64    @Test
65    public void selectAllEmployeeElementsThatAreDirectChildrenOfTheEmployeesElement() throws Exception {
66      XPathResult result = XpathCommand.testXPath("/root/employees/employee", EXAMPLE);
67      softly.assertThat(result.getType()).isEqualTo(Type.NODESET);
68      softly.assertThat(result.getResult()).isInstanceOf(List.class);
69      softly.assertThat((List<String>) result.getResult()).containsExactly(//
70          "<employee id=\"1\">Johnny Dapp</employee>", //
71          "<employee id=\"2\">Al Pacino</employee>", //
72          "<employee id=\"3\">Robert De Niro</employee>", //
73          "<employee id=\"4\">Kevin Spacey</employee>", //
74          "<employee id=\"5\">Denzel Washington</employee>"//
75      );
76    }
77  
78    @Test
79    public void selectTheTextualValueOfTheFirstEmployeeElement() throws Exception {
80      XPathResult result = XpathCommand.testXPath("string(//employee[1])", EXAMPLE);
81      softly.assertThat(result.getType()).isEqualTo(Type.STRING);
82      softly.assertThat(result.getResult()).isEqualTo("Johnny Dapp");
83    }
84  
85    @Test
86    public void selectTheTextualValueOfTheSecondEmployeeElement() throws Exception {
87      XPathResult result = XpathCommand.testXPath("//employee[2]/text()", EXAMPLE);
88      softly.assertThat(result.getType()).isEqualTo(Type.NODESET);
89      softly.assertThat(result.getResult()).isEqualTo(Collections.singletonList("Al Pacino"));
90    }
91  
92    @Test
93    public void selectTheNumberOfEmployeeElements() throws Exception {
94      XPathResult result = XpathCommand.testXPath("count(//employee)", EXAMPLE);
95      softly.assertThat(result.getType()).isEqualTo(Type.NUMBER);
96      softly.assertThat(result.getResult()).isEqualTo(5.0);
97    }
98  
99    @Test
100   public void selectTheNumberOfCompanyElements() throws Exception {
101     XPathResult result = XpathCommand.testXPath("count(//foo:company)", EXAMPLE);
102     softly.assertThat(result.getType()).isEqualTo(Type.NUMBER);
103     softly.assertThat(result.getResult()).isEqualTo(7.0);
104   }
105 
106   @Test
107   public void selectTheLocalNameOfTheFirstCompanyElement() throws Exception {
108     XPathResult result = XpathCommand.testXPath("local-name(//foo:company[1])", EXAMPLE);
109     softly.assertThat(result.getType()).isEqualTo(Type.STRING);
110     softly.assertThat(result.getResult()).isEqualTo("company");
111   }
112 
113   @Test
114   public void selectTrue() throws Exception {
115     XPathResult result = XpathCommand.testXPath("true()", EXAMPLE);
116     softly.assertThat(result.getType()).isEqualTo(Type.BOOLEAN);
117     softly.assertThat(result.getResult()).isEqualTo(true);
118   }
119 
120   @Test
121   public void selectFalse() throws Exception {
122     XPathResult result = XpathCommand.testXPath("false()", EXAMPLE);
123     softly.assertThat(result.getType()).isEqualTo(Type.BOOLEAN);
124     softly.assertThat(result.getResult()).isEqualTo(false);
125   }
126 
127   @Test
128   public void floor() throws Exception {
129     XPathResult result = XpathCommand.testXPath("floor(3.1415926)", EXAMPLE);
130     softly.assertThat(result.getType()).isEqualTo(Type.NUMBER);
131     softly.assertThat(result.getResult()).isEqualTo(3.0);
132   }
133 
134   @Test
135   public void ceiling() throws Exception {
136     XPathResult result = XpathCommand.testXPath("ceiling(3.1415926)", EXAMPLE);
137     softly.assertThat(result.getType()).isEqualTo(Type.NUMBER);
138     softly.assertThat(result.getResult()).isEqualTo(4.0);
139   }
