17  public class SearchResultPageTest {
18    String baseURI = "http://example.org/search/1/page/";
19    static Map<String, Object> map = Maps.newHashMap();
20  
21    static {
22      map.put("what", "ever");
23    }
24  
25    List<Map<String, Object>> ten_results = ImmutableList.<Map<String, Object>> builder()//
26        .add(new HashMap<String, Object>(map)).add(new HashMap<String, Object>(map)).add(new HashMap<String, Object>(map)).add(new HashMap<String, Object>(map)).add(new HashMap<String, Object>(map))//
27        .add(new HashMap<String, Object>(map)).add(new HashMap<String, Object>(map)).add(new HashMap<String, Object>(map)).add(new HashMap<String, Object>(map)).add(new HashMap<String, Object>(map))//
28        .build();
29  
30    List<Map<String, Object>> five_results = ImmutableList.<Map<String, Object>> builder()//
31        .add(new HashMap<String, Object>(map)).add(new HashMap<String, Object>(map)).add(new HashMap<String, Object>(map)).add(new HashMap<String, Object>(map)).add(new HashMap<String, Object>(map))//
32        .build();
33  
34    @Test
35    public void testFirstPageOfSeveral() {
36      SearchResultPage srp = new SearchResultPage(baseURI, 1, false);
37      srp.setResults(ten_results);
38      assertThat(srp.getPreviousPage()).isNull();
39      assertThat(srp.getNextPage()).hasToString(baseURI + 2);
40      List<Map<String, Object>> records = srp.getRecords();
41      Log.info("records={}", records);
42      assertThat(records).hasSize(10);
43      assertThat(records.get(0)).containsEntry("what", "ever");
44      assertThat(records.get(0)).containsEntry("_resultNumber", 1);
45    }
46  
47    @Test
48    public void testSecondPageOfThree() {
49      SearchResultPage srp = new SearchResultPage(baseURI, 2, false);
50      srp.setResults(ten_results);
51      assertThat(srp.getPreviousPage()).hasToString(baseURI + 1);
52      assertThat(srp.getNextPage()).hasToString(baseURI + 3);
53      List<Map<String, Object>> records = srp.getRecords();

55      assertThat(records).hasSize(10);

60    @Test
61    public void testLastPageOfSeveral() {
62      SearchResultPage srp = new SearchResultPage(baseURI, 2, true);
63      srp.setResults(five_results);
64      assertThat(srp.getPreviousPage()).hasToString(baseURI + 1);
65      assertThat(srp.getNextPage()).isNull();
66      List<Map<String, Object>> records = srp.getRecords();
67      Log.info("records={}", records);
68      assertThat(records).hasSize(5);
69      assertThat(records.get(0)).containsEntry("what", "ever");
70      assertThat(records.get(0)).containsEntry("_resultNumber", 11);
71      assertThat(records.get(4)).containsEntry("_resultNumber", 15);
72    }
