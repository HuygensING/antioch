34  import nl.knaw.huygens.alexandria.api.EndpointPaths;
35  import nl.knaw.huygens.alexandria.api.model.search.AlexandriaQuery;
36  import nl.knaw.huygens.alexandria.config.AlexandriaConfiguration;
37  import nl.knaw.huygens.alexandria.config.MockConfiguration;
38  import nl.knaw.huygens.alexandria.endpoint.EndpointPathResolver;
39  import nl.knaw.huygens.alexandria.endpoint.LocationBuilder;
40  
41  public class SearchResultTest {
42    @Test
43    public void testTotalPages() {
44      AlexandriaConfiguration config = new MockConfiguration();
45      SearchResult sr = new SearchResult(new LocationBuilder(config, new EndpointPathResolver()));
46      sr.setQuery(new AlexandriaQuery().setPageSize(10));
47      Map<String, Object> map = ImmutableMap.of("what", "ever");
48      List<Map<String, Object>> results = Collections.nCopies(11, map);
49      sr.setResults(results);
50      UUID id = UUID.randomUUID();
51      sr.setId(id);
52      assertThat(sr.getTotalResults()).isEqualTo(11);
53      assertThat(sr.getTotalPages()).isEqualTo(2);
54      String expected = config.getBaseURI() + EndpointPaths.SEARCHES + "/" + id + "/" + EndpointPaths.RESULTPAGES + "/1";
55      assertThat(sr.getFirstResultPage()).hasToString(expected);
56    }
