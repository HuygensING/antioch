17  public class OmittedOrMatchingTypeTest {
18    private final UUIDParam uuidParam = new UUIDParam(UUID.randomUUID().toString());
19  
20    private AlexandriaService service;
21    private AlexandriaAnnotation annotation;
22    private OmittedOrMatchingType.Validator validator;
23    private AnnotationPrototype mockPrototype;
24  
25    @Before
26    public void given() {
27      final AlexandriaAnnotationBody annotationBody = mock(AlexandriaAnnotationBody.class);
28      when(annotationBody.getType()).thenReturn("some type");
29  
30      annotation = mock(AlexandriaAnnotation.class);
31      when(annotation.getBody()).thenReturn(annotationBody);
32  
33      service = mock(AlexandriaService.class);
34      when(service.readAnnotation(uuidParam.getValue())).thenReturn(Optional.of(annotation));
35  
36      validator = new OmittedOrMatchingType.Validator(service, uuidParam);
37  
38      mockPrototype = mock(AnnotationPrototype.class);
39    }
40  
41    @Test
42    public void testOmittingTypeInPrototypeShouldClassifyAsValid() {
43      when(mockPrototype.getType()).thenReturn(Optional.empty());
44  
45      assertTrue("Omitting type in prototype should be valid", validator.isValid(mockPrototype, null));
46    }
47  
48    @Test
49    public void testEqualTypeInPrototypeShouldClassifyAsValid() {
50      when(mockPrototype.getType()).thenReturn(Optional.of("some type"));
51  
52      assertTrue("Equal type in prototype should be valid", validator.isValid(mockPrototype, null));
53    }
54  
55    @Test
56    public void testDifferentTypeInPrototypeShouldClassifyAsInvalid() {
57      when(mockPrototype.getType()).thenReturn(Optional.of("another type"));
58  
59      assertTrue("Different type in prototype should be invalid", !validator.isValid(mockPrototype, null));
60    }
61  
62  }
