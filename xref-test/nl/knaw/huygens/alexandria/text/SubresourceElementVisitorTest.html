25  import java.util.List;
26  import java.util.Map;
27  
28  import org.junit.Test;
29  
30  import com.google.common.collect.ImmutableList;
31  
32  public class SubresourceElementVisitorTest extends AlexandriaVisitorTest {
33  
34    @Test
35    public void testVisitor() {
36      // given
37      String xml = singleQuotesToDouble("<text xml:id='text-1'>"//
38          + "<p xml:id='p-1'>foo<note n='1'>probably by <persName>someone</persName></note></p>"//
39          + "<p xml:id='p-2'>bar<comment n='2'>most likely by <persName>someone else</persName></comment></p>"//
40          + "</text>");
41      String expectedBase = singleQuotesToDouble("<text xml:id='text-1'>"//
42          + "<p xml:id='p-1'>foo<alexandria_subtextplaceholder xml:id='alexandria:subtext-1'/></p>"//
43          + "<p xml:id='p-2'>bar<alexandria_subtextplaceholder xml:id='alexandria:subtext-2'/></p>"//
44          + "</text>");
45      String expectedSubText1 = singleQuotesToDouble("<note n='1'>probably by <persName>someone</persName></note>");
46      String expectedSubText2 = singleQuotesToDouble("<comment n='2'>most likely by <persName>someone else</persName></comment>");
47      List<String> subresourceElements = ImmutableList.of("note", "comment");
48      SEVContext context = new SEVContext();
49      SubresourceElementVisitor subresourceVisitor = new SubresourceElementVisitor(context, subresourceElements);
50  
51      // when
52      visitXml(xml, subresourceVisitor);
53  
54      // expect
55      softly.assertThat(context.getBaseText()).isEqualTo(expectedBase);
56      Map<String, String> subresourceTexts = context.getSubresourceTexts();
57      softly.assertThat(subresourceTexts).hasSize(2);
58      softly.assertThat(subresourceTexts).containsEntry("alexandria:subtext-1", expectedSubText1);
59      softly.assertThat(subresourceTexts).containsEntry("alexandria:subtext-2", expectedSubText2);
60    }
61  
62    @Test
63    public void testVisitorWithNestedSubresourceElements() {
64      // given
65      String xml = singleQuotesToDouble("<text xml:id='text-1'>"//
66          + "<p xml:id='p-1'>foo<note n='1'>"//
67          + "probably by <persName>someone</persName>"//
68          + "<note n='2'>most likely by <persName>someone else</persName></note>"//
69          + "</note></p>"//
70          + "</text>");
71      String expectedBase = singleQuotesToDouble("<text xml:id='text-1'>"//
72          + "<p xml:id='p-1'>foo<alexandria_subtextplaceholder xml:id='alexandria:subtext-1'/></p>"//
73          + "</text>");
74      String expectedSubText = singleQuotesToDouble("<note n='1'>"//
75          + "probably by <persName>someone</persName>"//
76          + "<note n='2'>most likely by <persName>someone else</persName></note>"//
77          + "</note>");
78      List<String> subresourceElements = ImmutableList.of("note");
79      SEVContext context = new SEVContext();
80      SubresourceElementVisitor subresourceVisitor = new SubresourceElementVisitor(context, subresourceElements);
81  
82      // when
83      visitXml(xml, subresourceVisitor);
84  
85      // expect
86      softly.assertThat(context.getBaseText()).isEqualTo(expectedBase);
87      Map<String, String> subresourceTexts = context.getSubresourceTexts();
88      softly.assertThat(subresourceTexts).hasSize(1);
89      softly.assertThat(subresourceTexts).containsEntry("alexandria:subtext-1", expectedSubText);
90    }
