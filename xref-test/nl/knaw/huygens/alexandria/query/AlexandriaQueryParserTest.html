81    public void testUserStory4a() {
82      String userId = "USERID";
83      AlexandriaQuery aQuery = new AlexandriaQuery();
84      aQuery.setFind("annotation");
85      aQuery.setWhere("type.eq(\"Tag\"),who.eq(\"" + userId + "\"),state.eq(\"CONFIRMED\")");
86      aQuery.setSort("when");
87      aQuery.setFields("when,value,resource.id,subresource.id");
88      ParsedAlexandriaQuery paq = AlexandriaQueryParser.parse(aQuery);
89  
90      AnnotationVF passAnnotation = mock(AnnotationVF.class);
91      AnnotationBodyVF passBody = mock(AnnotationBodyVF.class);
92      when(passBody.getType()).thenReturn("Tag");
93      // .put("resource.id", AnnotationVF::getResourceId)//
94      // .put("subresource.id", AnnotationVF::getSubResourceId)//
95  
96      when(passAnnotation.getUuid()).thenReturn("uuid");
97      when(passAnnotation.getBody()).thenReturn(passBody);
98      when(passAnnotation.getType()).thenReturn("Tag");
99      when(passAnnotation.getValue()).thenReturn("Value");
100     when(passAnnotation.getProvenanceWhen()).thenReturn("1");
101     when(passAnnotation.getProvenanceWho()).thenReturn(userId);
102     when(passAnnotation.getProvenanceWhy()).thenReturn("because");
103     when(passAnnotation.getResourceId()).thenReturn("resourceId");
104     when(passAnnotation.getSubResourceId()).thenReturn("subresourceId");
105     when(passAnnotation.getState()).thenReturn("CONFIRMED");
106 
107     AnnotationVF failAnnotation = mock(AnnotationVF.class);
108     when(failAnnotation.getBody()).thenReturn(passBody);
109     when(failAnnotation.getProvenanceWhen()).thenReturn("2");
110     when(failAnnotation.getProvenanceWho()).thenReturn(userId);
111     when(failAnnotation.getState()).thenReturn("TENTATIVE");
112 
113     // find: test VFClass
114     assertThat(paq.getVFClass()).isEqualTo(AnnotationVF.class);
115 
116     // test predicate

123     //

126     // assertThat(predicate.test(fail)).isFalse();
127 
128     // sort: test resultComparator
129     Comparator<AnnotationVF> resultComparator = paq.getResultComparator();
130     int compare1 = resultComparator.compare(passAnnotation, failAnnotation);
131     assertThat(compare1).isEqualTo(-1);
132     int compare2 = resultComparator.compare(failAnnotation, passAnnotation);
