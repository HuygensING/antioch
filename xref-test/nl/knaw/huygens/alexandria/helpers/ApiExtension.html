1   package nl.knaw.huygens.alexandria.helpers;
2   
3   import nu.xom.Attribute;
4   import nu.xom.Document;
5   import nu.xom.Element;
6   import org.concordion.api.Resource;
7   import org.concordion.api.extension.ConcordionExtender;
8   import org.concordion.api.extension.ConcordionExtension;
9   import org.concordion.api.listener.DocumentParsingListener;
10  import org.concordion.internal.listener.AssertResultRenderer;
11  
12  public class ApiExtension implements ConcordionExtension {
13    @Override
14    public void addTo(ConcordionExtender concordionExtender) {
15      registerIncludesJsonCommand(concordionExtender);
16      registerLinkedCSS(concordionExtender);
17      addHeader(concordionExtender);
18    }
19  
20    private void registerIncludesJsonCommand(ConcordionExtender concordionExtender) {
21      IncludesJsonCommand includesJson = new IncludesJsonCommand();
22      includesJson.addAssertEqualsListener(new AssertResultRenderer());
23      concordionExtender.withCommand("https://alexandria.huygens.knaw.nl", "includesJson", includesJson);
24    }
25  
26    private void registerLinkedCSS(ConcordionExtender concordionExtender) {
27      concordionExtender.withLinkedCSS("/concordion.css", new Resource("/concordion.css"));
28    }
29  
30    private void addHeader(ConcordionExtender concordionExtender) {
31      concordionExtender.withDocumentParsingListener(new DocumentParsingListener() {
32        @Override
33        public void beforeParsing(Document document) {
34          Element header = new Element("div");
35          header.addAttribute(new Attribute("class", "header"));
36          header.appendChild(title("Alexandria Text Annotation Repository"));
37          header.appendChild(subtitle("REST API Specification"));
38          Element body = (Element) document.query("//body").get(0);
39          body.insertChild(header, 0);
40        }
41  
42        private Element title(String value) {
43          Element title = new Element("h1");
44          title.appendChild(value);
45          return title;
46        }
47  
48        private Element subtitle(String value) {
49          Element title = new Element("p");
50          title.appendChild(value);
51          return title;
52        }
53      });
54    }
55  }
