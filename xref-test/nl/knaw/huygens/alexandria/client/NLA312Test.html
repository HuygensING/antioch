220   private UUID createResourceWithText(String xml) {
221     String resourceRef = "test";
222     UUID resourceUUID = createResource(resourceRef);
223     TextImportStatus textGraphImportStatus = setResourceText(resourceUUID, xml);
224     URI expectedURI = URI.create("http://localhost:2015/resources/" + resourceUUID + "/text/xml");
225     assertThat(textGraphImportStatus.getTextURI()).isEqualTo(expectedURI);
226     return resourceUUID;
227   }

229   protected UUID createResource(String resourceRef) {
230     ResourcePrototype resource = new ResourcePrototype().setRef(resourceRef);
231     UUID resourceUuid = UUID.randomUUID();
232     client.setResource(resourceUuid, resource);
233     return resourceUuid;
234   }
235 
236   protected TextImportStatus setResourceText(UUID resourceUuid, String xml) {
237     TextImportStatus textImportStatus = client.setResourceTextSynchronously(resourceUuid, xml);
238     return textImportStatus;
239   }
