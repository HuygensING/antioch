122   void assertRequestSucceeded(RestResult<?> result) {
123     assertThat(result).isNotNull();
124     assertThat(result.hasFailed())//
125         .as("Request went OK")//
126         .withFailMessage("request failed: %s", result.getFailureCause().orElse("something you should never see"))//
127         .isFalse();
128   }

130   protected UUID createResource(String resourceRef) {
131     ResourcePrototype resource = new ResourcePrototype().setRef(resourceRef);
132     UUID resourceUuid = UUID.randomUUID();
133     RestResult<Void> result = client.setResource(resourceUuid, resource);
134     assertRequestSucceeded(result);
135     return resourceUuid;
136   }

138   protected UUID createSubResource(UUID resourceUuid, String ref) {
139     SubResourcePrototype subresource = new SubResourcePrototype().setSub(ref);

146   protected TextImportStatus setResourceText(UUID resourceUuid, String xml) {
147     RestResult<URI> result = client.setResourceText(resourceUuid, xml);
148     assertThat(result).isNotNull();
149     assertThat(result.hasFailed()).isFalse();
150 
151     TextImportStatus textGraphImportStatus = null;
152     boolean goOn = true;
153     while (goOn) {
154       RestResult<TextImportStatus> result2 = client.getTextImportStatus(resourceUuid);
155       assertThat(result2.hasFailed()).isFalse();
156       textGraphImportStatus = result2.get();
157       goOn = !textGraphImportStatus.isDone();
158     }
159     return textGraphImportStatus;

185   void dumpDb() {
186     Log.info("dumping server graph as graphSON:");
187     try {
188       ByteArrayOutputStream outputstream = new ByteArrayOutputStream();
189       tinkerpopService.dumpToGraphSON(outputstream);
190       outputstream.flush();
191       outputstream.close();
192       String[] lines = outputstream.toString("UTF8").split("\n");
193       ObjectMapper mapper = new ObjectMapper();
194       for (String json : lines) {
195         Object object = mapper.readValue(json, Object.class);
196         System.out.println(mapper.writerWithDefaultPrettyPrinter().writeValueAsString(object));
197       }
198     } catch (IOException e) {
199       // TODO Auto-generated catch block
200       e.printStackTrace();
201     }
202     Log.info("dumping done");
203   }

220   static String prettyFormat(String input, int indent) {
221     try {
222       Source xmlInput = new StreamSource(new StringReader(input));
223       StringWriter stringWriter = new StringWriter();
224       StreamResult xmlOutput = new StreamResult(stringWriter);
225       TransformerFactory transformerFactory = TransformerFactory.newInstance();
226       // transformerFactory.setAttribute("indent-number", indent);
227       Transformer transformer = transformerFactory.newTransformer();
228       transformer.setOutputProperty(OutputKeys.INDENT, "yes");
229       transformer.transform(xmlInput, xmlOutput);
230       return xmlOutput.getWriter().toString();
231     } catch (Exception e) {
232       throw new RuntimeException(e); // simple exception handling, please review it
233     }
234   }
