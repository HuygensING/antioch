36  import nl.knaw.huygens.alexandria.api.model.CommandResponse;
37  import nl.knaw.huygens.alexandria.api.model.Commands;

39  public class CommandTest extends AlexandriaClientTest {
40  
41    @Before
42    public void before() {
43      client.setAuthKey(AUTHKEY);
44      client.setAutoConfirm(true);
45    }
46  
47    @Test
48    public void testAddUniqueIdCommmandWorks() {
49      UUID resourceUuid = createResource("xml");
50      String xml = "<text><p>Alinea 1</p><p>Alinea 2</p></text>";
51      String expectedXml = singleQuotesToDouble("<text xml:id='text-1'><p xml:id='p-1'>Alinea 1</p><p xml:id='p-2'>Alinea 2</p></text>");
52      setResourceText(resourceUuid, xml);
53      Map<String, Object> parameters = ImmutableMap.<String, Object> builder()//
54          .put("resourceIds", ImmutableList.of(resourceUuid))//
55          .put("elements", ImmutableList.of("text", "p"))//
56          .build();
57      RestResult<CommandResponse> result = client.addCommand(Commands.ADD_UNIQUE_ID, parameters);
58      assertRequestSucceeded(result);
59      CommandResponse commandResponse = result.get();
60      assertThat(commandResponse.success()).isTrue();
61  
62      RestResult<String> xmlReadResult = client.getTextAsString(resourceUuid);
63      assertRequestSucceeded(xmlReadResult);
64      String xml2 = xmlReadResult.get();
65      assertThat(xml2).isEqualTo(expectedXml);
66    }
