41    public void before() {
42      client.setAuthKey(AUTHKEY);

45    @Test
46    public void testSetTextRangeAnnotation() {
47      String xml = singleQuotesToDouble("<text><p xml:id='p-1'>This is a simple paragraph.</p></text>");
48      UUID resourceUUID = createResourceWithText(xml);
49      RestResult<URI> result = client.setAnnotator(resourceUUID, "ed", new Annotator().setCode("ed").setDescription("Eddy Wally"));
50      assertRequestSucceeded(result);
51  
52      UUID annotationUUID = UUID.randomUUID();
53      Position position = new Position()//
54          .setXmlId("p-1")//
55          .setOffset(6)//
56          .setLength(2);
57      TextRangeAnnotation textRangeAnnotation0 = new TextRangeAnnotation()//
58          .setId(annotationUUID)//
59          .setName("word")//
60          .setAnnotator("ed")//
61          .setPosition(position);
62      RestResult<TextRangeAnnotationInfo> putResult = client.setResourceTextRangeAnnotation(resourceUUID, textRangeAnnotation0);
63      assertRequestSucceeded(putResult);
64      TextRangeAnnotationInfo info = putResult.get();
65      assertThat(info.getAnnotates()).isEqualTo("is");
66      dumpDb();
67  
68      RestResult<TextRangeAnnotation> getResult = client.getResourceTextRangeAnnotation(resourceUUID, annotationUUID);
69      assertRequestSucceeded(getResult);
70      TextRangeAnnotation textRangeAnnotation1 = getResult.get();
71      assertThat(textRangeAnnotation1).isEqualToComparingFieldByFieldRecursively(textRangeAnnotation0);
72    }
73  
74    @Test
75    public void testSetTextRangeAnnotationWithInvalidXmlId() {
76      String xml = singleQuotesToDouble("<text><p xml:id='p-1'>This is a simple paragraph.</p></text>");
77      UUID resourceUUID = createResourceWithText(xml);
78      RestResult<URI> result = client.setAnnotator(resourceUUID, "ed", new Annotator().setCode("ed").setDescription("Eddy Wally"));
79      assertRequestSucceeded(result);
80  
81      UUID uuid = UUID.randomUUID();
82      Position position = new Position()//
83          .setXmlId("undefined")//
84          .setOffset(6)//
85          .setLength(2);
86      TextRangeAnnotation textAnnotation = new TextRangeAnnotation()//
87          .setId(uuid)//
88          .setName("word")//
89          .setAnnotator("ed")//
90          .setPosition(position);
91      RestResult<TextRangeAnnotationInfo> putResult = client.setResourceTextRangeAnnotation(resourceUUID, textAnnotation);
92      assertThat(putResult.hasFailed()).isTrue();
93      assertThat(putResult.getErrorMessage().get()).isEqualTo("The text does not contain an element with the specified xml:id.");
94    }
95  
96    private UUID createResourceWithText(String xml) {
97      String resourceRef = "test";
98      UUID resourceUUID = createResource(resourceRef);
99      TextImportStatus textGraphImportStatus = setResourceText(resourceUUID, xml);
100     URI expectedURI = URI.create("http://localhost:2016/resources/" + resourceUUID + "/text/xml");
101     assertThat(textGraphImportStatus.getTextURI()).isEqualTo(expectedURI);
102     return resourceUUID;
103   }
