50    @Test
51    public void testAddUniqueIdCommmandWorks() {
52      UUID resourceUuid = createResource("xml");
53      String xml = "<text><p>Alinea 1</p><p>Alinea 2</p></text>";
54      String expectedXml = singleQuotesToDouble("<text xml:id='text-1'><p xml:id='p-1'>Alinea 1</p><p xml:id='p-2'>Alinea 2</p></text>");
55      setResourceText(resourceUuid, xml);
56      Map<String, Object> parameters = ImmutableMap.<String, Object> builder()//
57          .put("resourceIds", ImmutableList.of(resourceUuid))//
58          .put("elements", ImmutableList.of("text", "p"))//
59          .build();
60      RestResult<CommandResponse> result = client.doCommand(Commands.ADD_UNIQUE_ID, parameters);
61      assertRequestSucceeded(result);
62      CommandResponse commandResponse = result.get();
63      assertThat(commandResponse.success()).isTrue();
64  
65      RestResult<String> xmlReadResult = client.getTextAsString(resourceUuid);
66      assertRequestSucceeded(xmlReadResult);
67      String xml2 = xmlReadResult.get();
68      assertThat(xml2).isEqualTo(expectedXml);
69    }

71    @Test
72    public void testXpathCommmandWorks() {
73      UUID resourceUuid = createResource("xml");
74      String xml = "<text><p>Alinea 1</p><p>Alinea 2</p></text>";
75      String expected = "Alinea 2";
76      setResourceText(resourceUuid, xml);
77      Map<String, Object> parameters = ImmutableMap.<String, Object> builder()//
78          .put("resourceIds", ImmutableList.of(resourceUuid))//
79          .put("xpath", "string(/text/p[2])")//
80          .build();
81      RestResult<CommandResponse> result = client.doCommand(Commands.XPATH, parameters);
82      assertRequestSucceeded(result);
83      CommandResponse commandResponse = result.get();
84      assertThat(commandResponse.success()).isTrue();
85      Map<String, Map<String, String>> result2 = (Map<String, Map<String, String>>) commandResponse.getResult();
86      Map<String, String> map = result2.get(resourceUuid.toString());
87      XPathResult returnedXPathResult = new XPathResult(XPathResult.Type.valueOf(map.get("type")), map.get("result"));
88      XPathResult expectedXPathResult = new XPathResult(XPathResult.Type.STRING, expected);
89      assertThat(returnedXPathResult).isEqualToComparingFieldByField(expectedXPathResult);
90    }
