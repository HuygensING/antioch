100   public AlexandriaClient(final URI alexandriaURI, SSLContext sslContext) {
101     this.alexandriaURI = alexandriaURI;
102     final ObjectMapper objectMapper = new ObjectMapper()//
103         .registerModule(new Jdk8Module())//
104         .registerModule(new JavaTimeModule());
105 
106     final JacksonJaxbJsonProvider jacksonProvider = new JacksonJaxbJsonProvider();
107     jacksonProvider.setMapper(objectMapper);
108 
109     PoolingHttpClientConnectionManager cm = new PoolingHttpClientConnectionManager();
110     cm.setMaxTotal(50);
111     cm.setDefaultMaxPerRoute(50);
112 
113     ApacheConnectorProvider connectorProvider = new ApacheConnectorProvider();
114     ClientConfig clientConfig = new ClientConfig(jacksonProvider)//
115         .connectorProvider(connectorProvider)//
116         .property(ApacheClientProperties.CONNECTION_MANAGER, cm)//
117         .property(ClientProperties.CONNECT_TIMEOUT, 600)//
118         .property(ClientProperties.READ_TIMEOUT, 600);
119 
120     if (sslContext == null) {
121       if ("https".equals(alexandriaURI.getScheme())) {
122         throw new RuntimeException("SSL connections need an SSLContext, use: new AlexandriaClient(uri, sslContext) instead.");
123       }
124       client = ClientBuilder.newClient(clientConfig);
125 
126     } else {
127       client = ClientBuilder.newBuilder()//
128           .sslContext(sslContext)//
129           .withConfig(clientConfig)//
130           .build();
131     }
132     rootTarget = client.target(alexandriaURI);
133   }

441   private WebTarget addViewParameters(final Map<String, String> viewParameters, WebTarget path) {
442     Set<Map.Entry<String, String>> entries = viewParameters.entrySet();
443     for (Map.Entry<String, String> entry : entries) {
444       String k = entry.getKey();
445       String v = entry.getValue();
446       path = path.queryParam("view." + k, v);
447     }
448     return path;
449   }

546     final Entity<AnnotationPrototype> entity = Entity.json(annotationPrototype);
547     final WebTarget path = rootTarget//
548         .path(annotatablePath)//
549         .path(annotatableUuid.toString())//
550         .path(EndpointPaths.ANNOTATIONS);
551     final Supplier<Response> responseSupplier = authorizedPost(path, entity);
552 
553     final RestRequester<UUID> requester = RestRequester.withResponseSupplier(responseSupplier);
554     final RestResult<UUID> annotateResult = requester//
555         .onStatus(Status.CREATED, this::uuidFromLocationHeader)//
556         .getResult();
557     if (autoConfirm && !annotateResult.hasFailed()) {
558       confirmAnnotation(annotateResult.get());
559     }
560     return annotateResult;
561   }

675     final String location = response.getHeaderString("Location");

678     CommandResponse commandResponse = new CommandResponse();

747   private Function<Response, RestResult<Void>> voidRestResult() {
748     return (response) -> {
749       response.bufferEntity(); // to notify connectors, such as the ApacheConnector, that the entity has been "consumed" and that it should release the current connection back into the Apache
750       // ConnectionManager pool (if being used). https://java.net/jira/browse/JERSEY-3149
751       return new RestResult<>();
752     };
753   }
