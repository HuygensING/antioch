66    public AlexandriaClient(final URI alexandriaURI, SSLContext sslContext) {
67      this.alexandriaURI = alexandriaURI;
68      final ObjectMapper objectMapper = new ObjectMapper()//
69          .registerModule(new Jdk8Module())//
70          .registerModule(new JavaTimeModule());
71  
72      final JacksonJaxbJsonProvider jacksonProvider = new JacksonJaxbJsonProvider();
73      jacksonProvider.setMapper(objectMapper);
74  
75      PoolingHttpClientConnectionManager cm = new PoolingHttpClientConnectionManager();
76      cm.setMaxTotal(50);
77      cm.setDefaultMaxPerRoute(50);
78  
79      ApacheConnectorProvider connectorProvider = new ApacheConnectorProvider();
80      ClientConfig clientConfig = new ClientConfig(jacksonProvider)//
81          .connectorProvider(connectorProvider)//
82          .property(ApacheClientProperties.CONNECTION_MANAGER, cm)//
83          .property(ClientProperties.CONNECT_TIMEOUT, 60000)//
84          .property(ClientProperties.READ_TIMEOUT, 60000);
85  
86      if (sslContext == null) {
87        if ("https".equals(alexandriaURI.getScheme())) {
88          throw new RuntimeException("SSL connections need an SSLContext, use: new AlexandriaClient(uri, sslContext) instead.");
89        }
90        client = ClientBuilder.newClient(clientConfig);
91  
92      } else {
93        client = ClientBuilder.newBuilder()//
94            .sslContext(sslContext)//
95            .withConfig(clientConfig)//
96            .build();
97      }
98      rootTarget = client.target(alexandriaURI);
99    }

383   private <E> RestResult<E> toEntityRestResult(final Response response, final Class<E> entityClass) {
384     final RestResult<E> result = new RestResult<>();
385     final E cargo = response.readEntity(entityClass);
386     result.setCargo(cargo);
387     return result;
388   }

390   private RestResult<UUID> uuidFromLocationHeader(final Response response) {
391     final RestResult<UUID> result = new RestResult<>();
392     final String location = response.getHeaderString("Location");
393     final UUID uuid = UUID.fromString(location.replaceFirst(".*/", ""));
394     result.setCargo(uuid);
395     return result;
396   }

431   private Function<Response, RestResult<Void>> voidRestResult() {
432     return (response) -> {
433       response.bufferEntity(); // to notify connectors, such as the ApacheConnector, that the entity has been "consumed" and that it should release the current connection back into the Apache
434       // ConnectionManager pool (if being used). https://java.net/jira/browse/JERSEY-3149
435       return new RestResult<>();
436     };
437   }
