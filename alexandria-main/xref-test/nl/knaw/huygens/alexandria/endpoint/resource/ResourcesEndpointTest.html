39    public void getResourcesRefNoLongerExists() {
40      when(SERVICE_MOCK.readResource(any(UUID.class))).thenReturn(Optional.empty());
41      Response response = target(ROOTPATH).path(UUID.randomUUID().toString()).path("/ref").request().get();
42      assertThat(response.getStatus()).isEqualTo(Status.NOT_FOUND.getStatusCode());
43    }

53    @Test
54    public void testPost() {
55      TentativeAlexandriaProvenance provenance = mock(TentativeAlexandriaProvenance.class);
56      UUID id = UUID.randomUUID();
57      AlexandriaResource resource = new AlexandriaResource(id, provenance);
58      when(SERVICE_MOCK.readResource(any(UUID.class))).thenReturn(Optional.of(resource));
59      Response response = target(ROOTPATH).request().post(jsonEntity("{'resource':{'ref':'REF'}}"));
60      Log.debug("response={}", response);
61      assertThat(response.getLocation().toString()).contains("/resources/" + id);
62      assertThat(response.getStatus()).isEqualTo(Status.CREATED.getStatusCode());
63    }

72    public void testTwoPutsWillNotProduceAnError() throws IOException {
73      AlexandriaService service = mock(AlexandriaService.class);
74      ResourceCreationRequestBuilder requestBuilder = new ResourceCreationRequestBuilder();
75      LocationBuilder locationBuilder = new LocationBuilder(new MockConfiguration(), new EndpointPathResolver());
76      ResourceEntityBuilder entityBuilder = new ResourceEntityBuilder(locationBuilder);
77      ResourcesEndpoint re = new ResourcesEndpoint(service, requestBuilder, locationBuilder, entityBuilder);
78  
79      ResourcePrototype prototype1 = deserialize("{'resource':{'id':'c28626d4-493a-4204-83d9-e9ae17e15654','ref':'Referentie1'}}");
80      UUIDParam uuidparam1=new UUIDParam("c28626d4-493a-4204-83d9-e9ae17e15654");
81      Response response1 = re.setResourceAtSpecificID(uuidparam1,prototype1);
82      assertThat(response1.getStatus()).isEqualTo(Status.CREATED.getStatusCode());
83  
84      ResourcePrototype prototype2 = deserialize("{'resource':{'id':'d1753214-493a-4204-83d9-e9ae17e15654','ref':'Referentie2'}}");
85      UUIDParam uuidparam2=new UUIDParam("d1753214-493a-4204-83d9-e9ae17e15654");
86      Response response2 = re.setResourceAtSpecificID(uuidparam2,prototype2);
87      assertThat(response2.getStatus()).isEqualTo(Status.CREATED.getStatusCode());
88    }
89  
90    private ResourcePrototype deserialize(String json1) throws IOException {
91      return om.readValue(json1.replace("'", "\""), ResourcePrototype.class);
92    }
