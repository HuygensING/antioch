32  import javax.ws.rs.core.PathSegment;
33  import javax.ws.rs.core.UriInfo;

35  import com.google.common.collect.Lists;
36  import junitparams.JUnitParamsRunner;
37  import junitparams.Parameters;
38  import nl.knaw.huygens.alexandria.endpoint.UUIDParam;
39  import nl.knaw.huygens.alexandria.endpoint.resource.MatchesPathId.Validator;
40  import org.junit.Test;
41  import org.junit.runner.RunWith;
42  
43  @RunWith(JUnitParamsRunner.class)
44  public class MatchesPathIdValidatorTest {
45  
46    @Test
47    @Parameters({ //
48        "fe944b98-18b1-11e5-9d51-1f35c61c7a23, fe944b98-18b1-11e5-9d51-1f35c61c7a23, true", // same UUIDs
49        "fe944b98-18b1-11e5-9d51-1f35c61c7a23, 27947b26-18b2-11e5-be1c-c7fb9bd13f9b, false" // different UUIDs
50    })
51    public void testValidationOfUUIDinURIShouldMatchUUIDinPrototype(String uriId, String protoId, boolean valid) {
52      final PathSegment mockedPathSegment = mock(PathSegment.class);
53      when(mockedPathSegment.getPath()).thenReturn(protoId);
54  
55      final UriInfo mockedUriInfo = mock(UriInfo.class);
56      when(mockedUriInfo.getPathSegments()).thenReturn(Lists.newArrayList(mockedPathSegment));
57  
58      final ResourcePrototype mockedPrototype = mock(ResourcePrototype.class);
59      when(mockedPrototype.getId()).thenReturn(new UUIDParam(uriId));
60  
61      assertThat(new Validator(mockedUriInfo).isValid(mockedPrototype, null), is(valid));
62  
63      verify(mockedUriInfo, atLeastOnce()).getPathSegments();
64      verify(mockedPrototype).getId();
65    }
66  
67  }
