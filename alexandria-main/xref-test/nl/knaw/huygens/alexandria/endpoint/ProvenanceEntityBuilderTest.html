19  public class ProvenanceEntityBuilderTest {
20    @Test
21    public void test_provenance_entity_creation_from_builder() {
22      UUID id = UUID.randomUUID();
23      String who = "who";
24      Instant when = Instant.now();
25      String why = "why";
26      String baseURI = "http://alexandria.ax";
27  
28      TentativeAlexandriaProvenance tprovenance = new TentativeAlexandriaProvenance(who, when, why);
29      AlexandriaResource resource = new AlexandriaResource(id, tprovenance);
30      AlexandriaProvenance provenance = resource.getProvenance();
31      AlexandriaConfiguration config = mock(AlexandriaConfiguration.class);
32      when(config.getBaseURI()).thenReturn(URI.create(baseURI));
33      LocationBuilder locationBuilder = new LocationBuilder(config, new EndpointPathResolver());
34      ProvenanceEntityBuilder builder = new ProvenanceEntityBuilder(locationBuilder);
35      ProvenanceEntity entity = builder.build(provenance);
36  
37      String expectedLocation = baseURI + "/" + EndpointPaths.RESOURCES + "/" + id;
38      URI what = entity.getWhat();
39      Log.info("entity={}", entity);
40      Log.info("what={}", what);
41      assertThat(what.toString()).isEqualTo(expectedLocation);
42      assertThat(entity.getWhen()).isEqualTo(when.toString());
43      assertThat(entity.getWhy()).isEqualTo(why);
44    }
