25  import java.text.MessageFormat;
26  import java.util.stream.Collectors;
27  
28  import nl.knaw.huygens.tei.Element;
29  
30  public class AnnotationData {
31    String annotatedBaseText = "";
32    XmlAnnotationLevel level = XmlAnnotationLevel.element;
33    String type = "";
34    String xpath = "";
35    Object value;
36  
37    public String getAnnotatedBaseText() {
38      return annotatedBaseText;
39    }
40  
41    public AnnotationData setAnnotatedBaseText(String annotatedBaseText) {
42      this.annotatedBaseText = annotatedBaseText;
43      return this;
44    }
45  
46    public XmlAnnotationLevel getLevel() {
47      return level;
48    }
49  
50    public AnnotationData setLevel(XmlAnnotationLevel level) {
51      this.level = level;
52      return this;
53    }
54  
55    public String getXPath() {
56      return xpath;
57    }
58  
59    public AnnotationData setXPath(String xpath) {
60      this.xpath = xpath;
61      return this;
62    }
63  
64    public String getType() {
65      return type;
66    }
67  
68    public AnnotationData setType(String type) {
69      this.type = type;
70      return this;
71    }
72  
73    public Object getValue() {
74      return value;
75    }
76  
77    public AnnotationData setValue(Object value) {
78      this.value = value;
79      return this;
80    }
81  
82    public String toVerbose() {
83      if (getLevel().equals(XmlAnnotationLevel.element)) {
84        Element element = (Element) value;
85        String prefix = ", with ";
86        String attributeAnnotationActions = element.getAttributes().entrySet().stream()//
87            .map(kv -> "attribute annotation '" + kv.getKey() + "'='" + kv.getValue() + "'")//
88            .collect(Collectors.joining(", and ", prefix, ""));
89  
90        if (attributeAnnotationActions.equals(prefix)) {
91          attributeAnnotationActions = "";
92        }
93  
94        return (MessageFormat.format(//
95            "adding element annotation on text ''{0}'', on xpath ''{1}'' element={2}{3}", //
96            annotatedBaseText, //
97            xpath, //
98            type, //
99            attributeAnnotationActions)//
100       );
101     }
102 
103     return "adding attribute annotation on xpath '" + xpath + "' : " + type + "=" + value;
104   }
105 
106 }
