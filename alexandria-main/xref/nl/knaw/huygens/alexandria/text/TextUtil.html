41    public static BaseLayerData extractBaseLayerData(String xml, BaseLayerDefinition def) {
42      Document document = Document.createFromXml(xml, true);
43      List<String> baseElementNames = def.getBaseElementDefinitions()//
44          .stream()//
45          .map(BaseElementDefinition::getName)//
46          .collect(toList());
47  
48      // determine existing base element ids
49      BaseLayerIdVisitor idVisitor = new BaseLayerIdVisitor(baseElementNames);
50      document.accept(idVisitor);
51      List<String> existingBaseElementIds = idVisitor.getBaseElementIds();
52  
53      // set xml:id on base elements where missing
54      AddMissingBaseLayerIdVisitor addIdVisitor = new AddMissingBaseLayerIdVisitor(existingBaseElementIds, baseElementNames);
55      document.accept(addIdVisitor);
56      String xmlWithIds = addIdVisitor.getContext().getResult();
57  
58      return extractSubResourceTexts(def, xmlWithIds, "");
59    }
60  
61    private static BaseLayerData extractSubResourceTexts(BaseLayerDefinition def, String xml, String parentId) {
62      // extract base layer + nearest layer of subresource xml
63      Document documentWithIds = Document.createFromXml(xml, true);
64      SEVContext context = new SEVContext();
65      SubresourceElementVisitor subresourceVisitor = new SubresourceElementVisitor(context, def.getSubresourceElements());
66      documentWithIds.accept(subresourceVisitor);
67      String baseText = context.getBaseText();
68  
69      Document baseDocument = Document.createFromXml(baseText, true);
70      BLVContext blContext = new BLVContext();
71      BaseLayerVisitor blVisitor = new BaseLayerVisitor(blContext, def, parentId);
72      baseDocument.accept(blVisitor);
73  
74      BaseLayerData baseLayerData = blVisitor.getBaseLayerData();
75      Map<String, String> subresourceXPathMap = blContext.getSubresourceXPathMap();
76      for (Entry<String, String> entry : context.getSubresourceTexts().entrySet()) {
77        String id = entry.getKey();
78        String text = entry.getValue();
79        String xpath = subresourceXPathMap.get(id);
80        BaseLayerData subLayerData = extractSubResourceTexts(def, text, xpath);
81        baseLayerData.getSubLayerData().add(subLayerData);
82        Log.info("subLayerData={}", subLayerData);
83        Log.info("make subresource with sub=\"xpath:{}\" and text={}", xpath, subLayerData.getBaseLayer());
84      }
85  
86      return baseLayerData;
87    }

89    public static String xpath(Element element) {
90      String xpath = "";
91      if (element.hasAttribute(XML_ID)) {
92        xpath = "//" + element.getName() + "[@xml:id='" + element.getAttribute(XML_ID) + "']";
93      } else {
94        Element parent = element.getParent();
95        if (parent == null) {
96          xpath = "/" + element.getName();
97        } else {
98          xpath = xpath(parent) + "/" + element.getName();
99        }
100     }
101     return xpath;
102   }

104   public static String substringXPath(Element parent, int start, int length) {
105     Log.info("offset: (start={},length={})", start, length);
106     return "substring(" + xpath(parent) + "," + start + "," + length + ")";
107   }
