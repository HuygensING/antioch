25  import java.util.Stack;
26  
27  import nl.knaw.huygens.tei.Comment;
28  import nl.knaw.huygens.tei.CommentHandler;
29  import nl.knaw.huygens.tei.DelegatingVisitor;
30  import nl.knaw.huygens.tei.Element;
31  import nl.knaw.huygens.tei.ElementHandler;
32  import nl.knaw.huygens.tei.ProcessingInstruction;
33  import nl.knaw.huygens.tei.ProcessingInstructionHandler;
34  import nl.knaw.huygens.tei.Text;
35  import nl.knaw.huygens.tei.Traversal;
36  import nl.knaw.huygens.tei.XmlContext;
37  import nl.knaw.huygens.tei.handlers.XmlTextHandler;
38  
39  public class XmlVisitor extends DelegatingVisitor<XmlContext> implements CommentHandler<XmlContext>, ElementHandler<XmlContext>, ProcessingInstructionHandler<XmlContext> {
40  
41    private static ParseResult result;
42    private Stack<Integer> startIndexStack = new Stack<>();
43    private Stack<Element> elementStack = new Stack<>();
44    private static boolean lastNodeWasText = false;
45  
46    public XmlVisitor(ParseResult result) {
47      super(new XmlContext());
48      XmlVisitor.result = result;
49      setDefaultElementHandler(this);
50      setCommentHandler(this);
51      setProcessingInstructionHandler(this);
52      setTextHandler(new TextSegmentHandler());
53    }
54  
55    @Override
56    public Traversal enterElement(Element element, XmlContext context) {
57      elementStack.add(element);
58      lastNodeWasText = false;
59      startIndexStack.push(result.getTextSegments().size());
60      if (element.hasNoChildren()) {
61        result.getTextSegments().add("");
62      }
63      return Traversal.NEXT;
64    }
65  
66    @Override
67    public Traversal leaveElement(Element element, XmlContext context) {
68      elementStack.pop();
69      lastNodeWasText = false;
70      XmlAnnotation xmlAnnotation = new XmlAnnotation(element.getName(), element.getAttributes(), elementStack.size())//
71          .setMilestone(element.hasNoChildren())//
72          .setFirstSegmentIndex(startIndexStack.pop())//
73          .setLastSegmentIndex(result.getTextSegments().size() - 1)//
74          ;
75      result.getXmlAnnotations().add(xmlAnnotation);
76      return Traversal.NEXT;
77    }
78  
79    @Override
80    public Traversal visitProcessingInstruction(ProcessingInstruction processingInstruction, XmlContext context) {
81      return Traversal.NEXT;
82    }
83  
84    @Override
85    public Traversal visitComment(Comment comment, XmlContext context) {
86      return Traversal.NEXT;
87    }
88  
89    public static class TextSegmentHandler extends XmlTextHandler<XmlContext> {
90      @Override
91      public Traversal visitText(Text text, XmlContext context) {
92        String filteredText = filterText(text.getText());
93        if (lastNodeWasText) {
94          int lastIndex = result.getTextSegments().size() - 1;
95          String segment = result.getTextSegments().get(lastIndex);
96          result.getTextSegments().set(lastIndex, segment + filteredText);
97        } else {
98          result.getTextSegments().add(filteredText);
99        }
100       lastNodeWasText = true;
101       return Traversal.NEXT;
102     }
103   }
104 
105 }
