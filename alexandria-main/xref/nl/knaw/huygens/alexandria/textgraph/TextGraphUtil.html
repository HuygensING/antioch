102   public static StreamingOutput xmlOutputStream(AlexandriaService service, UUID resourceId, String viewName, Map<String, String> viewParameters) {
103     if (StringUtils.isNotBlank(viewName)) {
104       TextView textView = service.getTextView(resourceId, viewName)//
105           .orElseThrow(() -> new NotFoundException("No view '" + viewName + "' found for this resource."));

170     public TextViewContext(TextView textView) {
171       elementViewMap = textView.getElementViewMap();
172     }

174     public boolean includeTag(String name, TextAnnotation textAnnotation) {
175       ElementView defaultElementView = elementViewMap.get(TextViewDefinition.DEFAULT_ATTRIBUTENAME);
176       ElementView elementView = elementViewMap.getOrDefault(name, defaultElementView);
177       ElementMode elementMode = elementView.getElementMode();
178       ElementMode defaultViewElementMode = defaultElementView.getElementMode();
179       Optional<AttributePreCondition> preCondition = elementView.getPreCondition();
180       boolean includeAccordingToElementMode = elementMode.equals(ElementMode.show);
181       boolean preConditionIsMet = preConditionIsMet(preCondition, textAnnotation);
182       if (!preConditionIsMet) {
183         boolean preConditionIsFirstOfFunction = preCondition.isPresent() //
184             && AttributeFunction.firstOf.equals(preCondition.get().getFunction());
185         includeAccordingToElementMode = !preConditionIsFirstOfFunction && defaultViewElementMode.equals(ElementMode.show);
186       }
187 
188       return notInsideIgnoredElement() && includeAccordingToElementMode;
189     }
190 
191     private boolean preConditionIsMet(Optional<AttributePreCondition> preCondition, TextAnnotation textAnnotation) {
192       if (preCondition.isPresent()) {
193         if (overruledTextAnnotations.contains(textAnnotation)) {
194           return false;
195         }
196         AttributePreCondition attributePreCondition = preCondition.get();
197         String attribute = attributePreCondition.getAttribute();
198         List<String> values = attributePreCondition.getValues();
199         String actualValue = textAnnotation.getAttributes().get(attribute);
200         switch (attributePreCondition.getFunction()) {
201         case is:
202           return values.contains(actualValue);
203         case isNot:
204           return !values.contains(actualValue);
205         case firstOf:
206           // TODO
207           break;
208         default:
209           // TODO
210           break;
211         }
212 
213       }
214       return true;
215     }

217     public Map<String, String> includedAttributes(TextAnnotation textAnnotation) {
218       ElementView elementView = elementViewMap.getOrDefault(textAnnotation.getName(), elementViewMap.get(TextViewDefinition.DEFAULT_ATTRIBUTENAME));
219       Map<String, String> allAttributes = textAnnotation.getAttributes();
220       Map<String, String> attributesToInclude = Maps.newHashMap();
221       AttributeMode attributeMode = elementView.getAttributeMode();
222 
223       switch (attributeMode) {
224       case showAll:
225         return allAttributes;
226 
227       case hideAll:
228         break;
229 
230       case showOnly:
231         allAttributes.forEach((k, v) -> {
232           if (elementView.getRelevantAttributes().contains(k)) {
233             attributesToInclude.put(k, v);
234           }
235         });
236         break;
237 
238       case hideOnly:
239         allAttributes.forEach((k, v) -> {
240           if (!elementView.getRelevantAttributes().contains(k)) {
241             attributesToInclude.put(k, v);
242           }
243         });
244         break;
245 
246       default:
247         throw new RuntimeException("unexpected attributemode: " + attributeMode);
248       }
249       return attributesToInclude;
250     }

252     public void pushWhenIgnoring(TextAnnotation textAnnotation) {
253       boolean ignoring = determineIgnoring(textAnnotation);
254       if (ignoring) {
255         ignoredAnnotationStack.push(textAnnotation);
256       }
257     }
258 
259     private boolean determineIgnoring(TextAnnotation textAnnotation) {
260       ElementView defaultElementView = elementViewMap.get(TextViewDefinition.DEFAULT_ATTRIBUTENAME);
261       ElementView elementView = elementViewMap.getOrDefault(textAnnotation.getName(), defaultElementView);
262       ElementMode elementMode = elementView.getElementMode();
263       ElementMode defaultElementMode = defaultElementView.getElementMode();
264       Optional<AttributePreCondition> preCondition = elementView.getPreCondition();
265       boolean preConditionIsMet = preConditionIsMet(preCondition, textAnnotation);
266       boolean hideThisElement = elementMode.equals(ElementMode.hide);
267       if (!preConditionIsMet) {
268         hideThisElement = defaultElementMode.equals(ElementMode.hide);
269       }
270       return hideThisElement;
271     }
272 
273     public void popWhenIgnoring(TextAnnotation textAnnotation) {
274       boolean ignoring = determineIgnoring(textAnnotation);
275       if (ignoring && !ignoredAnnotationStack.isEmpty()) {
276         ignoredAnnotationStack.pop();
277       }
278     }
279 
280     public boolean notInsideIgnoredElement() {
281       return ignoredAnnotationStack.isEmpty();
282     }

284     /**
285      * registers TextAnnotations in segment that:
286      * - annotate the same textrange
287      * - have the same name
288      * - that name is used in a firstOf function in the viewDefinition
289      * So they can be handled properly in includeTag
290      *
291      * @param segment
292      *          the TextGraphSegment to analyze
293      */
294     public void registerCompetingTextAnnotations(TextGraphSegment segment) {
295       List<String> relevantElementNames = elementViewMap.entrySet().stream()//
296           .filter(this::hasFirstOfAttributeFunction)//
297           .map(Entry::getKey)//
298           .collect(toList());
299       Set<TextAnnotation> segmentTextAnnotations = Sets.newHashSet(segment.textAnnotationsToOpen);
300       segmentTextAnnotations.addAll(segment.textAnnotationsToClose);
301       segmentTextAnnotations.removeIf(ta -> !relevantElementNames.contains(ta.getName()));
302       overruledTextAnnotations = segmentTextAnnotations.stream()//
303           .collect(groupingBy(this::textAnnotationGrouping))//
304           .values()//
305           .stream()//
306           .map(this::overruledTextAnnotations)//
307           .flatMap(List::stream)//
308           .collect(toList());
309     }

312     private boolean hasFirstOfAttributeFunction(Entry<String, ElementView> entry) {
313       Optional<AttributePreCondition> preCondition = entry.getValue().getPreCondition();
314       return preCondition.isPresent() && ElementView.AttributeFunction.firstOf.equals(preCondition.get().getFunction());
315     }

322     private List<TextAnnotation> overruledTextAnnotations(List<TextAnnotation> group) {
323       String elementName = group.get(0).getName();
324       AttributePreCondition attributePreCondition = elementViewMap.get(elementName)//
325           .getPreCondition().get();
326       String attribute = attributePreCondition.getAttribute();
327       List<String> prioritizedValues = Lists.newArrayList(attributePreCondition.getValues());
328       int originalSize = group.size();
329       do {
330         String value = prioritizedValues.remove(0);
331         group.removeIf(ta -> value.equals(ta.getAttributes().get(attribute)));
332       } while (group.size() == originalSize && !prioritizedValues.isEmpty());
333       return group;
334     }

349   public static void streamTextGraphSegment(Writer writer, TextGraphSegment segment, TextViewContext textViewContext) {
350     try {
351       textViewContext.registerCompetingTextAnnotations(segment);
352       writeOpenTags(writer, segment, textViewContext);
353       writeMilestoneTags(writer, segment, textViewContext);
354       writeText(writer, segment, textViewContext);
355       writeCloseTags(writer, segment, textViewContext);
356 
357     } catch (IOException ioe) {
358       throw new RuntimeException(ioe);
359     }
360   }

362   private static void writeCloseTags(Writer writer, TextGraphSegment segment, TextViewContext textViewContext) throws IOException {
363     for (TextAnnotation textAnnotation : segment.getTextAnnotationsToClose()) {
364       String name = textAnnotation.getName();
365       if (textViewContext.includeTag(name, textAnnotation)) {
366         String closeTag = getCloseTag(name);
367         writer.write(closeTag);
368       }
369       textViewContext.popWhenIgnoring(textAnnotation);
370     }
371   }

373   private static void writeText(Writer writer, TextGraphSegment segment, TextViewContext textViewContext) throws IOException {
374     if (textViewContext.notInsideIgnoredElement()) {
375       writeText(writer, segment);
376     }
377   }

379   private static void writeMilestoneTags(Writer writer, TextGraphSegment segment, TextViewContext textViewContext) throws IOException {
380     Optional<TextAnnotation> optionalMilestone = segment.getMilestoneTextAnnotation();
381     if (optionalMilestone.isPresent()) {
382       TextAnnotation milestone = optionalMilestone.get();
383       String name = milestone.getName();
384       if (textViewContext.includeTag(name, milestone)) {
385         String milestoneTag = getMilestoneTag(name, textViewContext.includedAttributes(milestone));
386         writer.write(milestoneTag);
387       }
388     }
389   }
390 
391   private static void writeOpenTags(Writer writer, TextGraphSegment segment, TextViewContext textViewContext) throws IOException {
392     List<TextAnnotation> textAnnotationsToOpen = textViewContext.getOrderedTextAnnotationsToOpen(segment);
393     for (TextAnnotation textAnnotation : textAnnotationsToOpen) {
394       String name = textAnnotation.getName();
395       if (textViewContext.includeTag(name, textAnnotation)) {
396         String openTag = getOpenTag(name, textViewContext.includedAttributes(textAnnotation));
397         writer.write(openTag);
398       }
399       textViewContext.pushWhenIgnoring(textAnnotation);
400     }
401   }

415   public static void appendAttributes(StringBuilder builder, Map<String, String> attributes) {
416     for (Map.Entry<String, String> entry : attributes.entrySet()) {
417       builder.append(' ').append(entry.getKey()).append('=');
418       builder.append('"');
419       appendAttributeValue(builder, entry.getValue());
420       builder.append('"');
421     }
422   }

435   private static StringBuilder openingTagBuilder(String name, Map<String, String> attributes) {
436     StringBuilder builder = new StringBuilder("<").append(name);
437     appendAttributes(builder, attributes);
438     return builder;
439   }

441   private static void appendAttributeValue(StringBuilder builder, String value) {
442     int n = value.length();
443     for (int i = 0; i < n; i++) {
444       char c = value.charAt(i);
445       switch (c) {
446       case '<':
447         builder.append("&lt;");
448         break;
449       case '>':
450         builder.append("&gt;");
451         break;
452       case '&':
453         builder.append("&amp;");
454         break;
455       default:
456         builder.append(c);
457         break;
458       }
459     }
