56    public static StreamingOutput streamXML(AlexandriaService service, UUID resourceId) {
57      return output -> {
58        Writer writer = createBufferedUTF8OutputStreamWriter(output);
59        Consumer<TextGraphSegment> action = segment -> streamTextGraphSegment(writer, segment);
60        stream(service, resourceId, writer, action);
61      };

64    public static void streamTextGraphSegment(Writer writer, TextGraphSegment segment) {
65      try {
66        if (segment.isMilestone()) {
67          TextAnnotation milestone = segment.getMilestone();
68          String name = milestone.getName();
69          String openTag = getMilestoneTag(name, milestone.getAttributes());
70          writer.write(openTag);
71  
72        } else {
73          for (TextAnnotation textAnnotation : segment.getTextAnnotationsToOpen()) {
74            String name = textAnnotation.getName();
75            String openTag = getOpenTag(name, textAnnotation.getAttributes());
76            writer.write(openTag);
77          }
78          writer.write(segment.getTextSegment());
79          for (TextAnnotation textAnnotation : segment.getTextAnnotationsToClose()) {
80            String name = textAnnotation.getName();
81            String closeTag = getCloseTag(name);
82            writer.write(closeTag);
83          }
84        }
85      } catch (IOException ioe) {
86        throw new RuntimeException(ioe);
87      }
88    }

113   protected static class TextViewContext {
114     private Map<String, ElementView> elementViewMap;
115     private Stack<TextAnnotation> ignoredAnnotationStack = new Stack<>();
116 
117     public TextViewContext(TextView textView) {
118       elementViewMap = textView.getElementViewMap();
119     }
120 
121     public boolean includeTag(String name, Map<String, String> attributes) {
122       ElementView defaultElementView = elementViewMap.get(TextViewDefinition.DEFAULT_ATTRIBUTENAME);
123       ElementView elementView = elementViewMap.getOrDefault(name, defaultElementView);
124       ElementMode elementMode = elementView.getElementMode();
125       ElementMode defaultViewElementMode = defaultElementView.getElementMode();
126       Optional<AttributePreCondition> preCondition = elementView.getPreCondition();
127       boolean includeAccordingToElementMode = elementMode.equals(ElementMode.show);
128       boolean preConditionIsMet = preConditionIsMet(preCondition, attributes);
129       if (!preConditionIsMet) {
130         includeAccordingToElementMode = defaultViewElementMode.equals(ElementMode.show);
131       }
132 
133       return notInsideIgnoredElement() && includeAccordingToElementMode;
134     }
135 
136     private boolean preConditionIsMet(Optional<AttributePreCondition> preCondition, Map<String, String> attributes) {
137       if (preCondition.isPresent()) {
138         AttributePreCondition attributePreCondition = preCondition.get();
139         String attribute = attributePreCondition.getAttribute();
140         List<String> values = attributePreCondition.getValues();
141         String actualValue = attributes.get(attribute);
142         switch (attributePreCondition.getFunction()) {
143           case is:
144             return values.contains(actualValue);
145           case isNot:
146             return !values.contains(actualValue);
147           case firstOf:
148             // TODO
149             break;
150         }
151 
152       }
153       return true;
154     }
155 
156     public Map<String, String> includedAttributes(TextAnnotation textAnnotation) {
157       ElementView elementView = elementViewMap.getOrDefault(textAnnotation.getName(), elementViewMap.get(TextViewDefinition.DEFAULT_ATTRIBUTENAME));
158       Map<String, String> allAttributes = textAnnotation.getAttributes();
159       Map<String, String> attributesToInclude = Maps.newHashMap();
160       AttributeMode attributeMode = elementView.getAttributeMode();
161 
162       switch (attributeMode) {
163         case showAll:
164           return allAttributes;
165 
166         case hideAll:
167           break;
168 
169         case showOnly:
170           allAttributes.forEach((k, v) -> {
171             if (elementView.getRelevantAttributes().contains(k)) {
172               attributesToInclude.put(k, v);
173             }
174           });
175           break;
176 
177         case hideOnly:
178           allAttributes.forEach((k, v) -> {
179             if (!elementView.getRelevantAttributes().contains(k)) {
180               attributesToInclude.put(k, v);
181             }
182           });
183           break;
184 
185         default:
186           throw new RuntimeException("unexpected attributemode: " + attributeMode);
187       }
188       return attributesToInclude;
189     }
190 
191     public void pushWhenIgnoring(TextAnnotation textAnnotation) {
192       boolean ignoring = determineIgnoring(textAnnotation);
193       if (ignoring) {
194         ignoredAnnotationStack.push(textAnnotation);
195       }
196     }
197 
198     private boolean determineIgnoring(TextAnnotation textAnnotation) {
199       ElementView defaultElementView = elementViewMap.get(TextViewDefinition.DEFAULT_ATTRIBUTENAME);
200       ElementView elementView = elementViewMap.getOrDefault(textAnnotation.getName(), defaultElementView);
201       ElementMode elementMode = elementView.getElementMode();
202       ElementMode defaultElementMode = defaultElementView.getElementMode();
203       Optional<AttributePreCondition> preCondition = elementView.getPreCondition();
204       boolean preConditionIsMet = preConditionIsMet(preCondition, textAnnotation.getAttributes());
205       boolean hideThisElement = elementMode.equals(ElementMode.hide);
206       if (!preConditionIsMet) {
207         hideThisElement = defaultElementMode.equals(ElementMode.hide);
208       }
209       return hideThisElement;
210     }
211 
212     public void popWhenIgnoring(TextAnnotation textAnnotation) {
213       boolean ignoring = determineIgnoring(textAnnotation);
214       if (ignoring && !ignoredAnnotationStack.isEmpty()) {
215         ignoredAnnotationStack.pop();
216       }
217     }
218 
219     public boolean notInsideIgnoredElement() {
220       return ignoredAnnotationStack.isEmpty();
221     }
222   }

224   public static void streamTextGraphSegment(Writer writer, TextGraphSegment segment, TextViewContext textViewContext) {
225     try {
226       if (segment.isMilestone()) {
227         TextAnnotation milestone = segment.getMilestone();
228         String name = milestone.getName();
229         if (textViewContext.includeTag(name, milestone.getAttributes())) {
230           String openTag = getMilestoneTag(name, textViewContext.includedAttributes(milestone));
231           writer.write(openTag);
232         }
233 
234       } else {
235         for (TextAnnotation textAnnotation : segment.getTextAnnotationsToOpen()) {
236           String name = textAnnotation.getName();
237           if (textViewContext.includeTag(name, textAnnotation.getAttributes())) {
238             String openTag = getOpenTag(name, textViewContext.includedAttributes(textAnnotation));
239             writer.write(openTag);
240           }
241           textViewContext.pushWhenIgnoring(textAnnotation);
242         }
243         if (textViewContext.notInsideIgnoredElement()) {
244           writer.write(segment.getTextSegment());
245         }
246         for (TextAnnotation textAnnotation : segment.getTextAnnotationsToClose()) {
247           String name = textAnnotation.getName();
248           if (textViewContext.includeTag(name, textAnnotation.getAttributes())) {
249             String closeTag = getCloseTag(name);
250             writer.write(closeTag);
251           }
252           textViewContext.popWhenIgnoring(textAnnotation);
253         }
254       }
255     } catch (IOException ioe) {
256       throw new RuntimeException(ioe);
257     }
258   }

260   public static String getMilestoneTag(String name, Map<String, String> attributes) {
261     return openingTagBuilder(name, attributes).append("/>").toString();
262   }
263 
264   public static String getOpenTag(String name, Map<String, String> attributes) {
265     return openingTagBuilder(name, attributes).append(">").toString();
266   }
267 
268   public static String getCloseTag(String name) {
269     return "</" + name + ">";
270   }

272   public static void appendAttributes(StringBuilder builder, Map<String, String> attributes) {
273     for (Map.Entry<String, String> entry : attributes.entrySet()) {
274       builder.append(' ').append(entry.getKey()).append('=');
275       builder.append('"');
276       appendAttributeValue(builder, entry.getValue());
277       builder.append('"');
278     }
279   }

308   private static Writer createBufferedUTF8OutputStreamWriter(OutputStream output) throws UnsupportedEncodingException {
309     return new BufferedWriter(new OutputStreamWriter(output, "UTF-8"));
310   }

312   private static void stream(AlexandriaService service, UUID resourceId, Writer writer, Consumer<TextGraphSegment> action) throws IOException {
313     service.runInTransaction(() -> service.getTextGraphSegmentStream(resourceId).forEach(action));
314     writer.flush();
315   }

317   private static StringBuilder openingTagBuilder(String name, Map<String, String> attributes) {
318     StringBuilder builder = new StringBuilder("<").append(name);
319     appendAttributes(builder, attributes);
320     return builder;
321   }

323   private static void appendAttributeValue(StringBuilder builder, String value) {
324     int n = value.length();
325     for (int i = 0; i < n; i++) {
326       char c = value.charAt(i);
327       switch (c) {
328         case '<':
329           builder.append("&lt;");
330           break;
331         case '>':
332           builder.append("&gt;");
333           break;
334         case '&':
335           builder.append("&amp;");
336           break;
337         default:
338           builder.append(c);
339           break;
340       }
341     }
342   }
