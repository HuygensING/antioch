3   import java.io.IOException;
4   import java.text.SimpleDateFormat;
5   import java.util.Date;
6   import java.util.Map;
7   import java.util.PropertyResourceBundle;
8   import java.util.TimeZone;
9   
10  import javax.inject.Singleton;
11  import javax.ws.rs.GET;
12  import javax.ws.rs.Path;
13  import javax.ws.rs.core.Response;
14  
15  import com.google.common.collect.Maps;
16  
17  import io.swagger.annotations.Api;
18  import io.swagger.annotations.ApiOperation;
19  import nl.knaw.huygens.alexandria.endpoint.JSONEndpoint;
20  
21  @Singleton
22  @Path("about")
23  @Api("about")
24  public class AboutEndpoint extends JSONEndpoint {
25    static Date starttime;
26  
27    public AboutEndpoint() {
28      if (starttime == null) {
29        init();
30      }
31    }
32  
33    // needs to be called at the start of the server
34    public static void init() {
35      starttime = new Date();
36    }
37  
38    /**
39     * Show information about the back-end
40     *
41     * @return about-data map
42     */
43    @GET
44    @ApiOperation("get information about the server (version,builddate,started)")
45    public Response getMetadata() {
46      Map<String, String> data = Maps.newLinkedHashMap();
47      data.put("version", getProperty("version"));
48      data.put("commit id", getProperty("commit_id"));
49      data.put("builddate", getProperty("builddate"));
50      SimpleDateFormat simpleDateFormat = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
51      simpleDateFormat.setTimeZone(TimeZone.getTimeZone("AMS"));
52      data.put("started", simpleDateFormat.format(this.starttime));
53      return Response.ok(data).build();
54    }
55  
56    // private methods
57    private static PropertyResourceBundle propertyResourceBundle;
58  
59    private static synchronized String getProperty(String key) {
60      if (propertyResourceBundle == null) {
61        try {
62          propertyResourceBundle = new PropertyResourceBundle(//
63              Thread.currentThread().getContextClassLoader().getResourceAsStream("about.properties"));
64        } catch (IOException e) {
65          e.printStackTrace();
66        }
67      }
68      return propertyResourceBundle.getString(key);
69    }
70  
71  }
