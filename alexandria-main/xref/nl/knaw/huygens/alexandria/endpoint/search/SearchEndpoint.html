35  import javax.ws.rs.PathParam;

37  import javax.ws.rs.core.Response;

39  import io.swagger.annotations.Api;
40  import io.swagger.annotations.ApiOperation;
41  import nl.knaw.huygens.Log;
42  import nl.knaw.huygens.alexandria.endpoint.JSONEndpoint;
43  import nl.knaw.huygens.alexandria.endpoint.LocationBuilder;
44  import nl.knaw.huygens.alexandria.endpoint.UUIDParam;
45  import nl.knaw.huygens.alexandria.exception.NotFoundException;
46  import nl.knaw.huygens.alexandria.model.search.AlexandriaQuery;
47  import nl.knaw.huygens.alexandria.service.AlexandriaService;
48  
49  @Singleton
50  @Path(SEARCHES)
51  @Api(SEARCHES)
52  public class SearchEndpoint extends JSONEndpoint {
53  
54    private final AlexandriaService service;
55    private final SearchResultEntityBuilder entityBuilder;
56    private final LocationBuilder locationBuilder;
57  
58    @Inject
59    public SearchEndpoint(final AlexandriaService service, //
60        final SearchResultEntityBuilder entityBuilder, //
61        final LocationBuilder locationBuilder) {
62      this.service = service;
63      this.entityBuilder = entityBuilder;
64      this.locationBuilder = locationBuilder;
65    }
66  
67    @POST
68    @Consumes(MediaType.APPLICATION_JSON)
69    @ApiOperation("create new SearchResult")
70    public Response createSearchResult(@NotNull @Valid AlexandriaQuery query) {
71      Log.trace("query=[{}]", query);
72      SearchResult searchResult = service.execute(query);
73      SearchResultCache.add(searchResult);
74      return Response.created(locationBuilder.locationOf(searchResult)).build();
75    }
76  
77    @GET
78    @Path("{uuid}")
79    @ApiOperation(value = "Get the SearchResult with the given uuid", response = SearchResult.class)
80    public Response getSearchResultsByID(@PathParam("uuid") final UUIDParam uuid) {
81      SearchResult search = getSearchResult(uuid);
82      return Response.ok(entityBuilder.build(search)).build();
83    }
84  
85    public static final String RESULTPAGES = "pages";
86  
87    @GET
88    @Path("{uuid}/" + RESULTPAGES + "/{pageNumber:[0-9]+}")
89    @ApiOperation(value = "Get the SearchResultPage with the given uuid and page number", response = SearchResultPage.class)
90    public Response getResultPage(@PathParam("uuid") final UUIDParam uuid, @PathParam("pageNumber") int pageNumber) {
91      SearchResult searchResult = getSearchResult(uuid);
92      int totalResultPages = searchResult.getTotalPages();
93      if (totalResultPages == 0) {
94        throw new NotFoundException("no result pages found");
95      }
96      if (pageNumber < 1 || pageNumber > totalResultPages) {
97        throw new NotFoundException("pageNumber should be between 1 and " + totalResultPages);
98      }
99      String baseURI = locationBuilder.locationOf(searchResult) + "/" + RESULTPAGES + "/";
100     SearchResultPage page = new SearchResultPage(baseURI, pageNumber, totalResultPages, searchResult.getPageSize());
101     page.setResults(searchResult.getRecordsForPage(pageNumber));
102     return Response.ok(page).build();
103   }
104 
105   private SearchResult getSearchResult(final UUIDParam uuid) {
106     return SearchResultCache.get(uuid.getValue()) //
107         .orElseThrow(() -> new NotFoundException("no SearchResult found with id " + uuid));
108   }
