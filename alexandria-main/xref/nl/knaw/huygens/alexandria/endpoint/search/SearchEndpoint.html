26  @Singleton
27  @Path(SEARCHES)
28  @Api(SEARCHES)
29  public class SearchEndpoint extends JSONEndpoint {
30  
31    private final AlexandriaService service;
32    private final SearchResultEntityBuilder entityBuilder;
33    private final LocationBuilder locationBuilder;
34  
35    @Inject
36    public SearchEndpoint(final AlexandriaService service,                                   //
37        final SearchResultEntityBuilder entityBuilder,                                  //
38        final LocationBuilder locationBuilder) {
39      this.service = service;
40      this.entityBuilder = entityBuilder;
41      this.locationBuilder = locationBuilder;
42    }
43  
44    @POST
45    @Consumes(MediaType.APPLICATION_JSON)
46    @ApiOperation("create new SearchResult")
47    public Response createSearchResult(@NotNull @Valid AlexandriaQuery query) {
48      Log.trace("query=[{}]", query);
49      SearchResult searchResult = service.execute(query);
50      SearchResultCache.add(searchResult);
51      return Response.created(locationBuilder.locationOf(searchResult)).build();
52    }
53  
54    @GET
55    @Path("{uuid}")
56    @ApiOperation(value = "Get the SearchResult with the given uuid", response = SearchEntity.class)
57    public Response getSearchResultsByID(@PathParam("uuid") final UUIDParam uuid) {
58      SearchResult search = SearchResultCache.get(uuid.getValue())//
59          .orElseThrow(() -> new NotFoundException("no SearchResult found with id " + uuid));
60      return Response.ok(entityBuilder.build(search)).build();
61    }
