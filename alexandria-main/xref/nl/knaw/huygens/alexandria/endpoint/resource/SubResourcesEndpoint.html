27  public class SubResourcesEndpoint extends JSONEndpoint {
28  
29    private final AlexandriaService service;
30    private final UUID parentUuid;
31    private final ResourceCreationRequestBuilder requestBuilder;
32    private final LocationBuilder locationBuilder;
33  
34    @Inject
35    public SubResourcesEndpoint(AlexandriaService service, ResourceCreationRequestBuilder requestBuilder, LocationBuilder locationBuilder, @PathParam("uuid") final UUIDParam uuidParam) {
36      this.locationBuilder = locationBuilder;
37      Log.trace("resourceService=[{}], uuidParam=[{}]", service, uuidParam);
38      this.service = service;
39      this.requestBuilder = requestBuilder;
40      this.parentUuid = uuidParam.getValue();
41    }
42  
43    @GET
44    @ApiOperation(value = "get subresources", response = ResourceEntity.class)
45    public Response get() {
46      final Set<AlexandriaResource> subresources = service.readSubResources(parentUuid);
47      final Set<SubResourceEntity> outgoing = subresources.stream()//
48          .map(sr -> SubResourceEntity.of(sr).withLocationBuilder(locationBuilder))//
49          .collect(toSet());
50      return Response.ok(outgoing).build();
51    }
52  
53    @POST
54    @Consumes(MediaType.APPLICATION_JSON)
55    @ApiOperation("add subresource")
56    public Response addSubResource(@NotNull @Valid SubResourcePrototype prototype) {
57      prototype.setState(AlexandriaState.TENTATIVE);
58      SubResourceCreationRequest request = requestBuilder.build(parentUuid, prototype);
59      AlexandriaResource resource = request.execute(service);
60      return Response.created(locationBuilder.locationOf(resource)).build();
61    }

63  }
