10  import java.lang.annotation.ElementType;
11  import java.lang.annotation.Retention;
12  import java.lang.annotation.RetentionPolicy;
13  import java.lang.annotation.Target;
14  import java.util.List;

16  import nl.knaw.huygens.Log;
17  import nl.knaw.huygens.alexandria.endpoint.UUIDParam;
18  
19  @Target({ElementType.FIELD, ElementType.PARAMETER})
20  @Retention(RetentionPolicy.RUNTIME)
21  @Constraint(validatedBy = MatchesPathId.Validator.class)
22  public @interface MatchesPathId {
23    String message() default "{nl.knaw.huygens.alexandria.endpoint.resource.MatchesPathId.message}";
24  
25    Class<?>[] groups() default {};
26  
27    Class<? extends Payload>[] payload() default {};
28  
29    class Validator implements ConstraintValidator<MatchesPathId, ResourcePrototype> {
30      final UriInfo uriInfo;
31  
32      @Inject
33      public Validator(UriInfo uriInfo) {
34        this.uriInfo = uriInfo;
35      }
36  
37      @Override
38      public void initialize(MatchesPathId constraintAnnotation) {
39        Log.info("initialize: annotation={}", constraintAnnotation);
40      }
41  
42      @Override
43      public boolean isValid(ResourcePrototype prototype, ConstraintValidatorContext context) {
44        if (prototype == null) {
45          return true; // Non-null validation, if desired, should be enforced via @NotNull
46        }
47  
48        final UUIDParam prototypeId = prototype.getId();
49  
50        // Either no id was passed in prototype, or it MUST equal the id in URI
51        return prototypeId == null || value(uriId()).equals(value(prototypeId));
52      }
53  
54      private String value(PathSegment segment) {
55        return segment.getPath();
56      }
57  
58      private String value(UUIDParam uuid) {
59        return String.valueOf(uuid.getValue());
60      }
61  
62      private PathSegment uriId() {
63        return uuidSegment(uriSegments());
64      }
65  
66      private PathSegment uuidSegment(List<PathSegment> pathSegments) {
67        final int uuidSegmentIndex = uriSegments().size() - 1;
68        return pathSegments.get(uuidSegmentIndex);
69      }
70  
71      private List<PathSegment> uriSegments() {
72        return uriInfo.getPathSegments();
73      }
74    }
75  
76  }
