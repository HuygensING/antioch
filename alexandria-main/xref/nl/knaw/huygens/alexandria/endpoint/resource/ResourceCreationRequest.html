16  class ResourceCreationRequest implements CreationRequest<AlexandriaResource> {
17  
18    private final ResourcePrototype prototype;
19    private boolean resourceCreated;
20  
21    private UUID uuid;
22  
23    ResourceCreationRequest(ResourcePrototype prototype) {
24      this.prototype = prototype;
25    }
26

36    @Override
37    public AlexandriaResource execute(AlexandriaService service) {
38      Log.trace("executing, service=[{}]", service);
39  
40      uuid = providedUUID().orElse(UUID.randomUUID());
41  
42      String ref = providedRef();
43      TentativeAlexandriaProvenance provenance = providedProvenance().orElse(TentativeAlexandriaProvenance.createDefault());
44      resourceCreated = service.createOrUpdateResource(uuid, ref, provenance, prototype.getState());
45      // Log.trace("resource: [{}]", resource);
46  
47      return service.readResource(uuid).get();
48    }
49  
50    public boolean wasExecutedAsIs() {
51      final boolean wasExecutedAsIs = providedUUID().isPresent();// && providedProvenance().isPresent();
52      Log.trace("wasExecutedAsIs: {}", wasExecutedAsIs);
53      return wasExecutedAsIs;
54    }
55  
56    public boolean newResourceWasCreated() {
57      Log.trace("newResourceWasCreated: {}", resourceCreated);
58      return resourceCreated;
59    }
60  
61    public UUID getUUID() {
62      return uuid;
63    }
64  
65    private String providedRef() {
66      return requireNonNull(prototype.getRef(), "Required 'ref' field was not validated for being non-null");
67    }
68  
69    private Optional<UUID> providedUUID() {
70      return Optional.ofNullable(prototype.getId()).map(UUIDParam::getValue);
71    }
72  
73    private Optional<TentativeAlexandriaProvenance> providedProvenance() {
74      return prototype.getProvenance().map(ProvenancePrototype::getValue);
75    }
76  }
