28  public class ResourceAnnotationsEndpoint extends AnnotatableObjectAnnotationsEndpoint {
29  
30    // TODO: how to remove this duplicated inject/constructor?
31    @Inject
32    public ResourceAnnotationsEndpoint(AlexandriaService service, //
33        AnnotationCreationRequestBuilder requestBuilder, //
34        LocationBuilder locationBuilder, //
35        @PathParam("uuid") final UUIDParam uuidParam) {
36      super(service, requestBuilder, locationBuilder, uuidParam);
37    }
38  
39    @Override
40    protected AbstractAnnotatable getAnnotatableObject() {
41      AlexandriaResource resource = service.readResource(uuid)//
42          .orElseThrow(ResourcesEndpoint.resourceNotFoundForId(uuid));
43      if (resource.isTentative()) {
44        throw ResourcesEndpoint.resourceIsTentativeException(uuid);
45      }
46      return resource;
47    }
48  
49    @Override
50    protected AnnotationCreationRequestBuilder getAnnotationCreationRequestBuilder() {
51      return requestBuilder.ofResource(uuid);
52    }
53  
54    @GET
55    @ApiOperation(value = "get the annotations of the resource and its immediate subresources", response = AnnotationEntity.class)
56    @Override
57    public Response get() {
58      Stream<AlexandriaAnnotation> resourceAnnotationsStream = getAnnotatableObject().getAnnotations().stream();
59  
60      Stream<AlexandriaAnnotation> subresourceAnnotationsStream = ((AlexandriaResource) getAnnotatableObject())//
61          .getSubResourcePointers().stream()//
62          .map(service::dereference)//
63          .flatMap(Optionals::stream)//
64          .flatMap(a -> ((AlexandriaResource) a).getAnnotations().stream());
65  
66      Stream<AlexandriaAnnotation> joinedStream = Stream.concat(resourceAnnotationsStream, subresourceAnnotationsStream);
67  
68      final Set<AnnotationEntity> annotationEntities = joinedStream//
69          .map((AlexandriaAnnotation a) -> AnnotationEntity.of(a).withLocationBuilder(locationBuilder))//
70          .collect(toSet());
71  
72      Map<String, Set<AnnotationEntity>> entity = ImmutableMap.of("annotations", annotationEntities);
73      return Response.ok(entity).build();
74    }
