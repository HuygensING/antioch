58    @Override
59    protected AbstractAnnotatable getAnnotatableObject() {
60      AlexandriaResource resource = service.readResource(uuid)//
61          .orElseThrow(ResourceValidatorFactory.resourceNotFoundForId(uuid));
62      if (resource.isTentative()) {
63        throw ResourceValidatorFactory.resourceIsTentativeException(uuid);
64      }
65      return resource;
66    }
67
