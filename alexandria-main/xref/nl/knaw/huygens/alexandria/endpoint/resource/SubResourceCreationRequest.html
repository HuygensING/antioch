2   
3   import static java.util.Objects.requireNonNull;
4   
5   import java.util.Optional;
6   import java.util.UUID;
7   
8   import nl.knaw.huygens.Log;
9   import nl.knaw.huygens.alexandria.endpoint.CreationRequest;
10  import nl.knaw.huygens.alexandria.endpoint.ProvenancePrototype;
11  import nl.knaw.huygens.alexandria.endpoint.UUIDParam;
12  import nl.knaw.huygens.alexandria.model.AlexandriaResource;
13  import nl.knaw.huygens.alexandria.model.TentativeAlexandriaProvenance;
14  import nl.knaw.huygens.alexandria.service.AlexandriaService;
15  
16  class SubResourceCreationRequest implements CreationRequest<AlexandriaResource> {
17  
18    private boolean subResourceWasCreated;
19    private UUID parentId;
20    private SubResourcePrototype prototype;
21    private UUID uuid;
22  
23    public SubResourceCreationRequest(UUID parentId, SubResourcePrototype prototype) {
24      this.parentId = parentId;
25      this.prototype = prototype;
26    }
27  
28    @Override
29    public AlexandriaResource execute(AlexandriaService service) {
30      Log.trace("executing, service=[{}]", service);
31  
32      uuid = providedUUID().orElse(UUID.randomUUID());
33  
34      String sub = providedSub();
35      TentativeAlexandriaProvenance provenance = providedProvenance().orElse(TentativeAlexandriaProvenance.createDefault());
36      service.createSubResource(uuid, parentId, sub, provenance, prototype.getState());
37  
38      return service.readResource(uuid).get();
39    }
40  
41    public boolean wasExecutedAsIs() {
42      final boolean wasExecutedAsIs = providedUUID().isPresent();// && providedProvenance().isPresent();
43      Log.trace("wasExecutedAsIs: {}", wasExecutedAsIs);
44      return wasExecutedAsIs;
45    }
46  
47    public boolean newResourceWasCreated() {
48      Log.trace("newResourceWasCreated: {}", subResourceWasCreated);
49      return subResourceWasCreated;
50    }
51  
52    public UUID getUUID() {
53      return uuid;
54    }
55  
56    private String providedSub() {
57      return requireNonNull(prototype.getSub(), "Required 'sub' field was not validated for being non-null");
58    }
59  
60    private Optional<UUID> providedUUID() {
61      return Optional.ofNullable(prototype.getId()).map(UUIDParam::getValue);
62    }
63  
64    private Optional<TentativeAlexandriaProvenance> providedProvenance() {
65      return prototype.getProvenance().map(ProvenancePrototype::getValue);
66    }
67  }
