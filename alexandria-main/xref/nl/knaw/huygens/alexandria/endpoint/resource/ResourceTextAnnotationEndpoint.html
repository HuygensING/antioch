62  public class ResourceTextAnnotationEndpoint extends JSONEndpoint {
63  
64    private LocationBuilder locationBuilder;
65    private AlexandriaService service;
66    private AlexandriaResource resource;
67    private UUID resourceUUID;
68  
69    @Inject
70    public ResourceTextAnnotationEndpoint(AlexandriaService service, //
71        ResourceValidatorFactory validatorFactory, //
72        LocationBuilder locationBuilder, //
73        @PathParam("uuid") final UUIDParam uuidParam) {
74      this.service = service;
75      this.locationBuilder = locationBuilder;
76      this.resource = validatorFactory.validateExistingResource(uuidParam).notTentative().hasText().get();
77      this.resourceUUID = resource.getId();
78    }
79  
80    @PUT
81    @Path("{annotationUUID}")
82    @Consumes(MediaType.APPLICATION_JSON)
83    public Response setAnnotation(//
84        @PathParam("annotationUUID") final UUIDParam uuidParam, //
85        @NotNull TextRangeAnnotation textRangeAnnotation//
86    ) {
87      textRangeAnnotation.setId(uuidParam.getValue());
88      String xml = getXML();
89      String annotated = validateTextRangeAnnotation(textRangeAnnotation, xml);
90  
91      service.setTextRangeAnnotation(resourceUUID, textRangeAnnotation);
92      URI location = locationBuilder.locationOf(resource, EndpointPaths.TEXT, EndpointPaths.ANNOTATIONS, uuidParam.getValue().toString());
93      TextRangeAnnotationInfo info = new TextRangeAnnotationInfo().setAnnotates(annotated);
94      return Response.created(location).entity(info).build();
95    }
96  
97    @GET
98    @Path("{annotationUUID}")
99    public Response getAnnotation(@PathParam("annotationUUID") final UUIDParam uuidParam) {
100     UUID annotationUUID = uuidParam.getValue();
101     TextRangeAnnotation textRangeAnnotation = service.readTextRangeAnnotation(resourceUUID, annotationUUID)//
102         .orElseThrow(() -> new NotFoundException("No annotation found for this resource with id " + annotationUUID));
103     return ok(textRangeAnnotation);
104   }
105 
106   private String getXML() {
107     StreamingOutput outputStream = TextGraphUtil.streamXML(service, resourceUUID);
108     return TextGraphUtil.asString(outputStream);
109   }
110 
111   private String validateTextRangeAnnotation(TextRangeAnnotation textAnnotation, String xml) {
112     String annotated = validatePosition(textAnnotation.getPosition(), xml);
113     validateName(textAnnotation.getName());
114     validateAnnotator(textAnnotation.getAnnotator());
115     if (service.overlapsWithExisitingTextRangeAnnotationForResource(textAnnotation, resourceUUID)) {
116       throw new ConflictException("Overlapping annotations with the same name and responsibility.");
117     }
118     return annotated;
119   }
120 
121   private void validateAnnotator(String annotator) {
122     if (StringUtils.isEmpty(annotator)) {
123       throw new BadRequestException("No annotator specified.");
124     }
125     Optional<String> validAnnotator = service.readResourceAnnotators(resourceUUID).parallelStream()//
126         .map(Annotator::getCode)//
127         .filter(annotator::equals)//
128         .findAny();
129     if (!validAnnotator.isPresent()) {
130       throw new BadRequestException("Resource has no annotator with code '" + annotator + "'.");
131     }
132   }
133 
134   private String validatePosition(Position position, String xml) {
135     QueryableDocument qDocument = QueryableDocument.createFromXml(xml, true);
136     validate(qDocument, //
137         "count(//*[@xml:id='" + position.getXmlId() + "'])", //
138         1d, //
139         "The text does not contain an element with the specified xml:id."//
140     );
141     String xpath = "substring(//*[@xml:id='" + position.getXmlId() + "']," + position.getOffset() + "," + position.getLength() + ")";
142     validate(qDocument, //
143         "string-length(" + xpath + ")", //
144         new Double(position.getLength()), //
145         "The specified offset/length is illegal."//
146     );
147 
148     String annotated = "";
149     try {
150       annotated = qDocument.evaluateXPathToString(xpath);
151     } catch (XPathExpressionException e) {
152       e.printStackTrace();
153     }
154 
155     return annotated;
156   }
157 
158   private void validate(QueryableDocument qDocument, String xpath, Double expectation, String errorMessage) {
159     Log.info("xpath = '{}'", xpath);
160     try {
161       Double evaluation = qDocument.evaluateXPathToDouble(xpath);
162       Log.info("evaluation = {}", evaluation);
163       if (!evaluation.equals(expectation)) {
164         throw new BadRequestException(errorMessage);
165       }
166 
167     } catch (XPathExpressionException e) {
168       e.printStackTrace();
169       throw new BadRequestException(errorMessage);
170     }
171   }
172 
173   private void validateName(String elementName) {
174     List<String> validationErrors = XMLUtil.validateElementName(elementName);
175     if (!validationErrors.isEmpty()) {
176       throw new BadRequestException("The specified annotation name is illegal.");
177     }
178 
179   }
