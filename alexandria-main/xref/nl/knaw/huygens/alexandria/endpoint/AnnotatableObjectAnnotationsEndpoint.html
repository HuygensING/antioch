28  public abstract class AnnotatableObjectAnnotationsEndpoint extends JSONEndpoint {
29  
30    protected final AlexandriaService service;
31    protected final LocationBuilder locationBuilder;
32    protected final AnnotationCreationRequestBuilder requestBuilder;
33    protected final UUID uuid;
34  
35    protected AnnotatableObjectAnnotationsEndpoint(AlexandriaService service,                   //
36        AnnotationCreationRequestBuilder requestBuilder,                   //
37        LocationBuilder locationBuilder,                   //
38        final UUIDParam uuidParam) {
39      Log.trace("resourceService=[{}], uuidParam=[{}]", service, uuidParam);
40      this.service = service;
41      this.requestBuilder = requestBuilder;
42      this.uuid = uuidParam.getValue();
43      this.locationBuilder = locationBuilder;
44    }
45  
46    protected abstract AbstractAnnotatable getAnnotableObject();
47  
48    protected abstract AnnotationCreationRequestBuilder getAnnotationCreationRequestBuilder();
49  
50    @GET
51    @ApiOperation(value = "get the annotations", response = AnnotationEntity.class)
52    public Response get() {
53      final Set<AnnotationEntity> annotationEntities = getAnnotableObject().getAnnotations().stream()//
54          .map((AlexandriaAnnotation a) -> AnnotationEntity.of(a).withLocationBuilder(locationBuilder))//
55          .collect(toSet());
56      Map<String, Set<AnnotationEntity>> entity = ImmutableMap.of("annotations", annotationEntities);
57      return Response.ok(entity).build();
58    }
59  
60    @POST
61    @Consumes(MediaType.APPLICATION_JSON)
62    @ApiOperation("add annotation")
63    public Response addAnnotation(@NotNull @Valid AnnotationPrototype prototype) {
64      prototype.setState(AlexandriaState.TENTATIVE);
65      AnnotationCreationRequest request = getAnnotationCreationRequestBuilder().build(prototype);
66      AlexandriaAnnotation annotation = request.execute(service);
67      return Response.created(locationBuilder.locationOf(annotation)).build();
68    }
