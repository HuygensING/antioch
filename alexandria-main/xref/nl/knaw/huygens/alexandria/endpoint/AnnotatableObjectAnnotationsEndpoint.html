55  public abstract class AnnotatableObjectAnnotationsEndpoint extends JSONEndpoint {
56  
57    protected final AlexandriaService service;
58    protected final LocationBuilder locationBuilder;
59    protected final AnnotationCreationRequestBuilder requestBuilder;
60    protected final UUID uuid;
61  
62    protected AnnotatableObjectAnnotationsEndpoint(AlexandriaService service, //
63        AnnotationCreationRequestBuilder requestBuilder, //
64        LocationBuilder locationBuilder, //
65        final UUIDParam uuidParam) {
66      // Log.trace("resourceService=[{}], uuidParam=[{}]", service, uuidParam);
67      this.service = service;
68      this.requestBuilder = requestBuilder;
69      this.uuid = uuidParam.getValue();
70      this.locationBuilder = locationBuilder;
71  
72      // throw exception if annotatableObject doesn't exist or is tentative
73      @SuppressWarnings("unused")
74      AbstractAnnotatable annotatableObject = getAnnotatableObject();
75    }
76  
77    protected abstract AbstractAnnotatable getAnnotatableObject();
78  
79    protected abstract AnnotationCreationRequestBuilder getAnnotationCreationRequestBuilder();
80  
81    @GET
82    @ApiOperation(value = "get the (non-deleted, non-deprecated) annotations", response = AnnotationEntity.class)
83    public Response get() {
84      final List<AnnotationEntity> annotationEntities = getAnnotatableObject().getAnnotations().stream()//
85          .filter(AlexandriaAnnotation::isActive)//
86          .sorted()//
87          .map((AlexandriaAnnotation a) -> AnnotationEntity.of(a).withLocationBuilder(locationBuilder))//
88          .collect(toList());
89      return ok(jsonWrap(annotationEntities));
90    }
91  
92    private Map<String, List<Map<String, AnnotationEntity>>> jsonWrap(final List<AnnotationEntity> annotationEntities) {
93      return ImmutableMap.of(//
94          JsonTypeNames.ANNOTATIONLIST,
95          annotationEntities.stream()//
96              .map(this::toAnnotationMap)//
97              .collect(toList())//
98      );
99    }
100 
101   private Map<String, AnnotationEntity> toAnnotationMap(AnnotationEntity e) {
102     return ImmutableMap.of(JsonTypeNames.ANNOTATION, e);
103   }
104 
105   @POST
106   @Consumes(MediaType.APPLICATION_JSON)
107   @ApiOperation("add annotation")
108   public Response addAnnotation(@NotNull @Valid AnnotationPrototype prototype) {
109     validateLocator(prototype);
110     prototype.setState(AlexandriaState.TENTATIVE);
111     AnnotationCreationRequest request = getAnnotationCreationRequestBuilder().build(prototype);
112     AlexandriaAnnotation annotation = request.execute(service);
113     return created(locationBuilder.locationOf(annotation));
114   }
115 
116   private void validateLocator(AnnotationPrototype prototype) {
117     if (prototype.getLocator().isPresent()) {
118       if (!(getAnnotatableObject() instanceof AlexandriaResource)) {
119         throw new BadRequestException("locators are only allowed on resource annotations");
120       }
121 
122       AlexandriaResource resource = (AlexandriaResource) getAnnotatableObject();
123       if (!resource.hasText()) {
124         throw new BadRequestException("The resource has no attached text to use the locator on.");
125       }
126 
127       String locatorString = prototype.getLocator().get();
128       try {
129         TextLocatorFactory textLocatorFactory = new TextLocatorFactory(service);
130         AlexandriaTextLocator locator = textLocatorFactory.fromString(locatorString);
131         textLocatorFactory.validate(locator, resource);
132       } catch (TextLocatorParseException e) {
133         throw new BadRequestException(e.getMessage());
134       }
135     }
136 
137   }
