25  import static java.util.stream.Collectors.joining;
26  import static java.util.stream.Collectors.toList;
27  
28  import java.util.ArrayList;
29  import java.util.List;
30  import java.util.UUID;
31  
32  import org.antlr.v4.runtime.ANTLRInputStream;
33  import org.antlr.v4.runtime.CharStream;
34  import org.antlr.v4.runtime.CommonTokenStream;
35  import org.antlr.v4.runtime.tree.ParseTree;
36  
37  import com.google.common.base.Joiner;
38  
39  import nl.knaw.huygens.alexandria.antlr.AQL2Lexer;
40  import nl.knaw.huygens.alexandria.antlr.AQL2Parser;
41  import nl.knaw.huygens.alexandria.antlr.QueryErrorListener;
42  import nl.knaw.huygens.alexandria.api.model.CommandStatus;
43  
44  public class AQL2Task implements Runnable {
45    private CommandStatus status;
46    private UUID uuid;
47    private String aql2Command;
48  
49    public AQL2Task(String aql2Command) {
50      this.aql2Command = aql2Command;
51      this.uuid = UUID.randomUUID();
52      this.status = new CommandStatus();
53    }
54  
55    public CommandStatus getStatus() {
56      return status;
57    }
58  
59    @Override
60    public void run() {
61      status.setStarted();
62      try {
63        Object result = process();
64        status.setResult(result);
65        status.setSuccess(true);
66  
67      } catch (Exception e) {
68        e.printStackTrace();
69        status.setSuccess(false);
70        status.setErrorMessage(e.getMessage());
71      }
72      status.setDone();
73    }
74  
75    private Object process() {
76      QueryVisitor visitor = new QueryVisitor();
77  
78      List<String> parseErrors = parse(visitor);
79  
80      if (!parseErrors.isEmpty()) {
81        throw new RuntimeException("parse error(s): " + Joiner.on("\n").join(parseErrors));
82      }
83  
84      String function = visitor.getFunction();
85      List<Object> parameters = visitor.getParameters();
86  
87      Object result;
88      switch (function) {
89      case "hello":
90        result = parameters.stream().map(this::hello).collect(joining("\n"));
91        break;
92  
93      case "bye":
94        result = parameters.stream().map(this::bye).collect(joining("\n"));
95        break;
96  
97      default:
98        throw new RuntimeException("unknown command: " + aql2Command);
99      }
100     return result;
101   }
102 
103   private List<String> parse(QueryVisitor visitor) {
104     List<String> parseErrors = new ArrayList<>();
105     QueryErrorListener errorListener = new QueryErrorListener();
106     CharStream stream = new ANTLRInputStream(aql2Command);
107     AQL2Lexer lex = new AQL2Lexer(stream);
108     lex.removeErrorListeners();
109     CommonTokenStream tokenStream = new CommonTokenStream(lex);
110     AQL2Parser parser = new AQL2Parser(tokenStream);
111     parser.removeErrorListeners();
112     parser.addErrorListener(errorListener);
113     parser.setBuildParseTree(true);
114     ParseTree tree = parser.root();
115     // Log.info("tree={}", tree.toStringTree(parser));
116     if (errorListener.heardErrors()) {
117       parseErrors.addAll(errorListener.getParseErrors().stream()//
118           // .map(AlexandriaQueryParser::clarifyParseError)//
119           .collect(toList()));
120       // result = "error";
121     }
122 
123     visitor.visit(tree);
124     parseErrors.addAll(errorListener.getParseErrors());
125     return parseErrors;
126   }
127 
128   public UUID getUUID() {
129     return uuid;
130   }
131 
132   public String hello(Object parameter) {
133     return "Hello and welcome, " + parameter + "!";
134   }
135 
136   public String bye(Object parameter) {
137     return "Goodbye " + parameter + "!";
138   }

140 }
