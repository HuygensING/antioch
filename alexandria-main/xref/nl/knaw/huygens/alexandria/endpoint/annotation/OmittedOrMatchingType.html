21  @Target({ElementType.FIELD, ElementType.PARAMETER})
22  @Retention(RetentionPolicy.RUNTIME)
23  @Constraint(validatedBy = OmittedOrMatchingType.Validator.class)
24  public @interface OmittedOrMatchingType {
25    String message() default "{nl.knaw.huygens.alexandria.endpoint.annotation.OmittedOrMatchingType.message}";
26  
27    Class<?>[] groups() default {};
28  
29    Class<? extends Payload>[] payload() default {};
30  
31    class Validator implements ConstraintValidator<OmittedOrMatchingType, AnnotationPrototype> {
32      private static final boolean OMITTED_TYPE_CONSIDERED_VALID = true;
33  
34      private final String currentType;
35  
36      @Inject
37      public Validator(AlexandriaService service, @PathParam("uuid") UUIDParam uuidParam) {
38        currentType = service.readAnnotation(uuidParam.getValue()) //
39            .map(AlexandriaAnnotation::getBody) //
40            .map(AlexandriaAnnotationBody::getType) //
41            .orElseThrow(annotationNotFoundForId(uuidParam));
42      }
43  
44      @Override
45      public void initialize(OmittedOrMatchingType constraintAnnotation) {
46      }
47  
48      @Override
49      public boolean isValid(AnnotationPrototype prototype, ConstraintValidatorContext context) {
50        if (prototype == null) {
51          return true; // Non-null validation, if desired, should be enforced via @NotNull
52        }
53  
54        return prototype.getType().map(currentType::equals).orElse(OMITTED_TYPE_CONSIDERED_VALID);
55      }
56    }
57  }
