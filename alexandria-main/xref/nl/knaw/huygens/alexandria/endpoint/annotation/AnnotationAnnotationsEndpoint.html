15  public class AnnotationAnnotationsEndpoint extends AnnotatableObjectAnnotationsEndpoint {
16  
17    // TODO: how to remove this duplicated inject/constructor?
18    @Inject
19    public AnnotationAnnotationsEndpoint(AlexandriaService service, //
20        AnnotationCreationRequestBuilder requestBuilder, //
21        LocationBuilder locationBuilder, //
22        @PathParam("uuid") final UUIDParam uuidParam) {
23      super(service, requestBuilder, locationBuilder, uuidParam);
24      Log.trace("resourceService=[{}], uuidParam=[{}]", service, uuidParam);
25    }
26  
27    @Override
28    protected AbstractAnnotatable getAnnotatableObject() {
29      AlexandriaAnnotation annotation = service.readAnnotation(uuid)//
30          .orElseThrow(AnnotationsEndpoint.annotationNotFoundForId(uuid));
31      if (annotation.isTentative()){
32        throw AnnotationsEndpoint.annotationIsTentative(uuid);
33      }
34      return annotation;
35    }
36  
37    @Override
38    protected AnnotationCreationRequestBuilder getAnnotationCreationRequestBuilder() {
39      return requestBuilder.ofAnnotation(uuid);
40    }

42  }
