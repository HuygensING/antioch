3   import java.net.URI;
4   
5   import javax.ws.rs.Produces;
6   import javax.ws.rs.core.MediaType;
7   import javax.ws.rs.core.Response;
8   import javax.ws.rs.core.Response.Status;
9   
10  //rfc4627: JSON text SHALL be encoded in Unicode. The default encoding is UTF-8.
11  @Produces(MediaType.APPLICATION_JSON)
12  public abstract class JSONEndpoint {
13    protected Response created(URI location) {
14      return Response.created(location).build();
15    }
16  
17    protected Response methodNotImplemented() {
18      return Response.status(Status.NOT_IMPLEMENTED).build();
19    }
20  
21    protected Response noContent() {
22      return Response.noContent().build();
23    }
24  
25    protected Response ok() {
26      return Response.ok().build();
27    }
28  
29    protected Response ok(Object entity) {
30      return Response.ok(entity).build();
31    }
32  }
