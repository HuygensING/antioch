25  import static java.util.stream.Collectors.toList;
26  import static java.util.stream.Collectors.toSet;
27  
28  import java.util.Arrays;
29  import java.util.List;
30  import java.util.Optional;
31  import java.util.Set;
32  import java.util.stream.Stream;
33  
34  // Thanks to Christoffer Sawicki (qerub) for https://gist.github.com/qerub/9299fce67ab74e114633
35  public class Optionals {
36    private Optionals() {
37    }
38  
39    public static void main(String[] args) {
40      Optional<String> opt1 = Optional.of("1");
41      Optional<String> opt2 = Optional.of("2");
42      Optional<String> opt3 = Optional.empty();
43  
44      // Transform an optional into something else via a stream:
45      Set<String> res1 = stream(opt1).collect(toSet());
46  
47      // Flatten a collection of optionals:
48      List<String> res2 = Stream.of(opt1, opt2, opt3).flatMap(Optionals::stream).collect(toList());
49  
50      // Combining multiple optionals, keeping the first present value:
51      Optional<String> res3 = or(opt1, opt2, opt3);
52    }
53  
54    /*
55     * This will be included in Java 9, but until then...
56     * (https://bugs.openjdk.java.net/browse/JDK-8050820)
57     */
58    public static <T> Stream<T> stream(Optional<T> self) {
59      return self.map(Stream::of).orElseGet(Stream::empty);
60    }
61  
62    @SafeVarargs
63    public static <T> Optional<T> or(Optional<T>... xs) {
64      return Arrays.stream(xs).flatMap(Optionals::stream).findFirst();
65    }
66  }
