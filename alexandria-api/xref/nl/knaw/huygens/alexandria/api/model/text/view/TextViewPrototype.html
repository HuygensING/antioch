25  import java.util.ArrayList;
26  import java.util.List;
27  
28  import com.fasterxml.jackson.annotation.JsonIgnore;
29  import com.fasterxml.jackson.annotation.JsonTypeName;
30  
31  import nl.knaw.huygens.alexandria.api.JsonTypeNames;
32  import nl.knaw.huygens.alexandria.api.model.JsonWrapperObject;
33  import nl.knaw.huygens.alexandria.api.model.Prototype;
34  
35  @JsonTypeName(JsonTypeNames.TEXTVIEW)
36  public class TextViewPrototype extends JsonWrapperObject implements Prototype {
37    private String description = "";
38    private List<ElementDefinition> includedElements = new ArrayList<>();
39    private List<String> excludedElementTags = new ArrayList<>();
40    private List<String> ignoredElements = new ArrayList<>();
41  
42    public void setDescription(String description) {
43      this.description = description;
44    }
45  
46    public String getDescription() {
47      return description;
48    }
49  
50    public TextViewPrototype setIncludedElements(List<ElementDefinition> includedElements) {
51      this.includedElements = includedElements;
52      return this;
53    }
54  
55    public List<ElementDefinition> getIncludedElements() {
56      return includedElements;
57    }
58  
59    public void setExcludedElements(List<String> excludedElementTags) {
60      this.excludedElementTags = excludedElementTags;
61    }
62  
63    public List<String> getExcludedElementTags() {
64      return excludedElementTags;
65    }
66  
67    public TextViewPrototype setIgnoredElements(List<String> ignoredElements) {
68      this.ignoredElements = ignoredElements;
69      return this;
70    }
71  
72    public List<String> getIgnoredElements() {
73      return ignoredElements;
74    }
75  
76    @JsonIgnore
77    public boolean isValid() {
78      // defining both includedElements and excludedElements makes no sense
79      return includedElements.isEmpty() || excludedElementTags.isEmpty();
80    }
81  }
