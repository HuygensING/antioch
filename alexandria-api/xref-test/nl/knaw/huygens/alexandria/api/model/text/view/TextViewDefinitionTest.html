51    @Test
52    public void testJsonSerializing() throws IOException {
53      TextViewDefinition d = new TextViewDefinition();
54      d.setDescription("Test definition");
55      d.getElementViewDefinitions().put(TextViewDefinition.DEFAULT_ATTRIBUTENAME, new ElementViewDefinition().setElementMode(ElementView.ElementMode.show).setAttributeMode("showAll"));
56      d.getElementViewDefinitions().put("note", new ElementViewDefinition().setElementMode(ElementView.ElementMode.hide));
57      d.getElementViewDefinitions().put("persName", new ElementViewDefinition().setElementMode(ElementView.ElementMode.show).setWhen("attribute(resp).is(\"#ed\")"));
58  
59      TextViewDefinitionParser tvdp = new TextViewDefinitionParser(d);
60      assertThat(tvdp.isValid()).isTrue();
61  
62      String json = om.writeValueAsString(d);
63      String expected = singleQuotesToDouble("{'textView':"//
64          + "{"//
65          + "'description':'Test definition',"//
66          + "'annotationLayers':{}," + "'annotationLayerDepthOrder':[],"
67          + "'elements':{"//
68          + "':default':{'elementMode':'show','attributeMode':'showAll'},"//
69          + "'note':{'elementMode':'hide'},"//
70          + "'persName':{'elementMode':'show','when':'attribute(resp).is(\\'#ed\\')'}"//
71          + "}"//
72          + "}}");
73  
74      TextViewDefinition d2 = om.readValue(json, TextViewDefinition.class);
75      assertThat(json).isEqualTo(expected);
76      assertThat(d2).isEqualToComparingFieldByFieldRecursively(d);
77    }

79    // http://www.w3schools.com/xml/xml_elements.asp :
80    // XML elements must follow these naming rules:
81    //
82    // Element names are case-sensitive
83    // Element names must start with a letter or underscore
84    // Element names cannot start with the letters xml (or XML, or Xml, etc)
85    // Element names can contain letters, digits, hyphens, underscores, and periods
86    // Element names cannot contain spaces
87    @Test
88    public void testValidationOfInvalidElementNamesFails() throws IOException {
89      TextViewDefinition d = new TextViewDefinition();
90      d.getElementViewDefinitions().put("3lement", new ElementViewDefinition().setElementMode(ElementView.ElementMode.show));
91      d.getElementViewDefinitions().put("xmlIsDaBomb", new ElementViewDefinition().setElementMode(ElementView.ElementMode.show));
92      d.getElementViewDefinitions().put("element?", new ElementViewDefinition().setElementMode(ElementView.ElementMode.show));
93      d.getElementViewDefinitions().put("e le ment", new ElementViewDefinition().setElementMode(ElementView.ElementMode.show));
94      TextViewDefinitionParser tvdp = new TextViewDefinitionParser(d);
95      assertThat(tvdp.isValid()).isFalse();
96      assertThat(tvdp.getErrors()).containsExactly(//
97          "\"3lement\" is not a valid element name: element names must start with a letter or underscore.", //
98          "\"xmlIsDaBomb\" is not a valid element name: element names cannot start with the letters xml (or XML, or Xml, etc).", //
99          "\"element?\" is not a valid element name: element names can only contain letters, digits, hyphens, underscores, and periods.", //
100         "\"e le ment\" is not a valid element name: element names cannot contain spaces."//
101     );
102   }

133   public void testParsingOfDefinitionWithJustAttributeModePasses() throws IOException {
134     TextViewDefinition tvd = new TextViewDefinition();
135     ElementViewDefinition evd = new ElementViewDefinition()//
136         .setAttributeMode("showOnly xml:id ref");
137     tvd.getElementViewDefinitions().put("p", evd);
138 
139     TextViewDefinitionParser tvdp = new TextViewDefinitionParser(tvd);

197   @Test
198   public void testParsingOfIsWhen() throws IOException {
199     TextViewDefinition tvd = new TextViewDefinition();
200     ElementViewDefinition evd = new ElementViewDefinition()//
201         .setElementMode(ElementView.ElementMode.show)//
202         .setWhen("attribute({resp}).is('#ed1')");
203     tvd.getElementViewDefinitions().put("element", evd);
204 
205     TextViewDefinitionParser tvdp = new TextViewDefinitionParser(tvd);
206     TextView textView = tvdp.getTextView().get();
207     Map<String, ElementView> elementViewMap = textView.getElementViewMap();
208 
209     assertThat(tvdp.isValid()).isTrue();
210     assertThat(elementViewMap).containsKey("element");
211 
212     ElementView elementView = elementViewMap.get("element");
213     assertThat(elementView.getElementMode()).isNotNull();
214     assertThat(elementView.getElementMode()).isEqualTo(ElementView.ElementMode.show);
215     assertThat(elementView.getAttributeMode()).isEqualTo(AttributeMode.showAll); // default
216     assertThat(elementView.getRelevantAttributes()).isEmpty();
217 
218     AttributePreCondition preCondition = elementView.getPreCondition().get();
219     assertThat(preCondition.getAttribute()).isEqualTo("{resp}");
220     assertThat(preCondition.getFunction()).isEqualTo(AttributeFunction.is);
221     assertThat(preCondition.getValues()).containsExactly("#ed1");
222     testJsonSerialization(textView);
223   }
